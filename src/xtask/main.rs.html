1use std::{
2  fs,
3  path::{Path, PathBuf},
4};
5
6use anyhow::{Context, Result};
7use clap::{CommandFactory, Parser, Subcommand};
8use clap_complete::{generate_to, shells};
9use clap_mangen::Man;
10
11#[derive(Parser)]
12#[command(author, version, about)]
13struct Xtask {
14  #[command(subcommand)]
15  command: Commands,
16}
17
18#[derive(Subcommand)]
19enum Commands {
20  /// Build distribution artifacts for the ndg CLI
21  Dist {
22    /// Output directory for generated files.
23    #[arg(short, long, default_value = "dist", value_parser = clap::value_parser!(std::path::PathBuf))]
24    output_dir: PathBuf,
25
26    /// Only generate shell completions.
27    #[arg(long, conflicts_with = "manpage_only")]
28    completions_only: bool,
29
30    /// Only generate manpage.
31    #[arg(long, conflicts_with = "completions_only")]
32    manpage_only: bool,
33  },
34}
35
36fn main() -> Result<()> {
37  let xtask = Xtask::parse();
38
39  match xtask.command {
40    Commands::Dist {
41      output_dir,
42      completions_only,
43      manpage_only,
44    } => {
45      if completions_only {
46        generate_completions(&output_dir)?;
47      } else if manpage_only {
48        generate_manpage(&output_dir)?;
49      } else {
50        generate_completions(&output_dir)?;
51        generate_manpage(&output_dir)?;
52      }
53    },
54  }
55
56  Ok(())
57}
58
59/// Generate shell completions for various shells.
60fn generate_completions(output_dir: &Path) -> Result<()> {
61  let completions_dir = output_dir.join("completions");
62  fs::create_dir_all(&completions_dir)?;
63  let mut cmd = ndg::cli::Cli::command();
64  generate_to(shells::Bash, &mut cmd, "ndg", &completions_dir)?;
65  generate_to(shells::Zsh, &mut cmd, "ndg", &completions_dir)?;
66  generate_to(shells::Fish, &mut cmd, "ndg", &completions_dir)?;
67  generate_to(shells::PowerShell, &mut cmd, "ndg", &completions_dir)?;
68  println!(
69    "Shell completions generated in {}",
70    completions_dir.display()
71  );
72  Ok(())
73}
74
75/// Generate manpage for the NDG CLI.
76fn generate_manpage(output_dir: &Path) -> Result<()> {
77  let man_dir = output_dir.join("man");
78  fs::create_dir_all(&man_dir)?;
79  let cmd = ndg::cli::Cli::command();
80  let man = Man::new(cmd);
81  let file_path = man_dir.join("ndg.1");
82  let mut file = fs::File::create(&file_path).with_context(|| {
83    format!("Failed to create manpage file at {}", file_path.display())
84  })?;
85  man
86    .render(&mut file)
87    .with_context(|| "Failed to render manpage")?;
88  println!("Manpage generated in {}", man_dir.display());
89  Ok(())
90}

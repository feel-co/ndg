1//! # NDG-Commonmark
2//!
3//! This is a high-performance Markdown processor designed for Nix, `NixOS` and
4//! Nixpkgs documentation featuring AST-based processing with role markup,
5//! autolinks, GFM support and more.
6//!
7//! ## Processing
8//!
9//! ```rust
10//! use ndg_commonmark::{ProcessorPreset, process_markdown_string};
11//!
12//! let result = process_markdown_string(
13//!   "# Hello World\n\nThis is **bold** text.",
14//!   ProcessorPreset::Basic,
15//! );
16//! println!("HTML: {}", result.html);
17//! println!("Title: {:?}", result.title);
18//! ```
19//!
20//! ## API
21//!
22//! ```rust
23//! use ndg_commonmark::{MarkdownOptions, MarkdownProcessor};
24//!
25//! let processor = MarkdownProcessor::new(MarkdownOptions::default());
26//! let result = processor.render("# Hello World\n\nThis is **bold** text.");
27//!
28//! println!("HTML: {}", result.html);
29//! println!("Title: {:?}", result.title);
30//! println!("Headers: {:?}", result.headers);
31//! ```
32//!
33//! ## Builder Pattern
34//!
35//! ```rust
36//! use ndg_commonmark::{MarkdownOptionsBuilder, MarkdownProcessor};
37//!
38//! let options = MarkdownOptionsBuilder::new()
39//!   .gfm(true)
40//!   .nixpkgs(true)
41//!   .highlight_code(true)
42//!   .highlight_theme(Some("github"))
43//!   .build();
44//!
45//! let processor = MarkdownProcessor::new(options);
46//! ```

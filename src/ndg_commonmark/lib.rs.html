1//! # NDG-Commonmark
2//!
3//! This is a high-performance Markdown processor designed for Nix, `NixOS` and Nixpkgs documentation
4//! featuring AST-based processing with role markup, autolinks, GFM support and more.
5//!
6//! ## Processing
7//!
8//! ```rust
9//! use ndg_commonmark::{process_markdown_string, ProcessorPreset};
10//!
11//! let result = process_markdown_string("# Hello World\n\nThis is **bold** text.", ProcessorPreset::Basic);
12//! println!("HTML: {}", result.html);
13//! println!("Title: {:?}", result.title);
14//! ```
15//!
16//! ## API
17//!
18//! ```rust
19//! use ndg_commonmark::{MarkdownProcessor, MarkdownOptions};
20//!
21//! let processor = MarkdownProcessor::new(MarkdownOptions::default());
22//! let result = processor.render("# Hello World\n\nThis is **bold** text.");
23//!
24//! println!("HTML: {}", result.html);
25//! println!("Title: {:?}", result.title);
26//! println!("Headers: {:?}", result.headers);
27//! ```
28//!
29//! ## Builder Pattern
30//!
31//! ```rust
32//! use ndg_commonmark::{MarkdownProcessor, MarkdownOptionsBuilder};
33//!
34//! let options = MarkdownOptionsBuilder::new()
35//!     .gfm(true)
36//!     .nixpkgs(true)
37//!     .highlight_code(true)
38//!     .highlight_theme(Some("github"))
39//!     .build();
40//!
41//! let processor = MarkdownProcessor::new(options);
42//! ```

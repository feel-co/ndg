38    #[test]
39    fn test_html_escaped_roles() {
40        // Test that HTML characters in role content are properly escaped
41        #[cfg(any(feature = "nixpkgs", feature = "ndg-flavored"))]
42        {
43            let result =
44                super::extensions::format_role_markup("option", "hjem.users.<name>.enable", None);
45
46            // Should escape < and > characters in content
47            assert!(result.contains("&lt;name&gt;"));
48            // Should not contain unescaped HTML in code content
49            assert!(!result.contains("<code>hjem.users.<name>.enable</code>"));
50            // Should contain escaped content in code
51            assert!(result.contains("<code>hjem.users.&lt;name&gt;.enable</code>"));
52            // Should have properly formatted option ID in href with preserved special chars
53            assert!(result.contains("option-hjem-users-<name>-enable"));
54        }
55    }

57    #[test]
58    fn test_html_escape_util() {

61
62        // html-escape crate doesn't escape single quotes by default
63        assert_eq!(escaped, "test&lt;&gt;&amp;\"'");

66    #[test]
67    fn test_various_role_types_with_html_characters() {
68        #[cfg(any(feature = "nixpkgs", feature = "ndg-flavored"))]
69        {
70            let content = "<script>alert('xss')</script>";
71
72            let command_result = super::extensions::format_role_markup("command", content, None);
73            assert!(command_result.contains("&lt;script&gt;"));
74            assert!(!command_result.contains("<script>alert"));
75
76            let env_result = super::extensions::format_role_markup("env", content, None);
77            assert!(env_result.contains("&lt;script&gt;"));
78            assert!(!env_result.contains("<script>alert"));
79
80            let file_result = super::extensions::format_role_markup("file", content, None);
81            assert!(file_result.contains("&lt;script&gt;"));
82            assert!(!file_result.contains("<script>alert"));
83        }
84    }

86    #[test]
87    fn test_option_role_escaping() {
88        // Test the specific reported issue: {option}`hjem.users.<name>.enable`
89        #[cfg(any(feature = "nixpkgs", feature = "ndg-flavored"))]
90        {
91            let result =
92                super::extensions::format_role_markup("option", "hjem.users.<name>.enable", None);
93
94            // Should not produce broken HTML like: <code>hjem.users.<name>.enable</name></code>
95            assert!(!result.contains("</name>"));
96
97            // Should properly escape the angle brackets in display text
98            assert!(result.contains("&lt;name&gt;"));
99
100            // Should produce valid HTML structure
101            assert!(result.contains("<code>hjem.users.&lt;name&gt;.enable</code>"));
102
103            // Should preserve special characters in the option ID (the actual anchor)
104            assert!(result.contains("options.html#option-hjem-users-<name>-enable"));
105        }
106    }

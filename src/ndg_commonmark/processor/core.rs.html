90  /// Highlight all code blocks in HTML using the configured syntax highlighter
91  #[must_use]
92  pub fn highlight_codeblocks(&self, html: &str) -> String {
93    if !self.options.highlight_code || self.syntax_manager.is_none() {
94      return html.to_string();
95    }
96
97    use kuchikikiki::parse_html;
98    use tendril::TendrilSink;
99
100    let document = parse_html().one(html);
101
102    // Collect all code blocks first to avoid DOM modification during iteration
103    let mut code_blocks = Vec::new();
104    for pre_node in document.select("pre > code").unwrap() {
105      let code_node = pre_node.as_node();
106      if let Some(element) = code_node.as_element() {
107        let class_attr = element
108          .attributes
109          .borrow()
110          .get("class")
111          .map(std::string::ToString::to_string);
112        let language = class_attr
113          .as_deref()
114          .and_then(|s| s.strip_prefix("language-"))
115          .unwrap_or("text");
116        let code_text = code_node.text_contents();
117
118        if let Some(pre_parent) = code_node.parent() {
119          code_blocks.push((
120            pre_parent.clone(),
121            code_node.clone(),
122            code_text,
123            language.to_string(),
124          ));
125        }
126      }
127    }
128
129    // Process each code block
130    for (pre_element, _code_node, code_text, language) in code_blocks {
131      if let Some(highlighted) = self.highlight_code_html(&code_text, &language)
132      {
133        // Wrap highlighted HTML in <pre><code> with appropriate classes
134        let wrapped_html = format!(
135          r#"<pre class="highlight"><code class="language-{language}">{highlighted}</code></pre>"#
136        );
137        let fragment = parse_html().one(wrapped_html.as_str());
138        pre_element.insert_after(fragment);
139        pre_element.detach();
140      }
141      // Do not add highlight/language-* classes if not highlighted
142    }
143
144    let mut buf = Vec::new();
145    document.serialize(&mut buf).unwrap();
146    String::from_utf8(buf).unwrap_or_default()
147  }

149  /// Highlight code using the configured syntax highlighter, returns HTML
150  /// string
151  fn highlight_code_html(&self, code: &str, language: &str) -> Option<String> {
152    if !self.options.highlight_code {
153      return None;
154    }
155
156    let syntax_manager = self.syntax_manager.as_ref()?;

163  /// Render Markdown to HTML, extracting headers and title.
164  #[must_use]
165  pub fn render(&self, markdown: &str) -> MarkdownResult {
166    // Clear previous includes
167    self.included_files.borrow_mut().clear();
168
169    let preprocessed = self.preprocess(markdown);
170    let (headers, title) = self.extract_headers(&preprocessed);
171    let html = self.process_html_pipeline(&preprocessed);
172
173    MarkdownResult {
174      html,
175      headers,
176      title,
177      included_files: self.included_files.borrow().clone(),
178    }
179  }

181  /// Process the HTML generation and post-processing pipeline.
182  fn process_html_pipeline(&self, content: &str) -> String {
183    let mut html = self.convert_to_html(content);
184
185    // Apply feature-specific post-processing
186    if cfg!(feature = "ndg-flavored") {
187      #[cfg(feature = "ndg-flavored")]
188      {
189        html = super::extensions::process_option_references(&html);
190      }
191    }
192
193    if self.options.nixpkgs {
194      html = self.process_manpage_references_html(&html);
195    }
196
197    if self.options.highlight_code {
198      html = self.highlight_codeblocks(&html);
199    }
200
201    self.kuchiki_postprocess(&html)
202  }

204  /// Preprocess the markdown content with all enabled transformations.
205  fn preprocess(&self, content: &str) -> String {
206    let mut processed = content.to_string();
207
208    if self.options.nixpkgs {
209      processed = self.apply_nixpkgs_preprocessing(&processed);
210    }
211
212    if self.options.nixpkgs || cfg!(feature = "ndg-flavored") {
213      processed = super::extensions::process_role_markup(
214        &processed,
215        self.manpage_urls.as_ref(),
216      );
217    }
218
219    processed
220  }

241  /// Extract headers and title from the markdown content.
242  #[must_use]
243  pub fn extract_headers(
244    &self,
245    content: &str,
246  ) -> (Vec<Header>, Option<String>) {
247    let arena = Arena::new();
248    let options = self.comrak_options();
249
250    // Normalize custom anchors with no heading level to h2
251    let mut normalized = String::with_capacity(content.len());
252    for line in content.lines() {
253      let trimmed = line.trim_end();
254      if !trimmed.starts_with('#') {
255        if let Some(anchor_start) = trimmed.rfind("{#") {
256          if let Some(anchor_end) = trimmed[anchor_start..].find('}') {
257            let text = trimmed[..anchor_start].trim_end();
258            let id = &trimmed[anchor_start + 2..anchor_start + anchor_end];
259            normalized.push_str(&format!("## {text} {{#{id}}}\n"));
260            continue;
261          }
262        }
263      }
264      normalized.push_str(line);
265      normalized.push('\n');
266    }
267
268    let root = parse_document(&arena, &normalized, &options);
269
270    let mut headers = Vec::new();
271    let mut found_title = None;
272
273    for node in root.descendants() {
274      if let NodeValue::Heading(NodeHeading { level, .. }) =
275        &node.data.borrow().value
276      {
277        let mut text = String::new();
278        let mut explicit_id = None;
279
280        for child in node.children() {
281          match &child.data.borrow().value {
282            NodeValue::Text(t) => text.push_str(t),
283            NodeValue::Code(t) => text.push_str(&t.literal),
284            NodeValue::Link(..) => text.push_str(&extract_inline_text(child)),
285            NodeValue::Emph => text.push_str(&extract_inline_text(child)),
286            NodeValue::Strong => text.push_str(&extract_inline_text(child)),
287            NodeValue::Strikethrough => {
288              text.push_str(&extract_inline_text(child))
289            },
290            NodeValue::Superscript => {
291              text.push_str(&extract_inline_text(child))
292            },
293            NodeValue::Subscript => text.push_str(&extract_inline_text(child)),
294            NodeValue::FootnoteReference(..) => {
295              text.push_str(&extract_inline_text(child));
296            },
297            NodeValue::HtmlInline(html) => {
298              // Look for explicit anchor in HTML inline node: {#id}
299              let html_str = html.as_str();
300              if let Some(start) = html_str.find("{#") {
301                if let Some(end) = html_str[start..].find('}') {
302                  let anchor = &html_str[start + 2..start + end];
303                  explicit_id = Some(anchor.to_string());
304                }
305              }
306            },
307            NodeValue::Image(..) => {},
308            _ => {},
309          }
310        }
311
312        // Check for trailing {#id} in heading text
313        let trimmed = text.trim_end();
314        let (final_text, id) = if let Some(start) = trimmed.rfind("{#") {
315          if let Some(end) = trimmed[start..].find('}') {
316            let anchor = &trimmed[start + 2..start + end];
317            (trimmed[..start].trim_end().to_string(), anchor.to_string())
318          } else {
319            (
320              text.clone(),
321              explicit_id.unwrap_or_else(|| utils::slugify(&text)),
322            )
323          }
324        } else {
325          (
326            text.clone(),
327            explicit_id.unwrap_or_else(|| utils::slugify(&text)),
328          )
329        };
330        if *level == 1 && found_title.is_none() {
331          found_title = Some(final_text.clone());
332        }
333        headers.push(Header {
334          text: final_text,
335          level: *level,
336          id,
337        });
338      }
339    }
340
341    (headers, found_title)
342  }

344  /// Convert markdown to HTML using comrak and configured options.
345  fn convert_to_html(&self, content: &str) -> String {
346    // Process directly without panic catching for better performance
347    let arena = Arena::new();
348    let options = self.comrak_options();
349    let root = parse_document(&arena, content, &options);
350
351    // Apply AST transformations
352    let prompt_transformer = PromptTransformer;
353    prompt_transformer.transform(root);
354
355    let mut html_output = Vec::new();
356    comrak::format_html(root, &options, &mut html_output).unwrap_or_default();
357    let html = String::from_utf8(html_output).unwrap_or_default();
358
359    // Post-process HTML to handle header anchors
360    self.process_header_anchors_html(&html)
361  }

363  /// Process header anchors in HTML by finding {#id} syntax and converting to
364  /// proper id attributes
365  fn process_header_anchors_html(&self, html: &str) -> String {
366    use std::sync::LazyLock;
367
368    use regex::Regex;
369
370    static HEADER_ANCHOR_RE: LazyLock<Regex> = LazyLock::new(|| {
371      Regex::new(r"<h([1-6])>(.*?)\s*\{#([a-zA-Z0-9_-]+)\}(.*?)</h[1-6]>")
372        .unwrap_or_else(|e| {
373          log::error!("Failed to compile HEADER_ANCHOR_RE regex: {e}");
374          utils::never_matching_regex()
375        })
376    });
377
378    HEADER_ANCHOR_RE
379      .replace_all(html, |caps: &regex::Captures| {
380        let level = &caps[1];
381        let prefix = &caps[2];
382        let id = &caps[3];
383        let suffix = &caps[4];
384        format!("<h{level} id=\"{id}\">{prefix}{suffix}</h{level}>")
385      })
386      .to_string()
387  }

389  /// Build comrak options from `MarkdownOptions` and feature flags.
390  fn comrak_options(&self) -> ComrakOptions<'_> {
391    let mut options = ComrakOptions::default();
392    if self.options.gfm {
393      options.extension.table = true;
394      options.extension.footnotes = true;
395      options.extension.strikethrough = true;
396      options.extension.tasklist = true;
397      options.extension.superscript = true;
398      options.extension.autolink = true;
399    }
400    options.render.unsafe_ = true;
401    // Enable description lists but keep custom header processing
402    options.extension.header_ids = None;
403    options.extension.description_lists = true;
404    options
405  }

442  /// Process list item ID markers: <li><!-- nixos-anchor-id:ID -->
443  fn process_list_item_id_markers(&self, document: &kuchikikiki::NodeRef) {
444    let mut to_modify = Vec::new();
445
446    for comment in document.inclusive_descendants() {
447      if let Some(comment_node) = comment.as_comment() {
448        let comment_text = comment_node.borrow();
449        if let Some(id_start) = comment_text.find("nixos-anchor-id:") {
450          let id = comment_text[id_start + 16..].trim();
451          if !id.is_empty()
452            && id
453              .chars()
454              .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
455          {
456            // Check if this comment is inside an <li> element
457            if let Some(parent) = comment.parent() {
458              if let Some(element) = parent.as_element() {
459                if element.name.local.as_ref() == "li" {
460                  to_modify.push((comment.clone(), id.to_string()));
461                }
462              }
463            }
464          }
465        }
466      }
467    }
468
469    for (comment_node, id) in to_modify {
470      let span = kuchikikiki::NodeRef::new_element(
471        markup5ever::QualName::new(
472          None,
473          markup5ever::ns!(html),
474          local_name!("span"),
475        ),
476        vec![
477          (
478            kuchikikiki::ExpandedName::new("", "id"),
479            kuchikikiki::Attribute {
480              prefix: None,
481              value:  id,
482            },
483          ),
484          (
485            kuchikikiki::ExpandedName::new("", "class"),
486            kuchikikiki::Attribute {
487              prefix: None,
488              value:  "nixos-anchor".into(),
489            },
490          ),
491        ],
492      );
493      comment_node.insert_after(span);
494      comment_node.detach();
495    }
496  }

498  /// Process header anchors with comments: <h1>text<!-- anchor: id --></h1>
499  fn process_header_anchor_comments(&self, document: &kuchikikiki::NodeRef) {
500    let mut to_modify = Vec::new();
501
502    for comment in document.inclusive_descendants() {
503      if let Some(comment_node) = comment.as_comment() {
504        let comment_text = comment_node.borrow();
505        if let Some(anchor_start) = comment_text.find("anchor:") {
506          let id = comment_text[anchor_start + 7..].trim();
507          if !id.is_empty()
508            && id
509              .chars()
510              .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
511          {
512            // Check if this comment is inside a header element
513            if let Some(parent) = comment.parent() {
514              if let Some(element) = parent.as_element() {
515                let tag_name = element.name.local.as_ref();
516                if matches!(tag_name, "h1" | "h2" | "h3" | "h4" | "h5" | "h6") {
517                  to_modify.push((
518                    parent.clone(),
519                    comment.clone(),
520                    id.to_string(),
521                  ));
522                }
523              }
524            }
525          }
526        }
527      }
528    }
529
530    for (header_element, comment_node, id) in to_modify {
531      if let Some(element) = header_element.as_element() {
532        element
533          .attributes
534          .borrow_mut()
535          .insert(local_name!("id"), id);
536        comment_node.detach();
537      }
538    }
539  }

541  /// Process remaining inline anchors in list items: <li>[]{#id}content</li>
542  fn process_list_item_inline_anchors(&self, document: &kuchikikiki::NodeRef) {
543    for li_node in document.select("li").unwrap() {
544      let li_element = li_node.as_node();
545
546      // Check if this list item contains code elements
547      let has_code = li_element.select("code, pre").is_ok()
548        && li_element.select("code, pre").unwrap().next().is_some();
549      if has_code {
550        continue; // Skip list items with code blocks
551      }
552
553      let text_content = li_element.text_contents();
554
555      if let Some(anchor_start) = text_content.find("[]{#") {
556        if let Some(anchor_end) = text_content[anchor_start..].find('}') {
557          let id = &text_content[anchor_start + 4..anchor_start + anchor_end];
558          if !id.is_empty()
559            && id
560              .chars()
561              .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
562          {
563            let remaining_content =
564              &text_content[anchor_start + anchor_end + 1..];
565
566            // Clear current content and rebuild
567            for child in li_element.children() {
568              child.detach();
569            }
570
571            let span = kuchikikiki::NodeRef::new_element(
572              markup5ever::QualName::new(
573                None,
574                markup5ever::ns!(html),
575                local_name!("span"),
576              ),
577              vec![
578                (
579                  kuchikikiki::ExpandedName::new("", "id"),
580                  kuchikikiki::Attribute {
581                    prefix: None,
582                    value:  id.into(),
583                  },
584                ),
585                (
586                  kuchikikiki::ExpandedName::new("", "class"),
587                  kuchikikiki::Attribute {
588                    prefix: None,
589                    value:  "nixos-anchor".into(),
590                  },
591                ),
592              ],
593            );
594            li_element.append(span);
595            if !remaining_content.is_empty() {
596              li_element
597                .append(kuchikikiki::NodeRef::new_text(remaining_content));
598            }
599          }
600        }
601      }
602    }
603  }

669  /// Process remaining standalone inline anchors throughout the document
670  fn process_remaining_inline_anchors(&self, document: &kuchikikiki::NodeRef) {
671    let mut text_nodes_to_process = Vec::new();
672
673    for node in document.inclusive_descendants() {
674      if let Some(text_node) = node.as_text() {
675        // Check if this text node is inside a code block
676        let mut parent = node.parent();
677        let mut in_code = false;
678        while let Some(p) = parent {
679          if let Some(element) = p.as_element() {
680            if element.name.local == local_name!("code")
681              || element.name.local == local_name!("pre")
682            {
683              in_code = true;
684              break;
685            }
686          }
687          parent = p.parent();
688        }
689
690        // Only process if not in code
691        if !in_code {
692          let text_content = text_node.borrow().clone();
693          if text_content.contains("[]{#") {
694            text_nodes_to_process.push((node.clone(), text_content));
695          }
696        }
697      }
698    }
699
700    for (text_node, text_content) in text_nodes_to_process {
701      let mut last_end = 0;
702      let mut new_children = Vec::new();
703
704      // Simple pattern matching for []{#id}
705      let chars = text_content.chars().collect::<Vec<_>>();
706      let mut i = 0;
707      while i < chars.len() {
708        if i + 4 < chars.len()
709          && chars[i] == '['
710          && chars[i + 1] == ']'
711          && chars[i + 2] == '{'
712          && chars[i + 3] == '#'
713        {
714          // Found start of anchor pattern
715          let anchor_start = i;
716          i += 4; // skip "[]{#"
717
718          let mut id = String::new();
719          while i < chars.len() && chars[i] != '}' {
720            if chars[i].is_alphanumeric() || chars[i] == '-' || chars[i] == '_'
721            {
722              id.push(chars[i]);
723              i += 1;
724            } else {
725              break;
726            }
727          }
728
729          if i < chars.len() && chars[i] == '}' && !id.is_empty() {
730            // Valid anchor found
731            let anchor_end = i + 1;
732
733            // Add text before anchor
734            if anchor_start > last_end {
735              let before_text: String =
736                chars[last_end..anchor_start].iter().collect();
737              if !before_text.is_empty() {
738                new_children.push(kuchikikiki::NodeRef::new_text(before_text));
739              }
740            }
741
742            // Add span element
743            let span = kuchikikiki::NodeRef::new_element(
744              markup5ever::QualName::new(
745                None,
746                markup5ever::ns!(html),
747                local_name!("span"),
748              ),
749              vec![
750                (
751                  kuchikikiki::ExpandedName::new("", "id"),
752                  kuchikikiki::Attribute {
753                    prefix: None,
754                    value:  id,
755                  },
756                ),
757                (
758                  kuchikikiki::ExpandedName::new("", "class"),
759                  kuchikikiki::Attribute {
760                    prefix: None,
761                    value:  "nixos-anchor".into(),
762                  },
763                ),
764              ],
765            );
766            new_children.push(span);
767
768            last_end = anchor_end;
769            i = anchor_end;
770          } else {
771            i += 1;
772          }
773        } else {
774          i += 1;
775        }
776      }
777
778      // Add remaining text
779      if last_end < chars.len() {
780        let after_text: String = chars[last_end..].iter().collect();
781        if !after_text.is_empty() {
782          new_children.push(kuchikikiki::NodeRef::new_text(after_text));
783        }
784      }
785
786      // Replace text node if we found anchors
787      if !new_children.is_empty() {
788        for child in new_children {
789          text_node.insert_before(child);
790        }
791        text_node.detach();
792      }
793    }
794  }

796  /// Process empty auto-links: [](#anchor) -> <a href="#anchor">Anchor</a>
797  fn process_empty_auto_links(&self, document: &kuchikikiki::NodeRef) {
798    for link_node in document.select("a").unwrap() {
799      let link_element = link_node.as_node();
800      if let Some(element) = link_element.as_element() {
801        let href = element
802          .attributes
803          .borrow()
804          .get(local_name!("href"))
805          .map(std::string::ToString::to_string);
806        let text_content = link_element.text_contents();
807
808        if let Some(href_value) = href {
809          if href_value.starts_with('#') && text_content.trim().is_empty() {
810            // Empty link with anchor - add humanized text
811            let display_text = self.humanize_anchor_id(&href_value);
812            link_element.append(kuchikikiki::NodeRef::new_text(display_text));
813          }
814        }
815      }
816    }
817  }
818
819  /// Process empty HTML links that have no content
820  fn process_empty_html_links(&self, document: &kuchikikiki::NodeRef) {
821    for link_node in document.select("a[href^='#']").unwrap() {
822      let link_element = link_node.as_node();
823      let text_content = link_element.text_contents();
824
825      if text_content.trim().is_empty() {
826        if let Some(element) = link_element.as_element() {
827          if let Some(href) =
828            element.attributes.borrow().get(local_name!("href"))
829          {
830            let display_text = self.humanize_anchor_id(href);
831            link_element.append(kuchikikiki::NodeRef::new_text(display_text));
832          }
833        }
834      }
835    }
836  }

838  /// Convert an anchor ID to human-readable text
839  fn humanize_anchor_id(&self, anchor: &str) -> String {
840    // Strip the leading #
841    let cleaned = anchor.trim_start_matches('#');
842
843    // Remove common prefixes
844    let without_prefix = cleaned
845      .trim_start_matches("sec-")
846      .trim_start_matches("ssec-")
847      .trim_start_matches("opt-");
848
849    // Replace separators with spaces
850    let spaced = without_prefix.replace(['-', '_'], " ");
851
852    // Capitalize each word
853    spaced
854      .split_whitespace()
855      .map(|word| {
856        let mut chars = word.chars();
857        chars.next().map_or_else(String::new, |c| {
858          c.to_uppercase().collect::<String>() + chars.as_str()
859        })
860      })
861      .collect::<Vec<String>>()
862      .join(" ")
863  }

866/// Extract all inline text from a heading node.
867pub fn extract_inline_text<'a>(node: &'a AstNode<'a>) -> String {
868  let mut text = String::new();
869  for child in node.children() {
870    match &child.data.borrow().value {
871      NodeValue::Text(t) => text.push_str(t),
872      NodeValue::Code(t) => text.push_str(&t.literal),
873      NodeValue::Link(..) => text.push_str(&extract_inline_text(child)),
874      NodeValue::Emph => text.push_str(&extract_inline_text(child)),
875      NodeValue::Strong => text.push_str(&extract_inline_text(child)),
876      NodeValue::Strikethrough => text.push_str(&extract_inline_text(child)),
877      NodeValue::Superscript => text.push_str(&extract_inline_text(child)),
878      NodeValue::Subscript => text.push_str(&extract_inline_text(child)),
879      NodeValue::FootnoteReference(..) => {
880        text.push_str(&extract_inline_text(child))
881      },
882      NodeValue::HtmlInline(_) => {},
883      NodeValue::Image(..) => {},
884      _ => {},
885    }
886  }
887  text
888}

890/// Collect all markdown files from the input directory
891pub fn collect_markdown_files(input_dir: &Path) -> Vec<PathBuf> {
892  let mut files = Vec::with_capacity(100);
893
894  for entry in WalkDir::new(input_dir)
895    .follow_links(true)
896    .into_iter()
897    .filter_map(Result::ok)
898  {
899    let path = entry.path();
900    if path.is_file() && path.extension().is_some_and(|ext| ext == "md") {
901      files.push(path.to_owned());
902    }
903  }
904
905  trace!("Found {} markdown files to process", files.len());
906  files
907}

922/// Standalone HTML post-processing function to avoid borrowing issues.
923fn kuchiki_postprocess_html<F>(html: &str, transform_fn: F) -> String
924where
925  F: FnOnce(&kuchikikiki::NodeRef),
926{
927  process_safe(
928    html,
929    |html| {
930      use tendril::TendrilSink;
931
932      let document = kuchikikiki::parse_html().one(html);
933      transform_fn(&document);
934
935      let mut out = Vec::new();
936      document.serialize(&mut out).ok();
937      String::from_utf8(out).unwrap_or_default()
938    },
939    html,
940  )

18/// Represents a `NixOS` configuration option.
19///
20/// This struct holds all relevant metadata for a single `NixOS` module option,
21/// including its name, type, description, default and example values,
22/// declaration location, and visibility flags. Used for rendering options
23/// documentation.
24#[derive(Debug, Clone, Default)]
25pub struct NixOption {
26  /// Option name (e.g., "services.nginx.enable")
27  pub name: String,
28
29  /// Option type (e.g., "boolean", "string", "signed integer", etc.)
30  pub type_name: String,
31
32  /// Option description
33  pub description: String,
34
35  /// Option default value (JSON)
36  pub default: Option<Value>,
37
38  /// Option default value as text
39  pub default_text: Option<String>,
40
41  /// Option example value (JSON)
42  pub example: Option<Value>,
43
44  /// Option example value as text
45  pub example_text: Option<String>,
46
47  /// Path to the file where the option is declared
48  pub declared_in: Option<String>,
49
50  /// Option declaration URL for hyperlink
51  pub declared_in_url: Option<String>,
52
53  /// Whether this option is internal
54  pub internal: bool,
55
56  /// Whether this option is read-only
57  pub read_only: bool,
58}

60/// Process options from a JSON file and generate the options documentation
61/// page.
62///
63/// Reads the options JSON file, parses all options, sorts and formats them,
64/// and writes the rendered HTML to the output directory specified in the
65/// config.
66///
67/// # Arguments
68///
69/// * `config` - The loaded configuration for documentation generation.
70/// * `options_path` - Path to the options.json file.
71///
72/// # Errors
73///
74/// Returns an error if the file cannot be read, parsed, or written.
75#[allow(
76  clippy::cognitive_complexity,
77  reason = "Main processing logic with multiple steps"
78)]
79pub fn process_options(config: &Config, options_path: &Path) -> Result<()> {
80  // Read options JSON
81  let json_content = fs::read_to_string(options_path).wrap_err_with(|| {
82    format!("Failed to read options file: {}", options_path.display())
83  })?;
84
85  let options_data: Value = serde_json::from_str(&json_content)
86    .wrap_err("Failed to parse options JSON")?;
87
88  // First pass: collect all option names for validation
89  let mut valid_options = HashSet::new();
90  if let Value::Object(ref map) = options_data {
91    for key in map.keys() {
92      valid_options.insert(key.clone());
93    }
94  }
95
96  // Create processor once with validation enabled
97  let processor = create_processor(config, Some(valid_options));
98
99  // Extract options
100  let mut options = HashMap::new();
101
102  if let Value::Object(map) = options_data {
103    for (key, value) in map {
104      if let Value::Object(option_data) = &value {
105        let mut option = NixOption {
106          name:            key.clone(),
107          type_name:       String::new(),
108          description:     String::new(),
109          default:         None,
110          default_text:    None,
111          example:         None,
112          example_text:    None,
113          declared_in:     None,
114          declared_in_url: None,
115          internal:        false,
116          read_only:       false,
117        };
118
119        // Process fields from the option data
120        if let Some(Value::String(type_name)) = option_data.get("type") {
121          option.type_name.clone_from(type_name);
122        }
123
124        if let Some(Value::String(desc)) = option_data.get("description") {
125          let processed_desc = escape_html_in_markdown(desc);
126          let result = processor.render(&processed_desc);
127          option.description = result.html;
128        }
129
130        // Handle default values
131        if let Some(default_val) = option_data.get("default") {
132          if let Some(extracted_value) = extract_value(default_val, true) {
133            option.default_text = Some(extracted_value);
134          } else {
135            option.default = Some(default_val.clone());
136          }
137        }
138
139        if let Some(Value::String(text)) = option_data.get("defaultText") {
140          option.default_text = Some(text.clone());
141        }
142
143        // Handle example values
144        if let Some(example_val) = option_data.get("example") {
145          if let Some(extracted_value) = extract_value(example_val, true) {
146            option.example_text = Some(extracted_value);
147          } else {
148            option.example = Some(example_val.clone());
149          }
150        }
151
152        if let Some(Value::String(text)) = option_data.get("exampleText") {
153          option.example_text = Some(text.clone());
154        }
155
156        // Process declarations with location handling
157        if let Some(Value::Array(decls)) = option_data.get("declarations") {
158          if !decls.is_empty() {
159            let (display, url) = format_location(&decls[0], &config.revision);
160            option.declared_in = display;
161            option.declared_in_url = url;
162          }
163        }
164
165        // Read-only status
166        if let Some(Value::Bool(read_only)) = option_data.get("readOnly") {
167          option.read_only = *read_only;
168        }
169
170        // Internal status
171        if let Some(Value::Bool(internal)) = option_data.get("internal") {
172          option.internal = *internal;
173        }
174
175        if let Some(Value::Bool(visible)) = option_data.get("visible") {
176          if !visible {
177            option.internal = true;
178          }
179        }
180
181        // Use loc as fallback if no declaration
182        if option.declared_in.is_none() {
183          if let Some(Value::Array(loc_data)) = option_data.get("loc") {
184            if !loc_data.is_empty() {
185              // For loc data, join the parts to form a path
186              let parts: Vec<String> = loc_data
187                .iter()
188                .filter_map(|v| {
189                  if let Value::String(s) = v {
190                    Some(s.clone())
191                  } else {
192                    None
193                  }
194                })
195                .collect();
196
197              if !parts.is_empty() {
198                option.declared_in = Some(parts.join("."));
199                debug!("Set declared_in from loc: {}", parts.join("."));
200              }
201            }
202          }
203        }
204
205        // Always ensure a fallback for declared_in
206        if option.declared_in.is_none() {
207          option.declared_in = Some("configuration.nix".to_string());
208          debug!("Using fallback declared_in for {key}");
209        }
210
211        options.insert(key, option);
212      }
213    }
214  }
215
216  // Sort options by priority (enable > package > other)
217  let mut sorted_options: Vec<_> = options.into_iter().collect();
218  sorted_options.sort_by(|(name_a, _), (name_b, _)| {
219    // Calculate priority for name_a
220    let priority_a = if name_a.starts_with("enable") {
221      0
222    } else if name_a.starts_with("package") {
223      1
224    } else {
225      2
226    };
227
228    // Calculate priority for name_b
229    let priority_b = if name_b.starts_with("enable") {
230      0
231    } else if name_b.starts_with("package") {
232      1
233    } else {
234      2
235    };
236
237    // Compare by priority first, then by name
238    priority_a.cmp(&priority_b).then_with(|| name_a.cmp(name_b))
239  });
240
241  // Convert back to HashMap preserving the new order
242  let customized_options = sorted_options
243    .iter()
244    .map(|(key, opt)| {
245      let option_clone = opt.clone();
246      (key.clone(), option_clone)
247    })
248    .map(|(key, mut opt)| {
249      opt.name = html_escape::encode_text(&opt.name).to_string();
250      (key, opt)
251    })
252    .collect();
253
254  // Render options page
255  let html = template::render_options(config, &customized_options)?;
256  let output_path = config.output_dir.join("options.html");
257  fs::write(&output_path, html).wrap_err_with(|| {
258    format!("Failed to write options file: {}", output_path.display())
259  })?;
260
261  Ok(())
262}

264/// Format location with URL handling based on nixos-render-docs.
265///
266/// Converts a location value from the options JSON into a display string and
267/// an optional URL, supporting both local and GitHub-based references.
268///
269/// # Arguments
270///
271/// * `loc_value` - The JSON value representing the location.
272/// * `revision` - The git revision for GitHub links.
273///
274/// # Returns
275///
276/// A tuple of (display string, URL).
277fn format_location(
278  loc_value: &Value,
279  revision: &str,
280) -> (Option<String>, Option<String>) {
281  match loc_value {
282    // Handle string path
283    Value::String(path) => {
284      let path_str = path.as_str();
285
286      if path_str.starts_with('/') {
287        // Local filesystem path handling
288        let url = format!("file://{path_str}");
289
290        if path_str.contains("nixops") && path_str.contains("/nix/") {
291          let suffix_index = path_str.find("/nix/").map_or(0, |i| i + 5);
292          let suffix = &path_str[suffix_index..];
293          (Some(format!("<nixops/{suffix}>")), Some(url))
294        } else {
295          (Some(path_str.to_string()), Some(url))
296        }
297      } else {
298        // Path is relative to nixpkgs repo
299        let url = if revision == "local" {
300          format!("https://github.com/NixOS/nixpkgs/blob/master/{path_str}")
301        } else {
302          format!("https://github.com/NixOS/nixpkgs/blob/{revision}/{path_str}")
303        };
304
305        // Format display name
306        let display = format!("<nixpkgs/{path_str}>");
307        (Some(display), Some(url))
308      }
309    },
310
311    // Handle object with name and url
312    Value::Object(obj) => {
313      let display = if let Some(Value::String(name)) = obj.get("name") {
314        Some(html_escape::encode_text(name).to_string())
315      } else {
316        None
317      };
318
319      let url = obj.get("url").and_then(|u| {
320        if let Value::String(url_str) = u {
321          Some(url_str.clone())
322        } else {
323          None
324        }
325      });
326
327      (display, url)
328    },
329
330    // For other values, return None
331    _ => (None, None),
332  }
333}

335/// Escape HTML tags in markdown text before it's processed by the markdown
336/// processor, but preserve code blocks.
337///
338/// This function ensures that angle brackets are escaped outside of code
339/// blocks, preventing accidental HTML injection in documentation.
340///
341/// # Arguments
342///
343/// * `text` - The markdown text to escape.
344///
345/// # Returns
346///
347/// The escaped string, safe for markdown rendering.
348fn escape_html_in_markdown(text: &str) -> String {
349  // Split the text on backticks (`) to separate code blocks from regular text
350  let mut result = String::with_capacity(text.len());
351
352  // Track code block state
353  let mut in_code_block = false;
354  let mut in_inline_code = false;
355  let mut backquote_counter = 0;
356
357  for c in text.chars() {
358    if c == '`' {
359      // Manage backtick counting for code blocks
360      backquote_counter += 1;
361
362      if backquote_counter == 3 && !in_inline_code {
363        // Toggle fenced code block state
364        in_code_block = !in_code_block;
365        backquote_counter = 0; // Reset counter after handling triple
366      // backticks
367      } else if backquote_counter == 1 && !in_code_block {
368        // Toggle inline code state
369        in_inline_code = !in_inline_code;
370      }
371
372      result.push(c);
373    } else {
374      // Reset backtick counter when not a backtick
375      if backquote_counter > 0 && backquote_counter < 3 {
376        // We've seen some backticks but not enough for a code block
377        // These are either inline code delimiters or just literal backticks
378        backquote_counter = 0;
379      }
380
381      // Handle character based on context
382      if c == '<' && !in_code_block && !in_inline_code {
383        result.push_str("&lt;");
384      } else if c == '>' && !in_code_block && !in_inline_code {
385        result.push_str("&gt;");
386      } else {
387        result.push(c);
388      }
389    }
390  }
391
392  result

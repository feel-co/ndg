13/// Represents a `NixOS` configuration option
14#[derive(Debug, Clone, Default)]
15pub struct NixOption {
16  /// Option name (e.g., "services.nginx.enable")
17  pub name: String,
18
19  /// Option type (e.g., "boolean", "string", "signed integer", etc.)
20  pub type_name: String,
21
22  /// Option description
23  pub description: String,
24
25  /// Option default value (JSON)
26  pub default: Option<Value>,
27
28  /// Option default value as text
29  pub default_text: Option<String>,
30
31  /// Option example value (JSON)
32  pub example: Option<Value>,
33
34  /// Option example value as text
35  pub example_text: Option<String>,
36
37  /// Path to the file where the option is declared
38  pub declared_in: Option<String>,
39
40  /// Option declaration URL for hyperlink
41  pub declared_in_url: Option<String>,
42
43  /// Whether this option is internal
44  pub internal: bool,
45
46  /// Whether this option is read-only
47  pub read_only: bool,
48}

50/// Process options from a JSON file
51pub fn process_options(config: &Config, options_path: &Path) -> Result<()> {
52  // Read options JSON
53  let json_content = fs::read_to_string(options_path).context(format!(
54    "Failed to read options file: {}",
55    options_path.display()
56  ))?;
57
58  let options_data: Value = serde_json::from_str(&json_content)
59    .context("Failed to parse options JSON")?;
60
61  // Extract options
62  let mut options = HashMap::new();
63
64  if let Value::Object(map) = options_data {
65    for (key, value) in map {
66      if let Value::Object(option_data) = &value {
67        let mut option = NixOption {
68          name:            key.clone(),
69          type_name:       String::new(),
70          description:     String::new(),
71          default:         None,
72          default_text:    None,
73          example:         None,
74          example_text:    None,
75          declared_in:     None,
76          declared_in_url: None,
77          internal:        false,
78          read_only:       false,
79        };
80
81        // Process fields from the option data
82        if let Some(Value::String(type_name)) = option_data.get("type") {
83          option.type_name.clone_from(type_name);
84        }
85
86        if let Some(Value::String(desc)) = option_data.get("description") {
87          let processed_desc = escape_html_in_markdown(desc);
88          let processor = create_processor_from_config(config);
89          let result = processor.render(&processed_desc);
90          option.description = result.html;
91        }
92
93        // Handle default values
94        if let Some(default_val) = option_data.get("default") {
95          if let Some(extracted_value) = extract_value_from_json(default_val) {
96            option.default_text = Some(extracted_value);
97          } else {
98            option.default = Some(default_val.clone());
99          }
100        }
101
102        if let Some(Value::String(text)) = option_data.get("defaultText") {
103          option.default_text = Some(text.clone());
104        }
105
106        // Handle example values
107        if let Some(example_val) = option_data.get("example") {
108          if let Some(extracted_value) = extract_value_from_json(example_val) {
109            option.example_text = Some(extracted_value);
110          } else {
111            option.example = Some(example_val.clone());
112          }
113        }
114
115        if let Some(Value::String(text)) = option_data.get("exampleText") {
116          option.example_text = Some(text.clone());
117        }
118
119        // Process declarations with location handling
120        if let Some(Value::Array(decls)) = option_data.get("declarations") {
121          if !decls.is_empty() {
122            let (display, url) = format_location(&decls[0], &config.revision);
123            option.declared_in = display;
124            option.declared_in_url = url;
125          }
126        }
127
128        // Read-only status
129        if let Some(Value::Bool(read_only)) = option_data.get("readOnly") {
130          option.read_only = *read_only;
131        }
132
133        // Internal status
134        if let Some(Value::Bool(internal)) = option_data.get("internal") {
135          option.internal = *internal;
136        }
137
138        if let Some(Value::Bool(visible)) = option_data.get("visible") {
139          if !visible {
140            option.internal = true;
141          }
142        }
143
144        // Use loc as fallback if no declaration
145        if option.declared_in.is_none() {
146          if let Some(Value::Array(loc_data)) = option_data.get("loc") {
147            if !loc_data.is_empty() {
148              // For loc data, join the parts to form a path
149              let parts: Vec<String> = loc_data
150                .iter()
151                .filter_map(|v| {
152                  if let Value::String(s) = v {
153                    Some(s.clone())
154                  } else {
155                    None
156                  }
157                })
158                .collect();
159
160              if !parts.is_empty() {
161                option.declared_in = Some(parts.join("."));
162                debug!("Set declared_in from loc: {}", parts.join("."));
163              }
164            }
165          }
166        }
167
168        // Always ensure a fallback for declared_in
169        if option.declared_in.is_none() {
170          option.declared_in = Some("configuration.nix".to_string());
171          debug!("Using fallback declared_in for {key}");
172        }
173
174        options.insert(key, option);
175      }
176    }
177  }
178
179  // Sort options by priority (enable > package > other)
180  let mut sorted_options: Vec<_> = options.into_iter().collect();
181  sorted_options.sort_by(|(name_a, _), (name_b, _)| {
182    // Calculate priority for name_a
183    let priority_a = if name_a.starts_with("enable") {
184      0
185    } else if name_a.starts_with("package") {
186      1
187    } else {
188      2
189    };
190
191    // Calculate priority for name_b
192    let priority_b = if name_b.starts_with("enable") {
193      0
194    } else if name_b.starts_with("package") {
195      1
196    } else {
197      2
198    };
199
200    // Compare by priority first, then by name
201    priority_a.cmp(&priority_b).then_with(|| name_a.cmp(name_b))
202  });
203
204  // Convert back to HashMap preserving the new order
205  let customized_options = sorted_options
206    .iter()
207    .map(|(key, opt)| {
208      let option_clone = opt.clone();
209      (key.clone(), option_clone)
210    })
211    .map(|(key, mut opt)| {
212      opt.name = opt.name.replace('<', "&lt;").replace('>', "&gt;");
213      (key, opt)
214    })
215    .collect();
216
217  // Render options page
218  let html = template::render_options(config, &customized_options)?;
219  let output_path = config.output_dir.join("options.html");
220  fs::write(&output_path, html).context(format!(
221    "Failed to write options file: {}",
222    output_path.display()
223  ))?;
224
225  Ok(())
226}

228/// Format location with URL handling based on nixos-render-docs
229fn format_location(
230  loc_value: &Value,
231  revision: &str,
232) -> (Option<String>, Option<String>) {
233  match loc_value {
234    // Handle string path
235    Value::String(path) => {
236      let path_str = path.as_str();
237
238      if path_str.starts_with('/') {
239        // Local filesystem path handling
240        let url = format!("file://{path_str}");
241
242        if path_str.contains("nixops") && path_str.contains("/nix/") {
243          let suffix_index = path_str.find("/nix/").map_or(0, |i| i + 5);
244          let suffix = &path_str[suffix_index..];
245          (Some(format!("<nixops/{suffix}>")), Some(url))
246        } else {
247          (Some(path_str.to_string()), Some(url))
248        }
249      } else {
250        // Path is relative to nixpkgs repo
251        let url = if revision == "local" {
252          format!("https://github.com/NixOS/nixpkgs/blob/master/{path_str}")
253        } else {
254          format!("https://github.com/NixOS/nixpkgs/blob/{revision}/{path_str}")
255        };
256
257        // Format display name
258        let display = format!("<nixpkgs/{path_str}>");
259        (Some(display), Some(url))
260      }
261    },
262
263    // Handle object with name and url
264    Value::Object(obj) => {
265      let display = if let Some(Value::String(name)) = obj.get("name") {
266        Some(name.replace('<', "&lt;").replace('>', "&gt;"))
267      } else {
268        None
269      };
270
271      let url = obj.get("url").and_then(|u| {
272        if let Value::String(url_str) = u {
273          Some(url_str.clone())
274        } else {
275          None
276        }
277      });
278
279      (display, url)
280    },
281
282    // For other values, return None
283    _ => (None, None),
284  }
285}

287/// Escape HTML tags in markdown text before it's processed by the markdown
288/// processor, but preserve code blocks
289fn escape_html_in_markdown(text: &str) -> String {
290  // Split the text on backticks (`) to separate code blocks from regular text
291  let mut result = String::with_capacity(text.len());
292
293  // Track code block state
294  let mut in_code_block = false;
295  let mut in_inline_code = false;
296  let mut backquote_counter = 0;
297
298  for c in text.chars() {
299    if c == '`' {
300      // Manage backtick counting for code blocks
301      backquote_counter += 1;
302
303      if backquote_counter == 3 && !in_inline_code {
304        // Toggle fenced code block state
305        in_code_block = !in_code_block;
306        backquote_counter = 0; // Reset counter after handling triple
307      // backticks
308      } else if backquote_counter == 1 && !in_code_block {
309        // Toggle inline code state
310        in_inline_code = !in_inline_code;
311      }
312
313      result.push(c);
314    } else {
315      // Reset backtick counter when not a backtick
316      if backquote_counter > 0 && backquote_counter < 3 {
317        // We've seen some backticks but not enough for a code block
318        // These are either inline code delimiters or just literal backticks
319        backquote_counter = 0;
320      }
321
322      // Handle character based on context
323      if c == '<' && !in_code_block && !in_inline_code {
324        result.push_str("&lt;");
325      } else if c == '>' && !in_code_block && !in_inline_code {
326        result.push_str("&gt;");
327      } else {
328        result.push(c);
329      }
330    }
331  }
332
333  result
334}

336/// Extract the value from special JSON structures like literalExpression
337fn extract_value_from_json(value: &Value) -> Option<String> {
338  if let Value::Object(obj) = value {
339    // literalExpression and similar structured values
340    if let Some(Value::String(type_name)) = obj.get("_type") {
341      match type_name.as_str() {
342        // XXX: `literalDocBook` and `literalMD` have been deprecated as of
343        // 24.11 (I think) and they're supported here only for backwards compat.
344        // Those *will* be removed, at a later date.
345        "literalExpression" | "literalDocBook" | "literalMD" => {
346          if let Some(Value::String(text)) = obj.get("text") {
347            // For literalExpression, we should parse it as Nix code that may
348            // contain special characters Use code formatting
349            // markers (backticks) to indicate this is code The
350            // backticks will be used by the renderer to understand
351            // this needs special handling
352            if type_name.as_str() == "literalExpression" {
353              return Some(format!("`{}`", text.clone()));
354            }
355            return Some(text.to_string());
356          }
357        },
358        _ => {},
359      }
360    }
361  }
362
363  // For simple scalar values, just convert them to string
364  match value {
365    Value::String(s) => Some(s.clone()),
366    Value::Number(n) => Some(n.to_string()),
367    Value::Bool(b) => Some(b.to_string()),
368    Value::Null => Some("null".to_string()),
369    _ => None,
370  }

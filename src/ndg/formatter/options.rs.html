9/// Represents a `NixOS` configuration option
10#[derive(Debug, Clone, Default)]
11pub struct NixOption {
12    /// Option name (e.g., "services.nginx.enable")
13    pub name: String,
14
15    /// Option type (e.g., "boolean", "string", "signed integer", etc.)
16    pub type_name: String,
17
18    /// Option description
19    pub description: String,
20
21    /// Option default value (JSON)
22    pub default: Option<Value>,
23
24    /// Option default value as text
25    pub default_text: Option<String>,
26
27    /// Option example value (JSON)
28    pub example: Option<Value>,
29
30    /// Option example value as text
31    pub example_text: Option<String>,
32
33    /// Path to the file where the option is declared
34    pub declared_in: Option<String>,
35
36    /// Option declaration URL for hyperlink
37    pub declared_in_url: Option<String>,
38
39    /// Whether this option is internal
40    pub internal: bool,
41

46/// Process options from a JSON file
47pub fn process_options(config: &Config, options_path: &Path) -> Result<()> {
48    // Read options JSON
49    let json_content = fs::read_to_string(options_path).context(format!(
50        "Failed to read options file: {}",
51        options_path.display()
52    ))?;
53
54    let options_data: Value =
55        serde_json::from_str(&json_content).context("Failed to parse options JSON")?;
56
57    // Extract options
58    let mut options = HashMap::new();
59
60    if let Value::Object(map) = options_data {
61        for (key, value) in map {
62            if let Value::Object(option_data) = &value {
63                let mut option = NixOption {
64                    name: key.clone(),
65                    type_name: String::new(),
66                    description: String::new(),
67                    default: None,
68                    default_text: None,
69                    example: None,
70                    example_text: None,
71                    declared_in: None,
72                    declared_in_url: None,
73                    internal: false,
74                    read_only: false,
75                };
76
77                // Process fields from the option data
78                if let Some(Value::String(type_name)) = option_data.get("type") {
79                    option.type_name.clone_from(type_name);
80                }
81
82                if let Some(Value::String(desc)) = option_data.get("description") {
83                    let processed_desc = escape_html_in_markdown(desc);
84                    let options = ndg_commonmark::processor::MarkdownOptions::default();
85                    let processor = ndg_commonmark::processor::MarkdownProcessor::new(options);
86                    let result = processor.render(&processed_desc);
87                    option.description = result.html;
88                }
89
90                // Handle default values
91                if let Some(default_val) = option_data.get("default") {
92                    if let Some(extracted_value) = extract_value_from_json(default_val) {
93                        option.default_text = Some(extracted_value);
94                    } else {
95                        option.default = Some(default_val.clone());
96                    }
97                }
98
99                if let Some(Value::String(text)) = option_data.get("defaultText") {
100                    option.default_text = Some(text.clone());
101                }
102
103                // Handle example values
104                if let Some(example_val) = option_data.get("example") {
105                    if let Some(extracted_value) = extract_value_from_json(example_val) {
106                        option.example_text = Some(extracted_value);
107                    } else {
108                        option.example = Some(example_val.clone());
109                    }
110                }
111
112                if let Some(Value::String(text)) = option_data.get("exampleText") {
113                    option.example_text = Some(text.clone());
114                }
115
116                // Process declarations with location handling
117                if let Some(Value::Array(decls)) = option_data.get("declarations") {
118                    if !decls.is_empty() {
119                        let (display, url) = format_location(&decls[0], &config.revision);
120                        option.declared_in = display;
121                        option.declared_in_url = url;
122                    }
123                }
124
125                // Read-only status
126                if let Some(Value::Bool(read_only)) = option_data.get("readOnly") {
127                    option.read_only = *read_only;
128                }
129
130                // Internal status
131                if let Some(Value::Bool(internal)) = option_data.get("internal") {
132                    option.internal = *internal;
133                }
134
135                if let Some(Value::Bool(visible)) = option_data.get("visible") {
136                    if !visible {
137                        option.internal = true;
138                    }
139                }
140
141                // Use loc as fallback if no declaration
142                if option.declared_in.is_none() {
143                    if let Some(Value::Array(loc_data)) = option_data.get("loc") {
144                        if !loc_data.is_empty() {
145                            // For loc data, join the parts to form a path
146                            let parts: Vec<String> = loc_data
147                                .iter()
148                                .filter_map(|v| {
149                                    if let Value::String(s) = v {
150                                        Some(s.clone())
151                                    } else {
152                                        None
153                                    }
154                                })
155                                .collect();
156
157                            if !parts.is_empty() {
158                                option.declared_in = Some(parts.join("."));
159                                debug!("Set declared_in from loc: {}", parts.join("."));
160                            }
161                        }
162                    }
163                }
164
165                // Always ensure a fallback for declared_in
166                if option.declared_in.is_none() {
167                    option.declared_in = Some("configuration.nix".to_string());
168                    debug!("Using fallback declared_in for {key}");
169                }
170
171                options.insert(key, option);
172            }
173        }
174    }
175
176    // Sort options by priority (enable > package > other)
177    let mut sorted_options: Vec<_> = options.into_iter().collect();
178    sorted_options.sort_by(|(name_a, _), (name_b, _)| {
179        // Calculate priority for name_a
180        let priority_a = if name_a.starts_with("enable") {
181            0
182        } else if name_a.starts_with("package") {
183            1
184        } else {
185            2
186        };
187
188        // Calculate priority for name_b
189        let priority_b = if name_b.starts_with("enable") {
190            0
191        } else if name_b.starts_with("package") {
192            1
193        } else {
194            2
195        };
196
197        // Compare by priority first, then by name
198        priority_a.cmp(&priority_b).then_with(|| name_a.cmp(name_b))
199    });
200
201    // Convert back to HashMap preserving the new order
202    let customized_options = sorted_options
203        .iter()
204        .map(|(key, opt)| {
205            let option_clone = opt.clone();
206            (key.clone(), option_clone)
207        })
208        .map(|(key, mut opt)| {
209            opt.name = opt.name.replace('<', "&lt;").replace('>', "&gt;");
210            (key, opt)
211        })
212        .collect();
213
214    // Render options page
215    let html = template::render_options(config, &customized_options)?;
216    let output_path = config.output_dir.join("options.html");
217    fs::write(&output_path, html).context(format!(
218        "Failed to write options file: {}",
219        output_path.display()
220    ))?;
221
222    Ok(())
223}

225/// Format location with URL handling based on nixos-render-docs
226fn format_location(loc_value: &Value, revision: &str) -> (Option<String>, Option<String>) {
227    match loc_value {
228        // Handle string path
229        Value::String(path) => {
230            let path_str = path.as_str();
231
232            if path_str.starts_with('/') {
233                // Local filesystem path handling
234                let url = format!("file://{path_str}");
235
236                if path_str.contains("nixops") && path_str.contains("/nix/") {
237                    let suffix_index = path_str.find("/nix/").map_or(0, |i| i + 5);
238                    let suffix = &path_str[suffix_index..];
239                    (Some(format!("<nixops/{suffix}>")), Some(url))
240                } else {
241                    (Some(path_str.to_string()), Some(url))
242                }
243            } else {
244                // Path is relative to nixpkgs repo
245                let url = if revision == "local" {
246                    format!("https://github.com/NixOS/nixpkgs/blob/master/{path_str}")
247                } else {
248                    format!("https://github.com/NixOS/nixpkgs/blob/{revision}/{path_str}")
249                };
250
251                // Format display name
252                let display = format!("<nixpkgs/{path_str}>");
253                (Some(display), Some(url))
254            }
255        }
256
257        // Handle object with name and url
258        Value::Object(obj) => {
259            let display = if let Some(Value::String(name)) = obj.get("name") {
260                Some(name.replace('<', "&lt;").replace('>', "&gt;"))
261            } else {
262                None
263            };
264
265            let url = obj.get("url").and_then(|u| {
266                if let Value::String(url_str) = u {
267                    Some(url_str.clone())
268                } else {
269                    None
270                }
271            });
272
273            (display, url)
274        }
275
276        // For other values, return None
277        _ => (None, None),
278    }
279}

281/// Escape HTML tags in markdown text before it's processed by the markdown
282/// processor, but preserve code blocks
283fn escape_html_in_markdown(text: &str) -> String {
284    // Split the text on backticks (`) to separate code blocks from regular text
285    let mut result = String::with_capacity(text.len());
286
287    // Track code block state
288    let mut in_code_block = false;
289    let mut in_inline_code = false;
290    let mut backquote_counter = 0;
291
292    for c in text.chars() {
293        if c == '`' {
294            // Manage backtick counting for code blocks
295            backquote_counter += 1;
296
297            if backquote_counter == 3 && !in_inline_code {
298                // Toggle fenced code block state
299                in_code_block = !in_code_block;
300                backquote_counter = 0; // Reset counter after handling triple
301            // backticks
302            } else if backquote_counter == 1 && !in_code_block {
303                // Toggle inline code state
304                in_inline_code = !in_inline_code;
305            }
306
307            result.push(c);
308        } else {
309            // Reset backtick counter when not a backtick
310            if backquote_counter > 0 && backquote_counter < 3 {
311                // We've seen some backticks but not enough for a code block
312                // These are either inline code delimiters or just literal backticks
313                backquote_counter = 0;
314            }
315
316            // Handle character based on context
317            if c == '<' && !in_code_block && !in_inline_code {
318                result.push_str("&lt;");
319            } else if c == '>' && !in_code_block && !in_inline_code {
320                result.push_str("&gt;");
321            } else {
322                result.push(c);
323            }
324        }
325    }
326
327    result
328}

330/// Extract the value from special JSON structures like literalExpression
331fn extract_value_from_json(value: &Value) -> Option<String> {
332    if let Value::Object(obj) = value {
333        // literalExpression and similar structured values
334        if let Some(Value::String(type_name)) = obj.get("_type") {
335            match type_name.as_str() {
336                // XXX: `literalDocBook` and `literalMD` have been deprecated as of
337                // 24.11 (I think) and they're supported here only for backwards compat.
338                // Those *will* be removed, at a later date.
339                "literalExpression" | "literalDocBook" | "literalMD" => {
340                    if let Some(Value::String(text)) = obj.get("text") {
341                        // For literalExpression, we should parse it as Nix code that may contain
342                        // special characters Use code formatting markers
343                        // (backticks) to indicate this is code The backticks
344                        // will be used by the renderer to understand this needs special handling
345                        if type_name.as_str() == "literalExpression" {
346                            return Some(format!("`{}`", text.clone()));
347                        }
348                        return Some(text.to_string());
349                    }
350                }
351                _ => {}
352            }
353        }
354    }
355
356    // For simple scalar values, just convert them to string
357    match value {
358        Value::String(s) => Some(s.clone()),
359        Value::Number(n) => Some(n.to_string()),
360        Value::Bool(b) => Some(b.to_string()),
361        Value::Null => Some("null".to_string()),
362        _ => None,
363    }

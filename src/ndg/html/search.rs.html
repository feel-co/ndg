19/// Generate search index from markdown files
20pub fn generate_search_index(
21  config: &Config,
22  markdown_files: &[PathBuf],
23) -> Result<()> {
24  if !config.generate_search {
25    return Ok(());
26  }
27
28  info!("Generating search index...");
29
30  // Create search directory
31  let search_dir = config.output_dir.join("assets");
32  fs::create_dir_all(&search_dir)?;
33
34  // Create search index data
35  let mut documents = Vec::new();
36  let mut doc_id = 0;
37
38  // Process markdown files if available and input_dir is provided
39  if !markdown_files.is_empty() && config.input_dir.is_some() {
40    let input_dir = config.input_dir.as_ref().unwrap();
41    for file_path in markdown_files {
42      let content = fs::read_to_string(file_path).context(format!(
43        "Failed to read file for search indexing: {}",
44        file_path.display()
45      ))?;
46
47      let title = extract_title(&content).unwrap_or_else(|| {
48        file_path
49          .file_stem()
50          .unwrap_or_default()
51          .to_string_lossy()
52          .to_string()
53      });
54
55      // Use the existing markdown processor to handle all NDG-specific markup
56      let plain_text = utils::process_content_to_plain_text(&content, config);
57
58      let rel_path = file_path.strip_prefix(input_dir).context(format!(
59        "Failed to determine relative path for {}",
60        file_path.display()
61      ))?;
62
63      let mut output_path = rel_path.to_owned();
64      output_path.set_extension("html");
65
66      documents.push(SearchDocument {
67        id: doc_id.to_string(),
68        title,
69        content: plain_text,
70        path: output_path.to_string_lossy().to_string(),
71      });
72
73      doc_id += 1;
74    }
75  }
76
77  // Process options if available
78  if let Some(options_path) = &config.module_options {
79    // Try to load options data using serde_json::Value
80    if let Ok(options_content) = fs::read_to_string(options_path) {
81      if let Ok(options_data) = serde_json::from_str::<Value>(&options_content)
82      {
83        if let Some(options_obj) = options_data.as_object() {
84          for (key, option_value) in options_obj {
85            // Extract description from the option
86            let raw_description =
87              option_value["description"].as_str().unwrap_or("");
88
89            // Use the same clean processing as for markdown files
90            let plain_description =
91              utils::process_content_to_plain_text(raw_description, config);
92
93            // Create search entry for this option
94            documents.push(SearchDocument {
95              id:      doc_id.to_string(),
96              title:   format!("Option: {key}"),
97              content: plain_description,
98              path:    format!("options.html#option-{}", key.replace('.', "-")),
99            });
100            doc_id += 1;
101          }
102        }
103      }
104    }
105  }
106
107  // Write search index data.
108  // Always create a valid JSON array, even if empty.
109  let search_data_path = search_dir.join("search-data.json");
110  fs::write(&search_data_path, serde_json::to_string(&documents)?).context(
111    format!(
112      "Failed to write search data to {}",
113      search_data_path.display()
114    ),
115  )?;
116
117  // Create search page
118  create_search_page(config)?;
119
120  info!("Search index generated successfully");
121
122  Ok(())
123}

125/// Extract title from markdown content
126/// Extract title from markdown content (first H1)
127fn extract_title(content: &str) -> Option<String> {
128  ndg_commonmark::utils::extract_title_from_markdown(content)
129}

131/// Create search page
132pub fn create_search_page(config: &Config) -> Result<()> {
133  if !config.generate_search {
134    return Ok(());
135  }
136
137  info!("Creating search page...");
138
139  // Create context for the search page
140  let mut context = HashMap::new();
141  context.insert("title", format!("{} - Search", config.title));
142
143  // Render the search page using the template
144  let html = html::template::render_search(config, &context)?;
145
146  // Write the search page to the output directory
147  let search_page_path = config.output_dir.join("search.html");
148  fs::write(&search_page_path, &html).context(format!(
149    "Failed to write search page to {}",
150    search_page_path.display()
151  ))?;
152
153  Ok(())
154}

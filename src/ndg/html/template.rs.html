15/// Render a documentation page
16pub fn render(
17    config: &Config,
18    content: &str,
19    title: &str,
20    headers: &[Header],
21    rel_path: &Path,
22) -> Result<String> {
23    let mut tera = Tera::default();
24    let template_content = get_template_content(config, "default.html", DEFAULT_TEMPLATE)?;
25    tera.add_raw_template("default", &template_content)?;
26
27    // Generate table of contents from headers
28    let toc = generate_toc(headers);
29
30    // Generate document navigation
31    let doc_nav = generate_doc_nav(config, rel_path);
32
33    // Check if options are available
34    let has_options = if config.module_options.is_some() {
35        ""
36    } else {
37        "style=\"display:none;\""
38    };
39
40    // Generate custom scripts HTML
41    let custom_scripts = generate_custom_scripts(config, rel_path)?;
42
43    // Generate asset and navigation paths based on file location
44    let asset_paths = utils::generate_asset_paths(rel_path);
45
46    // Create context
47    let mut tera_context = tera::Context::new();
48    tera_context.insert("content", content);
49    tera_context.insert("title", title);
50    tera_context.insert("site_title", &config.title);
51    tera_context.insert("footer_text", &config.footer_text);

55    tera_context.insert("custom_scripts", &custom_scripts);
56    tera_context.insert("generate_search", &config.generate_search);
57
58    // Add asset paths
59    tera_context.insert(
60        "stylesheet_path",
61        asset_paths

73        asset_paths
74            .get("search_js_path")
75            .map_or("assets/search.js", std::string::String::as_str),
76    );
77    tera_context.insert(
78        "index_path",
79        asset_paths
80            .get("index_path")
81            .map_or("index.html", std::string::String::as_str),
82    );
83    tera_context.insert(
84        "options_path",

91        asset_paths

184/// Generate specialized TOC for options page
185fn generate_options_toc(
186    options: &HashMap<String, NixOption>,
187    config: &Config,
188    tera: &Tera,
189) -> Result<String> {
190    // Configured depth or default of 2
191    let depth = config.options_toc_depth;
192
193    let mut grouped_options: HashMap<String, Vec<&NixOption>> = HashMap::new();
194    let mut direct_parent_options: HashMap<String, &NixOption> = HashMap::new();
195
196    for option in options.values() {
197        let parent = get_option_parent(&option.name, depth);
198
199        // Check if this option exactly matches its parent category
200        if option.name == parent {
201            direct_parent_options.insert(parent.clone(), option);
202        }
203
204        // Add to grouped options
205        grouped_options.entry(parent).or_default().push(option);
206    }
207
208    // Separate categories into single options and dropdown categories
209    let mut single_options: Vec<tera::Value> = Vec::new();
210    let mut dropdown_categories: Vec<tera::Value> = Vec::new();
211
212    for (parent, opts) in &grouped_options {
213        let has_multiple_options = opts.len() > 1;
214        let has_child_options =
215            opts.len() > usize::from(direct_parent_options.contains_key(parent));
216
217        if !has_multiple_options && !has_child_options {
218            // Single option with no children
219            let option = opts[0];
220            let option_value = tera::to_value({
221                let mut map = tera::Map::new();
222                map.insert("name".to_string(), tera::to_value(&option.name)?);
223                map.insert("internal".to_string(), tera::to_value(option.internal)?);
224                map.insert("read_only".to_string(), tera::to_value(option.read_only)?);
225                map
226            })?;
227            single_options.push(option_value);
228        } else {
229            // Category with multiple options or child options
230            let mut category = tera::Map::new();
231            category.insert("name".to_string(), tera::to_value(parent)?);
232            category.insert("count".to_string(), tera::to_value(opts.len())?);
233
234            // Add parent option if it exists
235            if let Some(parent_option) = direct_parent_options.get(parent) {
236                let parent_option_value = tera::to_value({
237                    let mut map = tera::Map::new();
238                    map.insert("name".to_string(), tera::to_value(&parent_option.name)?);
239                    map.insert(
240                        "internal".to_string(),
241                        tera::to_value(parent_option.internal)?,
242                    );
243                    map.insert(
244                        "read_only".to_string(),
245                        tera::to_value(parent_option.read_only)?,
246                    );
247                    map
248                })?;
249                category.insert("parent_option".to_string(), parent_option_value);
250            }
251
252            // Add child options
253            let mut children = Vec::new();
254            let mut child_options: Vec<&NixOption> = opts
255                .iter()
256                .filter(|opt| opt.name != *parent)
257                .copied()
258                .collect();
259
260            // Sort by suffix
261            child_options.sort_by(|a, b| {
262                let a_suffix = a
263                    .name
264                    .strip_prefix(&format!("{parent}."))
265                    .unwrap_or(&a.name);
266                let b_suffix = b
267                    .name
268                    .strip_prefix(&format!("{parent}."))
269                    .unwrap_or(&b.name);
270                a_suffix.cmp(b_suffix)
271            });
272
273            for option in child_options {
274                let display_name = option
275                    .name
276                    .strip_prefix(&format!("{parent}."))
277                    .unwrap_or(&option.name);
278
279                let child_value = tera::to_value({
280                    let mut map = tera::Map::new();
281                    map.insert("name".to_string(), tera::to_value(&option.name)?);
282                    map.insert("display_name".to_string(), tera::to_value(display_name)?);
283                    map.insert("internal".to_string(), tera::to_value(option.internal)?);
284                    map.insert("read_only".to_string(), tera::to_value(option.read_only)?);
285                    map
286                })?;
287                children.push(child_value);
288            }
289
290            category.insert("children".to_string(), tera::to_value(children)?);
291            dropdown_categories.push(tera::to_value(category)?);
292        }
293    }
294
295    // Sort single options alphabetically
296    single_options.sort_by(|a, b| {
297        let a_name = a.get("name").and_then(|v| v.as_str()).unwrap_or("");
298        let b_name = b.get("name").and_then(|v| v.as_str()).unwrap_or("");
299        a_name.cmp(b_name)
300    });
301
302    // Sort dropdown categories
303    dropdown_categories.sort_by(|a, b| {
304        let a_name = a.get("name").and_then(|v| v.as_str()).unwrap_or("");
305        let b_name = b.get("name").and_then(|v| v.as_str()).unwrap_or("");
306
307        let a_components = a_name.split('.').count();
308        let b_components = b_name.split('.').count();
309
310        // Sort by component count first
311        match a_components.cmp(&b_components) {
312            std::cmp::Ordering::Equal => a_name.cmp(b_name), // Then alphabetically
313            other => other,
314        }
315    });
316
317    let mut tera_context = tera::Context::new();
318    tera_context.insert("single_options", &single_options);
319    tera_context.insert("dropdown_categories", &dropdown_categories);
320
321    // Render the template
322    let rendered = tera.render("options_toc", &tera_context)?;
323
324    Ok(rendered)
325}

327/// Extract the parent category from an option name with configurable depth
328fn get_option_parent(option_name: &str, depth: usize) -> String {
329    let parts: Vec<&str> = option_name.split('.').collect();
330    if parts.len() <= depth {
331        option_name.to_string()
332    } else {
333        parts[..depth].join(".")
334    }
335}

425/// Get the template content from file in template directory or use default
426fn get_template_content(config: &Config, template_name: &str, fallback: &str) -> Result<String> {
427    // Try to get the template from the configured template directory
428    if let Some(template_dir) = config.get_template_path() {
429        let template_path = template_dir.join(template_name);
430        if template_path.exists() {
431            return fs::read_to_string(&template_path).with_context(|| {
432                format!(
433                    "Failed to read custom template file: {}. Check file permissions and ensure the file is valid UTF-8",
434                    template_path.display()
435                )
436            });
437        }
438    }
439
440    // If template_path is specified but doesn't point to a directory with our
441    // template
442    if let Some(template_path) = &config.template_path {
443        if template_path.exists() && template_name == "default.html" {
444            // XXX: For backward compatibility
445            // If template_path is a file, use it for default.html
446            return fs::read_to_string(template_path).with_context(|| {
447                format!(
448                    "Failed to read custom template file: {}. Check file permissions and ensure the file is valid UTF-8",
449                    template_path.display()
450                )
451            });
452        }
453    }
454
455    // Use fallback embedded template if no custom template found
456    Ok(fallback.to_string())
457}

459/// Generate the document navigation HTM
460fn generate_doc_nav(config: &Config, current_file_rel_path: &Path) -> String {
461    let mut doc_nav = String::new();
462    let root_prefix = utils::calculate_root_relative_path(current_file_rel_path);
463
464    // Only process markdown files if input_dir is provided
465    if let Some(input_dir) = &config.input_dir {
466        let entries: Vec<_> = walkdir::WalkDir::new(input_dir)
467            .follow_links(true)
468            .into_iter()
469            .filter_map(std::result::Result::ok)
470            .filter(|e| e.path().is_file() && e.path().extension().is_some_and(|ext| ext == "md"))
471            .collect();
472
473        if !entries.is_empty() {
474            // Partition entries into special and regular
475            let mut special_entries = Vec::new();
476            let mut regular_entries = Vec::new();
477
478            for entry in &entries {
479                let path = entry.path();
480                if let Ok(rel_doc_path) = path.strip_prefix(input_dir) {
481                    let file_name = rel_doc_path
482                        .file_name()
483                        .and_then(|n| n.to_str())
484                        .unwrap_or("")
485                        .to_ascii_lowercase();
486                    if file_name == "index.md" || file_name == "readme.md" {
487                        special_entries.push(entry);
488                    } else {
489                        regular_entries.push(entry);
490                    }
491                }
492            }
493
494            // Helper closure to render a nav entry
495            let mut render_entry = |entry: &walkdir::DirEntry| {
496                let path = entry.path();
497                if let Ok(rel_doc_path) = path.strip_prefix(input_dir) {
498                    let mut html_path = rel_doc_path.to_path_buf();
499                    html_path.set_extension("html");
500
501                    // Create relative path from current file to target file
502                    let target_path = format!("{}{}", root_prefix, html_path.to_string_lossy());
503
504                    let page_title = fs::read_to_string(path).map_or_else(
505                        |_| {
506                            html_path
507                                .file_stem()
508                                .unwrap_or_default()
509                                .to_string_lossy()
510                                .to_string()
511                        },
512                        |content| {
513                            content.lines().next().map_or_else(
514                                || {
515                                    html_path
516                                        .file_stem()
517                                        .unwrap_or_default()
518                                        .to_string_lossy()
519                                        .to_string()
520                                },
521                                |first_line| {
522                                    first_line.strip_prefix("# ").map_or_else(
523                                        || {
524                                            html_path
525                                                .file_stem()
526                                                .unwrap_or_default()
527                                                .to_string_lossy()
528                                                .to_string()
529                                        },
530                                        ndg_commonmark::utils::clean_anchor_patterns,
531                                    )
532                                },
533                            )
534                        },
535                    );
536
537                    writeln!(
538                        doc_nav,
539                        "<li><a href=\"{target_path}\">{page_title}</a></li>"
540                    )
541                    .expect("Failed to write to doc_nav string");
542                }
543            };
544
545            // Render special entries first (index.md, README.md)
546            for entry in &special_entries {
547                render_entry(entry);
548            }
549            // Then render regular entries
550            for entry in &regular_entries {
551                render_entry(entry);
552            }
553        }
554    }
555
556    // Add link to options page if module_options is configured
557    if doc_nav.is_empty() && config.module_options.is_some() {
558        doc_nav.push_str(&format!(
559            "<li><a href=\"{root_prefix}options.html\">Module Options</a></li>\n"
560        ));
561    }
562
563    // Add search link only if search is enabled
564    if config.generate_search {
565        doc_nav.push_str(&format!(
566            "<li><a href=\"{root_prefix}search.html\">Search</a></li>\n"
567        ));
568    }
569
570    doc_nav
571}

573/// Generate custom scripts HTML
574fn generate_custom_scripts(config: &Config, current_file_rel_path: &Path) -> Result<String> {
575    let mut custom_scripts = String::new();
576    let root_prefix = utils::calculate_root_relative_path(current_file_rel_path);
577
578    // Add any user scripts from script_paths. This is additive, not replacing. To replace
579    // default content, the user should specify `--template-dir` or `--template` instead.
580    for script_path in &config.script_paths {
581        // Relative path to script
582        let script_relative_path = format!("{}{}", root_prefix, script_path.to_string_lossy());
583        write!(
584            custom_scripts,
585            "<script defer src=\"{script_relative_path}\"></script>"
586        )?;
587    }
588
589    Ok(custom_scripts)
590}

592/// Generate table of contents from headers
593fn generate_toc(headers: &[Header]) -> String {
594    let mut toc = String::new();
595    let mut current_level = 0;
596
597    for header in headers {
598        // Only include h1, h2, h3 in TOC
599        if header.level <= 3 {
600            // Adjust TOC nesting
601            while current_level < header.level - 1 {
602                toc.push_str("<ul>");
603                current_level += 1;
604            }
605            while current_level > header.level - 1 {
606                toc.push_str("</ul>");
607                current_level -= 1;
608            }
609
610            // Add TOC item
611            if current_level == header.level - 1 {
612                toc.push_str("<li>");
613            } else {
614                toc.push_str("</li><li>");
615            }
616
617            let mut visible_text = header.text.as_str();
618            if let Some(idx) = visible_text.rfind("{#") {
619                if visible_text.ends_with('}') {
620                    visible_text = visible_text[..idx].trim_end();
621                }
622            }
623            writeln!(toc, "<a href=\"#{}\">{}</a>", header.id, visible_text)
624                .expect("Failed to write to toc string");
625        }
626    }
627
628    // Close open tags
629    while current_level > 0 {
630        toc.push_str("</li></ul>");
631        current_level -= 1;
632    }
633    if !headers.is_empty() && !toc.is_empty() && !toc.ends_with("</li>") {
634        toc.push_str("</li>");
635    }
636
637    toc
638}

640/// Generate the options HTML content
641fn generate_options_html(options: &HashMap<String, NixOption>) -> String {
642    let mut options_html = String::with_capacity(options.len() * 500); // FIXME: Rough estimate for capacity
643
644    // Sort options by name
645    let mut option_keys: Vec<_> = options.keys().collect();
646    option_keys.sort();
647
648    for key in option_keys {
649        let option = &options[key];
650        let option_id = format!("option-{}", option.name.replace('.', "-"));
651
652        // Open option container with ID for direct linking
653        writeln!(options_html, "<div class=\"option\" id=\"{option_id}\">")
654            .expect("Failed to write to options_html string");
655
656        // Option name with anchor link and copy button
657        write!(
658            options_html,
659            "  <h3 class=\"option-name\">\n    <a href=\"#{}\" class=\"option-anchor\">{}</a>\n    <span \
660             class=\"copy-link\" title=\"Copy link to this option\"></span>\n    <span \
661             class=\"copy-feedback\">Link copied!</span>\n  </h3>\n",
662            option_id, option.name
663        )
664        .expect("Failed to write to options_html string");
665
666        // Option metadata (internal/readOnly)
667        let mut metadata = Vec::new();
668        if option.internal {
669            metadata.push("internal");
670        }
671        if option.read_only {
672            metadata.push("read-only");
673        }
674
675        if !metadata.is_empty() {
676            writeln!(
677                options_html,
678                "  <div class=\"option-metadata\">{}</div>",
679                metadata.join(", ")
680            )
681            .expect("Failed to write to options_html string");
682        }
683
684        // Option type
685        writeln!(
686            options_html,
687            "  <div class=\"option-type\">Type: <code>{}</code></div>",
688            option.type_name
689        )
690        .expect("Failed to write to options_html string");
691
692        // Option description
693        writeln!(
694            options_html,
695            "  <div class=\"option-description\">{}</div>",
696            option.description
697        )
698        .expect("Failed to write to options_html string");
699
700        // Add default value if available
701        add_default_value(&mut options_html, option);
702
703        // Add example if available
704        add_example_value(&mut options_html, option);
705
706        // Option declared in - now with hyperlink support
707        if let Some(declared_in) = &option.declared_in {
708            if let Some(url) = &option.declared_in_url {
709                writeln!(
710                    options_html,
711                    "  <div class=\"option-declared\">Declared in: <code><a href=\"{url}\" \
712                     target=\"_blank\">{declared_in}</a></code></div>"
713                )
714                .expect("Failed to write to options_html string");
715            } else {
716                writeln!(
717                    options_html,
718                    "  <div class=\"option-declared\">Declared in: <code>{declared_in}</code></div>"
719                )
720                .expect("Failed to write to options_html string");
721            }
722        }
723
724        // Close option div
725        options_html.push_str("</div>\n");
726    }
727
728    options_html
729}

731/// Add default value to options HTML
732fn add_default_value(html: &mut String, option: &NixOption) {
733    if let Some(default_text) = &option.default_text {
734        // Remove surrounding backticks if present (from literalExpression)
735        let clean_default = if default_text.starts_with('`')
736            && default_text.ends_with('`')
737            && default_text.len() > 2
738        {
739            &default_text[1..default_text.len() - 1]
740        } else {
741            default_text
742        };
743
744        writeln!(
745            html,
746            "  <div class=\"option-default\">Default: <code>{clean_default}</code></div>"
747        )
748        .expect("Failed to write to options HTML string");
749    } else if let Some(default_val) = &option.default {
750        writeln!(
751            html,
752            "  <div class=\"option-default\">Default: <code>{default_val}</code></div>"
753        )
754        .expect("Failed to write to options HTML string");
755    }
756}

758/// Add example value to options HTML
759fn add_example_value(html: &mut String, option: &NixOption) {
760    if let Some(example_text) = &option.example_text {
761        // Process the example text to preserve code formatting
762        if example_text.contains('\n') {
763            // Multi-line examples - preserve formatting with pre/code
764            // Process special characters to ensure valid HTML
765            let safe_example = example_text.replace('<', "&lt;").replace('>', "&gt;");
766
767            // Remove backticks if they're surrounding the entire content (from
768            // literalExpression)
769            let trimmed_example = if safe_example.starts_with('`')
770                && safe_example.ends_with('`')
771                && safe_example.len() > 2
772            {
773                &safe_example[1..safe_example.len() - 1]
774            } else {
775                &safe_example
776            };
777
778            writeln!(
779                html,
780                "  <div class=\"option-example\">Example: <pre><code>{trimmed_example}</code></pre></div>"
781            )
782            .expect("Failed to write to options HTML string");
783        } else {
784            // Check if this is already a code block (surrounded by backticks)
785            if example_text.starts_with('`')
786                && example_text.ends_with('`')
787                && example_text.len() > 2
788            {
789                // This is inline code - extract the content and properly escape it
790                let code_content = &example_text[1..example_text.len() - 1];
791                let safe_content = code_content.replace('<', "&lt;").replace('>', "&gt;");
792                writeln!(
793                    html,
794                    "  <div class=\"option-example\">Example: <code>{safe_content}</code></div>"
795                )
796                .expect("Failed to write to options HTML string");
797            } else {
798                // Regular inline example - still needs escaping
799                let safe_example = example_text.replace('<', "&lt;").replace('>', "&gt;");
800                writeln!(
801                    html,
802                    "  <div class=\"option-example\">Example: <code>{safe_example}</code></div>"
803                )
804                .expect("Failed to write to options HTML string");
805            }
806        }
807    } else if let Some(example_val) = &option.example {
808        let example_str = example_val.to_string();
809        let safe_example = example_str.replace('<', "&lt;").replace('>', "&gt;");
810        if example_str.contains('\n') {
811            // Multi-line JSON examples need special handling
812            writeln!(
813                html,
814                "  <div class=\"option-example\">Example: <pre><code>{safe_example}</code></pre></div>"
815            )
816            .expect("Failed to write to options HTML string");
817        } else {
818            // Single-line JSON examples
819            writeln!(
820                html,
821                "  <div class=\"option-example\">Example: <code>{safe_example}</code></div>"
822            )
823            .expect("Failed to write to options HTML string");
824        }
825    }

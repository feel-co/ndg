3/// Calculate the relative path prefix needed to reach the root from a given
4/// file path For example: "docs/subdir/file.html" would return "../"
5///              "docs/subdir/nested/file.html" would return "../../"
6#[must_use]
7pub fn calculate_root_relative_path(file_rel_path: &Path) -> String {
8  let depth = file_rel_path.components().count();
9  if depth <= 1 {
10    String::new() // file is at root level
11  } else {
12    "../".repeat(depth - 1)
13  }
14}

16/// Generate proper asset paths for templates based on file location
17#[must_use]
18pub fn generate_asset_paths(
19  file_rel_path: &Path,
20) -> HashMap<&'static str, String> {
21  let root_prefix = calculate_root_relative_path(file_rel_path);
22
23  let mut paths = HashMap::new();
24  paths.insert("stylesheet_path", format!("{root_prefix}assets/style.css"));
25  paths.insert("main_js_path", format!("{root_prefix}assets/main.js"));
26  paths.insert("search_js_path", format!("{root_prefix}assets/search.js"));
27
28  // Navigation paths
29  paths.insert("index_path", format!("{root_prefix}index.html"));
30  paths.insert("options_path", format!("{root_prefix}options.html"));
31  paths.insert("search_path", format!("{root_prefix}search.html"));
32
33  paths
34}

42/// Process content through the markdown pipeline and extract plain text
43#[must_use]
44pub fn process_content_to_plain_text(
45  content: &str,
46  config: &crate::config::Config,
47) -> String {
48  let processor = crate::utils::create_processor_from_config(config);
49  let result = processor.render(content);
50  strip_markdown(&result.html)
51    .replace('\n', " ")
52    .replace("  ", " ")
53    .trim()
54    .to_string()
55}

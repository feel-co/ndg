21/// Collects all included files from markdown documents in the input directory.
22///
23/// This scans all markdown files and collects the set of files that are
24/// included via NDG's markdown processor.
25///
26/// # Arguments
27///
28/// * `config` - The loaded configuration for documentation generation.
29///
30/// # Returns
31///
32/// A set of relative paths (to the input directory) of all included files.
33///
34/// # Errors
35///
36/// Returns an error if any markdown file cannot be read.
37pub fn collect_included_files(config: &Config) -> Result<HashSet<PathBuf>> {
38  let mut all_included_files: HashSet<PathBuf> = HashSet::new();
39
40  if let Some(ref input_dir) = config.input_dir {
41    let files = collect_markdown_files(input_dir);
42
43    for file_path in &files {
44      let content = fs::read_to_string(file_path).wrap_err_with(|| {
45        format!("Failed to read markdown file: {}", file_path.display())
46      })?;
47      let processor = create_processor_from_config(config);
48      let base_dir = file_path.parent().unwrap_or_else(|| {
49        config.input_dir.as_deref().unwrap_or(Path::new("."))
50      });
51      let processor = processor.with_base_dir(base_dir);
52      let result = processor.render(&content);
53
54      for inc in &result.included_files {
55        // Include paths are relative to the file's directory, not input_dir
56        let inc_path = base_dir.join(&inc.path);
57        if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
58          all_included_files.insert(inc_rel.to_path_buf());
59        }
60      }
61    }
62  }
63
64  Ok(all_included_files)
65}

67/// Processes all markdown files in the input directory and writes HTML output.
68///
69/// This function renders all markdown files, handles custom output paths,
70/// and writes the resulting HTML files to the output directory. Files that are
71/// included in other files are skipped unless they have custom output paths.
72///
73/// # Arguments
74///
75/// * `config` - The loaded configuration for documentation generation.
76///
77/// # Returns,
78///
79/// A vector of all processed markdown file paths.
80///
81/// # Errors
82///
83/// Returns an error if any file cannot be read, rendered, or written.
84/// Processes all markdown files in the input directory and writes HTML output.
85///
86/// This function renders all markdown files, handles custom output paths,
87/// and writes the resulting HTML files to the output directory. Files that are
88/// included in other files are skipped unless they have custom output paths.
89///
90/// # Errors
91///
92/// Returns an error if any file cannot be read, rendered, or written.
93///
94/// # Panics
95///
96/// Panics if `config.input_dir` is `None` when processing markdown files.
97pub fn process_markdown_files(config: &Config) -> Result<Vec<PathBuf>> {
98  if let Some(ref input_dir) = config.input_dir {
99    info!("Input directory: {}", input_dir.display());
100    let files = collect_markdown_files(input_dir);
101    info!("Found {} markdown files", files.len());
102
103    // Map: output html path -> OutputEntry
104    let mut output_map: HashMap<String, OutputEntry> = HashMap::new();
105
106    // Track custom outputs keyed by included file path
107    let mut pending_custom_outputs: HashMap<PathBuf, Vec<String>> =
108      HashMap::new();
109
110    // First pass: collect all custom outputs for included files
111    for file_path in &files {
112      let content = fs::read_to_string(file_path).wrap_err_with(|| {
113        format!("Failed to read markdown file: {}", file_path.display())
114      })?;
115      let processor = create_processor_from_config(config);
116
117      let base_dir = file_path.parent().unwrap_or_else(|| {
118        config.input_dir.as_deref().unwrap_or(Path::new("."))
119      });
120      let processor = processor.with_base_dir(base_dir);
121
122      let result = processor.render(&content);
123
124      // Track custom outputs for included files
125      for inc in &result.included_files {
126        if let Some(ref custom_output) = inc.custom_output {
127          let inc_path = base_dir.join(&inc.path);
128          if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
129            pending_custom_outputs
130              .entry(inc_rel.to_path_buf())
131              .or_default()
132              .push(custom_output.clone());
133          }
134        }
135      }
136    }
137
138    for file_path in &files {
139      let content = fs::read_to_string(file_path).wrap_err_with(|| {
140        format!("Failed to read markdown file: {}", file_path.display())
141      })?;
142      let processor = create_processor_from_config(config);
143
144      // Set base directory to file's parent directory for relative includes
145      let base_dir = file_path.parent().unwrap_or_else(|| {
146        config.input_dir.as_deref().unwrap_or(Path::new("."))
147      });
148      let processor = processor.with_base_dir(base_dir);
149
150      let result = processor.render(&content);
151
152      let input_dir = config.input_dir.as_ref().expect("input_dir required");
153      let rel_path = file_path
154        .strip_prefix(input_dir)
155        .expect("strip_prefix failed");
156      let mut output_path = rel_path.to_path_buf();
157      output_path.set_extension("html");
158      let output_path_str = output_path.to_string_lossy().to_string();
159
160      // Check if this file is included in another file
161      let is_included = config.excluded_files.contains(rel_path);
162
163      // Get any pending custom outputs for this file
164      let custom_outputs =
165        pending_custom_outputs.remove(rel_path).unwrap_or_default();
166
167      // If this file has custom outputs via html:into-file, write to those
168      // locations
169      if custom_outputs.is_empty() {
170        output_map.insert(
171          output_path_str,
172          (
173            result.html,
174            result.headers,
175            result.title,
176            file_path.clone(),
177            is_included,
178          ),
179        );
180      } else {
181        for custom in custom_outputs {
182          output_map.insert(
183            custom.clone(),
184            (
185              result.html.clone(),
186              result.headers.clone(),
187              result.title.clone(),
188              file_path.clone(),
189              false,
190            ),
191          );
192        }
193      }
194    }
195
196    // Write outputs, skipping those marked as included (unless they have custom
197    // output)
198    for (out_path, (html_content, headers, title, _src_md, is_included)) in
199      &output_map
200    {
201      if *is_included {
202        continue;
203      }
204
205      let rel_path = Path::new(out_path);
206      if rel_path.is_absolute() {
207        log::warn!(
208          "Output path '{out_path}' is absolute (would write to '{}'). \
209           Attempting to normalize to relative.",
210          rel_path.display()
211        );
212      }
213
214      // Always force rel_path to be relative, never absolute
215      let rel_path = rel_path
216        .strip_prefix(std::path::MAIN_SEPARATOR_STR)
217        .unwrap_or(rel_path);
218
219      let html = template::render(
220        config,
221        html_content,
222        title.as_deref().unwrap_or(&config.title),
223        headers,
224        rel_path,
225      )?;
226
227      let output_path = config.output_dir.join(rel_path);
228      if let Some(parent) = output_path.parent() {
229        fs::create_dir_all(parent).wrap_err_with(|| {
230          format!("Failed to create output directory: {}", parent.display())
231        })?;
232      }
233
234      fs::write(&output_path, html).wrap_err_with(|| {
235        format!("Failed to write output HTML: {}", output_path.display())
236      })?;
237    }
238
239    Ok(files)
240  } else {
241    info!("No input directory provided, skipping markdown processing");
242    Ok(Vec::new())
243  }
244}

246/// Creates a markdown processor from the NDG configuration.
247///
248/// This configures the processor with GFM, code highlighting, and manpage URL
249/// mappings as specified in the configuration.
250///
251/// # Arguments
252/// * `config` - The loaded configuration for documentation generation.
253///
254/// # Returns
255/// A configured `MarkdownProcessor`.
256#[must_use]
257pub fn create_processor_from_config(config: &Config) -> MarkdownProcessor {
258  let mut builder = MarkdownOptionsBuilder::new()
259    .gfm(true)
260    .highlight_code(config.highlight_code);
261
262  if let Some(mappings_path) = &config.manpage_urls_path {
263    builder = builder
264      .manpage_urls_path(Some(mappings_path.to_string_lossy().to_string()));
265  }
266
267  MarkdownProcessor::new(builder.build())
268}

270/// Extracts the page title from a markdown file.
271///
272/// This attempts to extract the first heading as the title, falling back to the
273/// file name if no heading is found or the file cannot be read.
274///
275/// # Arguments
276///
277/// * `file_path` - The path to the markdown file.
278/// * `html_path` - The corresponding HTML output path (used for fallback).
279///
280/// # Returns
281///
282/// The extracted title or the file stem as a fallback.
283#[must_use]
284pub fn extract_page_title(file_path: &Path, html_path: &Path) -> String {
285  let default_title = html_path
286    .file_stem()
287    .unwrap_or_default()
288    .to_string_lossy()
289    .to_string();
290
291  match fs::read_to_string(file_path) {
292    Ok(content) => {
293      ndg_commonmark::utils::extract_markdown_title(&content)
294        .unwrap_or(default_title)
295    },
296    Err(_) => default_title,
297  }

12/// Creates a fallback index page listing available documents.
13///
14/// This is used when no `index.md` is present. It generates a simple HTML
15/// listing of all processed markdown files, using their first heading or
16/// filename as the display title.
17///
18/// # Arguments
19///
20/// * `config` - The loaded configuration for documentation generation.
21/// * `markdown_files` - The list of processed markdown file paths.
22///
23/// # Returns
24///
25/// A [`String`] containing the fallback HTML content.
26#[must_use]
27pub fn create_fallback_index(
28  config: &Config,
29  markdown_files: &[PathBuf],
30) -> String {
31  let mut content = format!(
32    "<h1>{}</h1>\n<p>This is a fallback page created by ndg.</p>",
33    &config.title
34  );
35
36  // Add file listing if we have an input directory
37  if let Some(input_dir) = &config.input_dir {
38    if !markdown_files.is_empty() {
39      let mut file_list = String::with_capacity(markdown_files.len() * 100); // preallocate based on estimated size
40      file_list.push_str("<h2>Available Documents</h2>\n<ul>\n");
41
42      for file_path in markdown_files {
43        if let Ok(rel_path) = file_path.strip_prefix(input_dir) {
44          let mut html_path = rel_path.to_path_buf();
45          html_path.set_extension("html");
46
47          // Get page title from first heading or filename
48          let page_title = extract_page_title(file_path, &html_path);
49
50          writeln!(
51            file_list,
52            "  <li><a href=\"{}\">{}</a></li>",
53            html_path.to_string_lossy(),
54            page_title
55          )
56          .unwrap();
57        }
58      }
59
60      file_list.push_str("</ul>");
61      content.push_str(&file_list);
62    }
63  }
64
65  content
66}

68/// Process the options page for a given JSON file if configured, and generate
69/// options documentation for the processed Nix module options.
70///
71/// # Arguments
72///
73/// * `config` - The loaded configuration for documentation generation.
74///
75/// # Returns
76///
77/// Returns `Ok(true)` if options were processed, `Ok(false)` otherwise.
78///
79/// # Errors
80///
81/// Returns an error if processing fails.
82pub fn process_module_options(config: &Config) -> Result<bool> {
83  if let Some(options_path) = &config.module_options {
84    info!("Processing options.json from {}", options_path.display());
85    options::process_options(config, options_path)?;
86    Ok(true)
87  } else {
88    info!("Module options were not set, skipping options processing");
89    Ok(false)
90  }

12/// Creates a fallback index page listing available documents.
13///
14/// This is used when no `index.md` is present. It generates a simple HTML
15/// listing of all processed markdown files, using their first heading or
16/// filename as the display title.
17///
18/// # Arguments
19///
20/// * `config` - The loaded configuration for documentation generation.
21/// * `markdown_files` - The list of processed markdown file paths.
22///
23/// # Returns
24///
25/// A [`String`] containing the fallback HTML content.
26#[must_use]
27pub fn create_fallback_index(
28  config: &Config,
29  markdown_files: &[PathBuf],
30) -> String {
31  let mut content = format!(
32    "<h1>{}</h1>\n<p>This is a fallback page created by ndg.</p>",
33    &config.title
34  );
35
36  // Add file listing if we have an input directory
37  if let Some(input_dir) = &config.input_dir {
38    if !markdown_files.is_empty() {
39      let mut file_list = String::with_capacity(markdown_files.len() * 100); // preallocate based on estimated size
40      file_list.push_str("<h2>Available Documents</h2>\n<ul>\n");
41
42      for file_path in markdown_files {
43        if let Ok(rel_path) = file_path.strip_prefix(input_dir) {
44          // Skip included files that are not generated as HTML
45          if config.excluded_files.contains(rel_path) {
46            continue;
47          }
48
49          let mut html_path = rel_path.to_path_buf();
50          html_path.set_extension("html");
51
52          // Get page title from first heading or filename
53          let page_title = extract_page_title(file_path, &html_path);
54
55          writeln!(
56            file_list,
57            "  <li><a href=\"{}\">{}</a></li>",
58            html_path.to_string_lossy(),
59            page_title
60          )
61          .expect("failed to write file list entry");
62        }
63      }
64
65      file_list.push_str("</ul>");
66      content.push_str(&file_list);
67    }
68  }
69
70  content
71}

73/// Process the options page for a given JSON file if configured, and generate
74/// options documentation for the processed Nix module options.
75///
76/// # Arguments
77///
78/// * `config` - The loaded configuration for documentation generation.
79///
80/// # Returns
81///
82/// Returns `Ok(true)` if options were processed, `Ok(false)` otherwise.
83///
84/// # Errors
85///
86/// Returns an error if processing fails.
87pub fn process_module_options(config: &Config) -> Result<bool> {
88  if let Some(options_path) = &config.module_options {
89    info!("Processing options.json from {}", options_path.display());
90    options::process_options(config, options_path)?;
91    Ok(true)
92  } else {
93    info!("Module options were not set, skipping options processing");
94    Ok(false)
95  }
96}

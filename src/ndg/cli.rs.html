1use std::path::PathBuf;
2
3use clap::{Parser, Subcommand};
4
5/// Command line interface for ndg
6#[derive(Parser, Debug)]
7#[command(author, version, about = "NDG: Not a Docs Generator")]
8pub struct Cli {
9  /// Subcommand to execute (see [`Commands`])
10  #[command(subcommand)]
11  pub command: Option<Commands>,
12
13  /// Enable verbose debug logging
14  #[arg(short, long)]
15  pub verbose: bool,
16
17  /// Path to configuration file (TOML or JSON)
18  #[arg(short = 'c', long = "config")]
19  pub config_file: Option<PathBuf>,
20}
21
22/// All supported subcommands for the ndg CLI.
23#[derive(Subcommand, Debug)]
24pub enum Commands {
25  /// Initialize a new NDG configuration file
26  Init {
27    /// Path to create the configuration file at
28    #[arg(short, long, default_value = "ndg.toml")]
29    output: PathBuf,
30
31    /// Format of the configuration file.
32    #[arg(short = 'F', long, default_value = "toml", value_parser = ["toml", "json"])]
33    format: String,
34
35    /// Force overwrite if file already exists
36    #[arg(short, long)]
37    force: bool,
38  },
39
40  /// Export default templates to a directory for customization.
41  Export {
42    /// Output directory for template files.
43    #[arg(short, long, default_value = "templates")]
44    output_dir: PathBuf,
45
46    /// Whether to overwrite existing files.
47    #[arg(long)]
48    force: bool,
49
50    /// Specific templates to export (e.g., html, css, js). If not specified,
51    /// exports all.
52    #[arg(short, long, action = clap::ArgAction::Append)]
53    templates: Vec<String>,
54  },
55
56  /// Process documentation and generate HTML.
57  Html {
58    /// Path to the directory containing markdown files.
59    #[arg(short, long)]
60    input_dir: Option<PathBuf>,
61
62    /// Output directory for generated documentation.
63    #[arg(short, long)]
64    output_dir: Option<PathBuf>,
65
66    /// Number of threads to use for parallel processing.
67    #[arg(short = 'p', long = "jobs")]
68    jobs: Option<usize>,
69
70    /// Path to custom template file.
71    #[arg(short, long)]
72    template: Option<PathBuf>,
73
74    /// Path to directory containing template files. Templates override
75    /// built-in ones (default.html, options.html, etc.)
76    #[arg(long = "template-dir")]
77    template_dir: Option<PathBuf>,
78
79    /// Path to custom stylesheet (can be specified multiple times)
80    #[arg(short, long, action = clap::ArgAction::Append)]
81    stylesheet: Vec<PathBuf>,
82
83    /// Path to custom JavaScript file (can be specified multiple times)
84    #[arg(long, action = clap::ArgAction::Append)]
85    script: Vec<PathBuf>,
86
87    /// Title of the documentation. Will be used in various components via
88    /// the templating options.
89    #[arg(short = 'T', long)]
90    title: Option<String>,
91
92    /// Footer text for the documentation.
93    #[arg(short = 'f', long)]
94    footer: Option<String>,
95
96    /// Path to a JSON file containing module options in the same format
97    /// expected by nixos-render-docs.
98    #[arg(short = 'j', long)]
99    module_options: Option<PathBuf>,
100
101    /// Depth of parent categories in options section in the sidebar.
102    #[arg(long = "options-depth", value_parser = clap::value_parser!(usize))]
103    options_toc_depth: Option<usize>,
104
105    /// Path to manpage URL mappings JSON file.
106    #[arg(long = "manpage-urls")]
107    manpage_urls: Option<PathBuf>,
108
109    /// Whether to generate search data and render relevant components.
110    #[arg(short = 'S', long = "generate-search")]
111    generate_search: bool,
112
113    /// Whether to enable syntax highlighting for code blocks.
114    #[arg(long = "highlight-code")]
115    highlight_code: bool,
116
117    /// GitHub revision for linking to source files.
118    #[arg(long, default_value = "local")]
119    revision: Option<String>,
120  },
121
122  /// Generate manpage from options.
123  Manpage {
124    /// Path to a JSON file containing module options.
125    #[arg(short = 'j', long, required = true)]
126    module_options: PathBuf,
127
128    /// Output file for the generated manpage
129    #[arg(short, long)]
130    output_file: Option<PathBuf>,
131
132    /// Header text to include at the beginning of the manpage
133    #[arg(short = 'H', long)]
134    header: Option<String>,
135
136    /// Footer text to include at the end of the manpage
137    #[arg(short = 'F', long)]
138    footer: Option<String>,
139
140    /// Title for the manpage
141    #[arg(short = 'T', long)]
142    title: Option<String>,
143
144    /// Section number for the manpage
145    #[arg(short = 's', long, default_value = "5")]
146    section: u8,
147  },
148}
149
150impl Cli {
151  /// Parse command line arguments into a [`Cli`] struct.
152  #[must_use]
153  pub fn parse_args() -> Self {
154    Self::parse()
155  }

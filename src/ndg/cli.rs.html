1use std::path::PathBuf;
2
3use clap::{Command, CommandFactory, Parser, Subcommand};
4
5/// Command line interface for ndg
6#[derive(Parser, Debug)]
7#[command(author, version, about = "Nix Documentation Generator")]
8pub struct Cli {
9  #[command(subcommand)]
10  pub command: Option<Commands>,
11
12  /// Enable verbose debug logging
13  #[arg(short, long)]
14  pub verbose: bool,
15
16  /// Path to configuration file (TOML or JSON)
17  #[arg(short = 'c', long = "config")]
18  pub config_file: Option<PathBuf>,
19}
20
21#[derive(Subcommand, Debug)]
22pub enum Commands {
23  /// Initialize a new NDG configuration file
24  Init {
25    /// Path to create the configuration file at
26    #[arg(short, long, default_value = "ndg.toml")]
27    output: PathBuf,
28
29    /// Format of the configuration file (toml or json)
30    #[arg(short = 'F', long, default_value = "toml", value_parser = ["toml", "json"])]
31    format: String,
32
33    /// Force overwrite if file already exists
34    #[arg(short, long)]
35    force: bool,
36  },
37
38  /// Export default templates to a directory for customization
39  ExportTemplates {
40    /// Output directory for template files
41    #[arg(short, long, default_value = "templates")]
42    output_dir: PathBuf,
43
44    /// Overwrite existing files
45    #[arg(long)]
46    force: bool,
47  },
48
49  /// Generate shell completions and manpages
50  Generate {
51    /// Directory to output generated files
52    #[arg(short, long, default_value = "dist")]
53    output_dir: PathBuf,
54
55    /// Only generate shell completions
56    #[arg(long)]
57    completions_only: bool,
58
59    /// Only generate manpage
60    #[arg(long)]
61    manpage_only: bool,
62  },
63
64  /// Process documentation and generate HTML
65  Html {
66    /// Path to the directory containing markdown files
67    #[arg(short, long)]
68    input_dir: Option<PathBuf>,
69
70    /// Output directory for generated documentation
71    #[arg(short, long)]
72    output_dir: Option<PathBuf>,
73
74    /// Number of threads to use for parallel processing
75    #[arg(short = 'p', long = "jobs")]
76    jobs: Option<usize>,
77
78    /// Path to custom template file
79    #[arg(short, long)]
80    template: Option<PathBuf>,
81
82    /// Path to directory containing template files. Use 'ndg export-templates'
83    /// to get the default templates for customization. Templates in this
84    /// directory override built-in templates (default.html, options.html,
85    /// search.html, etc.)
86    #[arg(long = "template-dir")]
87    template_dir: Option<PathBuf>,
88
89    /// Path to custom stylesheet. This can be specified multiple times to
90    /// include multiple stylesheets in order.
91    #[arg(short, long, action = clap::ArgAction::Append)]
92    stylesheet: Vec<PathBuf>,
93
94    /// Path to custom Javascript file to include. This can be specified
95    /// multiple times to create multiple script tags in order.
96    #[arg(long, action = clap::ArgAction::Append)]
97    script: Vec<PathBuf>,
98
99    /// Title of the documentation. Will be used in various components via
100    /// the templating options.
101    #[arg(short = 'T', long)]
102    title: Option<String>,
103
104    /// Footer text for the documentation
105    #[arg(short = 'f', long)]
106    footer: Option<String>,
107
108    /// Path to a JSON file containing module options in the same format
109    /// expected by nixos-render-docs.
110    #[arg(short = 'j', long)]
111    module_options: Option<PathBuf>,
112
113    /// Depth of parent categories in options TOC
114    #[arg(long = "options-depth", value_parser = clap::value_parser!(usize))]
115    options_toc_depth: Option<usize>,
116
117    /// Path to manpage URL mappings JSON file
118    #[arg(long = "manpage-urls")]
119    manpage_urls: Option<PathBuf>,
120
121    /// Whether to generate search functionality
122    #[arg(short = 'S', long = "generate-search")]
123    generate_search: Option<bool>,
124
125    /// Whether to enable syntax highlighting for code blocks
126    #[arg(long = "highlight-code")]
127    highlight_code: Option<bool>,
128
129    /// GitHub revision for linking to source files (defaults to 'local')
130    #[arg(long)]
131    revision: Option<String>,
132  },
133
134  /// Generate manpage from options
135  Manpage {
136    /// Path to a JSON file containing module options
137    #[arg(short = 'j', long, required = true)]
138    module_options: PathBuf,
139
140    /// Output file for the generated manpage
141    #[arg(short, long)]
142    output_file: Option<PathBuf>,
143
144    /// Header text to include at the beginning of the manpage
145    #[arg(short = 'H', long)]
146    header: Option<String>,
147
148    /// Footer text to include at the end of the manpage
149    #[arg(short = 'F', long)]
150    footer: Option<String>,
151
152    /// Title for the manpage
153    #[arg(short = 'T', long)]
154    title: Option<String>,
155
156    /// Section number for the manpage
157    #[arg(short = 's', long, default_value = "5")]
158    section: u8,
159  },
160}
161
162impl Cli {
163  #[must_use]
164  pub fn parse_args() -> Self {
165    Self::parse()
166  }
167
168  #[must_use]
169  pub fn command() -> Command {
170    <Self as CommandFactory>::command()
171  }
172}

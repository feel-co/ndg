18/// Copy assets to output directory
19pub fn copy_assets(config: &Config) -> Result<()> {
20  // Create assets directory
21  let assets_dir = config.output_dir.join("assets");
22  fs::create_dir_all(&assets_dir)?;
23
24  // Generate and write CSS
25  let css = generate_css(config)?;
26  fs::write(assets_dir.join("style.css"), css)
27    .context("Failed to write CSS file")?;
28
29  // Copy main.js, which is always needed for the default templates
30  copy_template_asset(config, &assets_dir, "main.js", MAIN_JS)?;
31
32  // Copy custom assets if they exist
33  copy_custom_assets(config, &assets_dir)?;
34
35  // Copy script files to assets directory
36  copy_script_files(config, &assets_dir)?;
37
38  // Create search.js for search functionality
39  if config.generate_search {
40    copy_template_asset(config, &assets_dir, "search.js", SEARCH_JS)?;
41  }
42
43  Ok(())
44}

46/// Copies template asset while still allowing user override
47fn copy_template_asset(
48  config: &Config,
49  assets_dir: &Path,
50  filename: &str,
51  fallback_content: &str,
52) -> Result<()> {
53  let content = if let Some(path) = config.get_template_file(filename) {
54    if path.exists() {
55      fs::read_to_string(&path).with_context(|| {
56        format!("Failed to read {} from: {}", filename, path.display())
57      })?
58    } else {
59      fallback_content.to_string()
60    }
61  } else {
62    fallback_content.to_string()
63  };
64
65  fs::write(assets_dir.join(filename), content)
66    .with_context(|| format!("Failed to write {filename} to assets directory"))
67}

69/// Copy custom assets from the configured directory
70fn copy_custom_assets(config: &Config, assets_dir: &Path) -> Result<()> {
71  if let Some(custom_assets_dir) = &config.assets_dir {
72    if custom_assets_dir.exists() && custom_assets_dir.is_dir() {
73      debug!("Copying custom assets from {}", custom_assets_dir.display());
74
75      let options = fs_extra::dir::CopyOptions::new().overwrite(true);
76      fs_extra::dir::copy(custom_assets_dir, assets_dir, &options)
77        .context("Failed to copy custom assets")?;
78    }
79  }
80  Ok(())
81}

83/// Copy script files to the assets directory
84fn copy_script_files(config: &Config, assets_dir: &Path) -> Result<()> {
85  for script_path in &config.script_paths {
86    if script_path.exists() {
87      let file_name = script_path
88        .file_name()
89        .ok_or_else(|| anyhow::anyhow!("Invalid script filename"))?;
90      let dest_path = assets_dir.join(file_name);
91
92      fs::read(script_path)
93        .with_context(|| {
94          format!("Failed to read script file {}", script_path.display())
95        })
96        .and_then(|content| {
97          fs::write(&dest_path, content).with_context(|| {
98            format!("Failed to write script file to {}", dest_path.display())
99          })
100        })?;
101    }
102  }
103  Ok(())
104}

106/// Generate CSS from stylesheet
107fn generate_css(config: &Config) -> Result<String> {
108  // Use template CSS if available, otherwise use default CSS
109  let mut combined_css = if let Some(template_path) = config.get_template_path()
110  {
111    let template_css_path = template_path.join("default.css");
112    if template_css_path.exists() {
113      fs::read_to_string(&template_css_path).with_context(|| {
114        format!(
115          "Failed to read template CSS: {}",
116          template_css_path.display()
117        )
118      })?
119    } else {
120      // No template CSS found, use default
121      String::from(DEFAULT_CSS)
122    }
123  } else {
124    // No template directory specified, use default CSS
125    String::from(DEFAULT_CSS)
126  };
127
128  // Add any custom stylesheets provided via --stylesheet
129  if !config.stylesheet_paths.is_empty() {
130    // Process each stylesheet in order
131    for (index, stylesheet_path) in config.stylesheet_paths.iter().enumerate() {
132      if stylesheet_path.exists() {
133        let content =
134          fs::read_to_string(stylesheet_path).with_context(|| {
135            format!(
136              "Failed to read stylesheet {}: {}",
137              index + 1,
138              stylesheet_path.display()
139            )
140          })?;
141
142        // Process SCSS if needed
143        let processed_content =
144          if stylesheet_path.extension().is_some_and(|ext| ext == "scss") {
145            grass::from_string(content.clone(), &grass::Options::default())
146              .with_context(|| {
147                format!(
148                  "Failed to compile SCSS to CSS for stylesheet {}",
149                  index + 1
150                )
151              })?
152          } else {
153            content
154          };
155
156        // Add a comment to separate multiple stylesheets
157        combined_css.push_str("\n\n/* Custom Stylesheet ");
158        combined_css.push_str(&(index + 1).to_string());
159        combined_css.push_str(": ");
160        combined_css.push_str(&stylesheet_path.display().to_string());
161        combined_css.push_str(" */\n");
162        combined_css.push_str(&processed_content);
163      }
164    }
165  }
166
167  Ok(combined_css)
168}

209/// Collect all included files from markdown documents
210pub fn collect_included_files(
211  config: &Config,
212) -> Result<std::collections::HashSet<std::path::PathBuf>> {
213  use std::collections::HashSet;
214  let mut all_included_files: HashSet<std::path::PathBuf> = HashSet::new();
215
216  if let Some(ref input_dir) = config.input_dir {
217    let files = collect_markdown_files(input_dir);
218
219    for file_path in &files {
220      let content = std::fs::read_to_string(file_path)?;
221      let processor = create_processor_from_config(config);
222      let base_dir = file_path.parent().unwrap_or_else(|| {
223        config
224          .input_dir
225          .as_deref()
226          .unwrap_or(std::path::Path::new("."))
227      });
228      let processor = processor.with_base_dir(base_dir);
229      let result = processor.render(&content);
230
231      for inc in &result.included_files {
232        // Include paths are relative to the file's directory, not input_dir
233        let inc_path = base_dir.join(&inc.path);
234        if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
235          all_included_files.insert(inc_rel.to_path_buf());
236        }
237      }
238    }
239  }
240
241  Ok(all_included_files)
242}

244/// Process markdown files
245pub fn process_markdown_files(
246  config: &Config,
247) -> Result<Vec<std::path::PathBuf>> {
248  use std::collections::HashMap;
249  if let Some(ref input_dir) = config.input_dir {
250    info!("Input directory: {}", input_dir.display());
251    let files = collect_markdown_files(input_dir);
252    info!("Found {} markdown files", files.len());
253
254    // Map: output html path -> (content, headers, title, source md, is_include)
255    // FIXME: clippy warning about type complexity, it's not wrong tbh
256    let mut output_map: HashMap<
257      String,
258      (
259        String,
260        Vec<ndg_commonmark::Header>,
261        Option<String>,
262        std::path::PathBuf,
263        bool,
264      ),
265    > = HashMap::new();
266
267    // Track custom outputs keyed by included file path
268    let mut pending_custom_outputs: HashMap<std::path::PathBuf, Vec<String>> =
269      HashMap::new();
270
271    // First pass: collect all custom outputs for included files
272    for file_path in &files {
273      let content = std::fs::read_to_string(file_path)?;
274      let processor = create_processor_from_config(config);
275
276      let base_dir = file_path.parent().unwrap_or_else(|| {
277        config
278          .input_dir
279          .as_deref()
280          .unwrap_or(std::path::Path::new("."))
281      });
282      let processor = processor.with_base_dir(base_dir);
283
284      let result = processor.render(&content);
285
286      // Track custom outputs for included files
287      for inc in &result.included_files {
288        if let Some(ref custom_output) = inc.custom_output {
289          let inc_path = base_dir.join(&inc.path);
290          if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
291            pending_custom_outputs
292              .entry(inc_rel.to_path_buf())
293              .or_default()
294              .push(custom_output.clone());
295          }
296        }
297      }
298    }
299
300    for file_path in &files {
301      let content = std::fs::read_to_string(file_path)?;
302      let processor = create_processor_from_config(config);
303
304      // Set base directory to file's parent directory for relative includes
305      let base_dir = file_path.parent().unwrap_or_else(|| {
306        config
307          .input_dir
308          .as_deref()
309          .unwrap_or(std::path::Path::new("."))
310      });
311      let processor = processor.with_base_dir(base_dir);
312
313      let result = processor.render(&content);
314
315      let input_dir = config.input_dir.as_ref().expect("input_dir required");
316      let rel_path = file_path
317        .strip_prefix(input_dir)
318        .expect("strip_prefix failed");
319      let mut output_path = rel_path.to_path_buf();
320      output_path.set_extension("html");
321      let output_path_str = output_path.to_string_lossy().to_string();
322
323      // Check if this file is included in another file
324      let is_included = config.excluded_files.contains(rel_path);
325
326      // Get any pending custom outputs for this file
327      let custom_outputs =
328        pending_custom_outputs.remove(rel_path).unwrap_or_default();
329
330      // If this file has custom outputs via html:into-file, write to those
331      // locations
332      if custom_outputs.is_empty() {
333        output_map.insert(
334          output_path_str,
335          (
336            result.html,
337            result.headers,
338            result.title,
339            file_path.clone(),
340            is_included,
341          ),
342        );
343      } else {
344        for custom in custom_outputs {
345          output_map.insert(
346            custom.clone(),
347            (
348              result.html.clone(),
349              result.headers.clone(),
350              result.title.clone(),
351              file_path.clone(),
352              false,
353            ),
354          );
355        }
356      }
357    }
358
359    // Write outputs, skipping those marked as included (unless they have custom
360    // output)
361    use crate::html::template;
362
363    for (out_path, (html_content, headers, title, _src_md, is_included)) in
364      &output_map
365    {
366      if *is_included {
367        continue;
368      }
369      let rel_path = std::path::Path::new(out_path);
370      let html = template::render(
371        config,
372        html_content,
373        title.as_deref().unwrap_or(&config.title),
374        headers,
375        rel_path,
376      )?;
377      let output_path = config.output_dir.join(rel_path);
378      if let Some(parent) = output_path.parent() {
379        std::fs::create_dir_all(parent)?;
380      }
381      std::fs::write(&output_path, html)?;
382    }
383
384    Ok(files)
385  } else {
386    info!("No input directory provided, skipping markdown processing");
387    Ok(Vec::new())
388  }
389}

402/// Create a fallback index page
403#[must_use]
404pub fn create_fallback_index(
405  config: &Config,
406  markdown_files: &[std::path::PathBuf],
407) -> String {
408  let mut content = format!(
409    "<h1>{}</h1>\n<p>This is a fallback page created by ndg.</p>",
410    &config.title
411  );
412
413  // Add file listing if we have an input directory
414  if let Some(input_dir) = &config.input_dir {
415    if !markdown_files.is_empty() {
416      let mut file_list = String::with_capacity(markdown_files.len() * 100); // Preallocate based on estimated size
417      file_list.push_str("<h2>Available Documents</h2>\n<ul>\n");
418
419      for file_path in markdown_files {
420        if let Ok(rel_path) = file_path.strip_prefix(input_dir) {
421          let mut html_path = rel_path.to_path_buf();
422          html_path.set_extension("html");
423
424          // Get page title from first heading or filename
425          let page_title = extract_page_title(file_path, &html_path);
426
427          writeln!(
428            file_list,
429            "  <li><a href=\"{}\">{}</a></li>",
430            html_path.to_string_lossy(),
431            page_title
432          )
433          .unwrap();
434        }
435      }
436
437      file_list.push_str("</ul>");
438      content.push_str(&file_list);
439    }
440  }
441
442  content
443}

445/// Create a markdown processor from ndg config
446#[must_use]
447pub fn create_processor_from_config(config: &Config) -> MarkdownProcessor {
448  let mut builder = MarkdownOptionsBuilder::new()
449    .gfm(true)
450    .highlight_code(config.highlight_code);
451
452  if let Some(mappings_path) = &config.manpage_urls_path {
453    builder = builder
454      .manpage_urls_path(Some(mappings_path.to_string_lossy().to_string()));
455  }
456
457  MarkdownProcessor::new(builder.build())
458}

460/// Extract the page title from a markdown file
461#[must_use]
462pub fn extract_page_title(
463  file_path: &std::path::Path,
464  html_path: &std::path::Path,
465) -> String {
466  let default_title = html_path
467    .file_stem()
468    .unwrap_or_default()
469    .to_string_lossy()
470    .to_string();
471
472  match fs::read_to_string(file_path) {
473    Ok(content) => {
474      ndg_commonmark::utils::extract_markdown_title(&content)
475        .unwrap_or(default_title)
476    },
477    Err(_) => default_title,
478  }
479}

481/// Ensure that index.html exists in the output directory
482pub fn ensure_index(
483  config: &Config,
484  options_processed: bool,
485  markdown_files: &[std::path::PathBuf],
486) -> Result<()> {
487  let index_path = config.output_dir.join("index.html");
488
489  // Check if index.html already exists (already generated from index.md)
490  if index_path.exists() {
491    return Ok(());
492  }
493
494  // If options were processed, try to use options.html as index
495  if options_processed {
496    let options_path = config.output_dir.join("options.html");
497    if options_path.exists() {
498      info!("Using options.html as index.html");
499
500      // Copy the content of options.html to index.html
501      let options_content = fs::read(&options_path)?;
502      fs::write(&index_path, options_content)?;
503      return Ok(());
504    }
505  }
506
507  // Create a fallback index page if input directory was provided but no
508  // index.md existed
509  if config.input_dir.is_some() {
510    info!("No index.md found, creating fallback index.html");
511
512    let content = create_fallback_index(config, markdown_files);
513
514    // Create a simple HTML document using our template system
515    let headers = vec![ndg_commonmark::Header {
516      text:  config.title.clone(),
517      level: 1,
518      id:    "welcome".to_string(),
519    }];
520
521    let html = html::template::render(
522      config,
523      &content,
524      &config.title,
525      &headers,
526      &std::path::PathBuf::from("index.html"),
527    )?;
528
529    fs::write(&index_path, html)?;
530  }
531
532  Ok(())

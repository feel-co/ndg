14/// Copy assets to output directory
15pub fn copy_assets(config: &Config) -> Result<()> {
16    // Create assets directory
17    let assets_dir = config.output_dir.join("assets");
18    fs::create_dir_all(&assets_dir)?;
19
20    // Generate and write CSS
21    let css = generate_css(config)?;
22    fs::write(assets_dir.join("style.css"), css).context("Failed to write CSS file")?;
23
24    // Copy main.js, which is always needed for the default templates
25    copy_template_asset(config, &assets_dir, "main.js", MAIN_JS)?;
26
27    // Copy custom assets if they exist
28    copy_custom_assets(config, &assets_dir)?;
29
30    // Copy script files to assets directory
31    copy_script_files(config, &assets_dir)?;
32
33    // Create search.js for search functionality
34    if config.generate_search {
35        copy_template_asset(config, &assets_dir, "search.js", SEARCH_JS)?;
36    }
37
38    Ok(())
39}

41/// Copies template asset while still allowing user override
42fn copy_template_asset(
43    config: &Config,
44    assets_dir: &Path,
45    filename: &str,
46    fallback_content: &str,
47) -> Result<()> {
48    let content = if let Some(path) = config.get_template_file(filename) {
49        if path.exists() {
50            fs::read_to_string(&path)
51                .with_context(|| format!("Failed to read {} from: {}", filename, path.display()))?
52        } else {
53            fallback_content.to_string()
54        }
55    } else {
56        fallback_content.to_string()
57    };
58
59    fs::write(assets_dir.join(filename), content)
60        .with_context(|| format!("Failed to write {filename} to assets directory"))
61}

63/// Copy custom assets from the configured directory
64fn copy_custom_assets(config: &Config, assets_dir: &Path) -> Result<()> {
65    if let Some(custom_assets_dir) = &config.assets_dir {
66        if custom_assets_dir.exists() && custom_assets_dir.is_dir() {
67            debug!("Copying custom assets from {}", custom_assets_dir.display());
68
69            let options = fs_extra::dir::CopyOptions::new().overwrite(true);
70            fs_extra::dir::copy(custom_assets_dir, assets_dir, &options)
71                .context("Failed to copy custom assets")?;
72        }
73    }
74    Ok(())
75}

77/// Copy script files to the assets directory
78fn copy_script_files(config: &Config, assets_dir: &Path) -> Result<()> {
79    for script_path in &config.script_paths {
80        if script_path.exists() {
81            let file_name = script_path
82                .file_name()
83                .ok_or_else(|| anyhow::anyhow!("Invalid script filename"))?;
84            let dest_path = assets_dir.join(file_name);
85
86            fs::read(script_path)
87                .with_context(|| format!("Failed to read script file {}", script_path.display()))
88                .and_then(|content| {
89                    fs::write(&dest_path, content).with_context(|| {
90                        format!("Failed to write script file to {}", dest_path.display())
91                    })
92                })?;
93        }
94    }
95    Ok(())
96}

98/// Generate CSS from stylesheet
99fn generate_css(config: &Config) -> Result<String> {
100    // Use template CSS if available, otherwise use default CSS
101    let mut combined_css = if let Some(template_path) = config.get_template_path() {
102        let template_css_path = template_path.join("default.css");
103        if template_css_path.exists() {
104            fs::read_to_string(&template_css_path).with_context(|| {
105                format!(
106                    "Failed to read template CSS: {}",
107                    template_css_path.display()
108                )
109            })?
110        } else {
111            // No template CSS found, use default
112            String::from(DEFAULT_CSS)
113        }
114    } else {
115        // No template directory specified, use default CSS
116        String::from(DEFAULT_CSS)
117    };
118
119    // Add any custom stylesheets provided via --stylesheet
120    if !config.stylesheet_paths.is_empty() {
121        // Process each stylesheet in order
122        for (index, stylesheet_path) in config.stylesheet_paths.iter().enumerate() {
123            if stylesheet_path.exists() {
124                let content = fs::read_to_string(stylesheet_path).with_context(|| {
125                    format!(
126                        "Failed to read stylesheet {}: {}",
127                        index + 1,
128                        stylesheet_path.display()
129                    )
130                })?;
131
132                // Process SCSS if needed
133                let processed_content = if stylesheet_path
134                    .extension()
135                    .is_some_and(|ext| ext == "scss")
136                {
137                    grass::from_string(content.clone(), &grass::Options::default()).with_context(
138                        || format!("Failed to compile SCSS to CSS for stylesheet {}", index + 1),
139                    )?
140                } else {
141                    content
142                };
143
144                // Add a comment to separate multiple stylesheets
145                combined_css.push_str("\n\n/* Custom Stylesheet ");
146                combined_css.push_str(&(index + 1).to_string());
147                combined_css.push_str(": ");
148                combined_css.push_str(&stylesheet_path.display().to_string());
149                combined_css.push_str(" */\n");
150                combined_css.push_str(&processed_content);
151            }
152        }
153    }
154
155    Ok(combined_css)
156}

190/// Process markdown files
191pub fn process_markdown_files(config: &Config) -> Result<Vec<std::path::PathBuf>> {
192    if let Some(ref input_dir) = config.input_dir {
193        info!("Input directory: {}", input_dir.display());
194        let files = collect_markdown_files(input_dir);
195        info!("Found {} markdown files", files.len());
196
197        if !files.is_empty() {
198            // Process files in parallel
199            use rayon::prelude::*;
200
201            use crate::html::template;
202            files.par_iter().try_for_each(|file_path| {
203                let content = std::fs::read_to_string(file_path)?;
204                let processor = create_processor_from_config(config);
205                let result = processor.render(&content);
206
207                // The HTML already includes syntax highlighting
208                let highlighted_html = result.html;
209
210                let headers = result.headers;
211                let title = result.title;
212                let input_dir = config.input_dir.as_ref().expect("input_dir required");
213                let rel_path = file_path
214                    .strip_prefix(input_dir)
215                    .expect("strip_prefix failed");
216                let mut output_path = config.output_dir.join(rel_path);
217                output_path.set_extension("html");
218
219                let html = template::render(
220                    config,
221                    &highlighted_html,
222                    title.as_deref().unwrap_or(&config.title),
223                    &headers,
224                    rel_path,
225                )?;
226
227                if let Some(parent) = output_path.parent() {
228                    std::fs::create_dir_all(parent)?;
229                }
230                std::fs::write(&output_path, html)?;
231                Ok::<(), anyhow::Error>(())
232            })?;
233        }
234
235        Ok(files)
236    } else {
237        info!("No input directory provided, skipping markdown processing");
238        Ok(Vec::new())
239    }
240}

253/// Create a fallback index page
254#[must_use]
255pub fn create_fallback_index(config: &Config, markdown_files: &[std::path::PathBuf]) -> String {
256    let mut content = format!(
257        "<h1>{}</h1>\n<p>This is a fallback page created by ndg.</p>",
258        &config.title
259    );
260
261    // Add file listing if we have an input directory
262    if let Some(input_dir) = &config.input_dir {
263        if !markdown_files.is_empty() {
264            let mut file_list = String::with_capacity(markdown_files.len() * 100); // Preallocate based on estimated size
265            file_list.push_str("<h2>Available Documents</h2>\n<ul>\n");
266
267            for file_path in markdown_files {
268                if let Ok(rel_path) = file_path.strip_prefix(input_dir) {
269                    let mut html_path = rel_path.to_path_buf();
270                    html_path.set_extension("html");
271
272                    // Get page title from first heading or filename
273                    let page_title = extract_page_title(file_path, &html_path);
274
275                    writeln!(
276                        file_list,
277                        "  <li><a href=\"{}\">{}</a></li>",
278                        html_path.to_string_lossy(),
279                        page_title
280                    )
281                    .unwrap();
282                }
283            }
284
285            file_list.push_str("</ul>");
286            content.push_str(&file_list);
287        }
288    }
289
290    content
291}

293/// Create a markdown processor from ndg config
294#[must_use] pub fn create_processor_from_config(config: &Config) -> MarkdownProcessor {
295    let mut builder = MarkdownOptionsBuilder::new()
296        .gfm(true)
297        .highlight_code(config.highlight_code);
298
299    if let Some(mappings_path) = &config.manpage_urls_path {
300        builder = builder.manpage_urls_path(Some(mappings_path.to_string_lossy().to_string()));
301    }
302
303    MarkdownProcessor::new(builder.build())
304}

306/// Extract the page title from a markdown file
307#[must_use]
308pub fn extract_page_title(file_path: &std::path::Path, html_path: &std::path::Path) -> String {
309    let default_title = html_path
310        .file_stem()
311        .unwrap_or_default()
312        .to_string_lossy()
313        .to_string();
314
315    match fs::read_to_string(file_path) {
316        Ok(content) => {
317            ndg_commonmark::utils::extract_markdown_title(&content).unwrap_or(default_title)
318        }
319        Err(_) => default_title,
320    }
321}

323/// Ensure that index.html exists in the output directory
324pub fn ensure_index(
325    config: &Config,
326    options_processed: bool,
327    markdown_files: &[std::path::PathBuf],
328) -> Result<()> {
329    let index_path = config.output_dir.join("index.html");
330
331    // Check if index.html already exists (already generated from index.md)
332    if index_path.exists() {
333        return Ok(());
334    }
335
336    // If options were processed, try to use options.html as index
337    if options_processed {
338        let options_path = config.output_dir.join("options.html");
339        if options_path.exists() {
340            info!("Using options.html as index.html");
341
342            // Copy the content of options.html to index.html
343            let options_content = fs::read(&options_path)?;
344            fs::write(&index_path, options_content)?;
345            return Ok(());
346        }
347    }
348
349    // Create a fallback index page if input directory was provided but no index.md
350    // existed
351    if config.input_dir.is_some() {
352        info!("No index.md found, creating fallback index.html");
353
354        let content = create_fallback_index(config, markdown_files);
355
356        // Create a simple HTML document using our template system
357        let headers = vec![ndg_commonmark::Header {
358            text: config.title.clone(),
359            level: 1,
360            id: "welcome".to_string(),
361        }];
362
363        let html = html::template::render(
364            config,
365            &content,
366            &config.title,
367            &headers,
368            &std::path::PathBuf::from("index.html"),
369        )?;
370
371        fs::write(&index_path, html)?;
372    }
373
374    Ok(())

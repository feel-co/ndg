19/// Copy assets to output directory
20pub fn copy_assets(config: &Config) -> Result<()> {
21    // Create assets directory
22    let assets_dir = config.output_dir.join("assets");
23    fs::create_dir_all(&assets_dir)?;
24
25    // Generate and write CSS
26    let css = generate_css(config)?;
27    fs::write(assets_dir.join("style.css"), css).context("Failed to write CSS file")?;
28
29    // Copy main.js, which is always needed for the default templates
30    copy_template_asset(config, &assets_dir, "main.js", MAIN_JS)?;
31
32    // Copy custom assets if they exist
33    copy_custom_assets(config, &assets_dir)?;
34
35    // Copy script files to assets directory
36    copy_script_files(config, &assets_dir)?;
37
38    // Create search.js for search functionality
39    if config.generate_search {
40        copy_template_asset(config, &assets_dir, "search.js", SEARCH_JS)?;
41    }
42
43    Ok(())
44}

46/// Copies template asset while still allowing user override
47fn copy_template_asset(
48    config: &Config,
49    assets_dir: &Path,
50    filename: &str,
51    fallback_content: &str,
52) -> Result<()> {
53    let content = if let Some(path) = config.get_template_file(filename) {
54        if path.exists() {
55            fs::read_to_string(&path)
56                .with_context(|| format!("Failed to read {} from: {}", filename, path.display()))?
57        } else {
58            fallback_content.to_string()
59        }
60    } else {
61        fallback_content.to_string()
62    };
63
64    fs::write(assets_dir.join(filename), content)
65        .with_context(|| format!("Failed to write {filename} to assets directory"))
66}

68/// Copy custom assets from the configured directory
69fn copy_custom_assets(config: &Config, assets_dir: &Path) -> Result<()> {
70    if let Some(custom_assets_dir) = &config.assets_dir {
71        if custom_assets_dir.exists() && custom_assets_dir.is_dir() {
72            debug!("Copying custom assets from {}", custom_assets_dir.display());
73
74            let options = fs_extra::dir::CopyOptions::new().overwrite(true);
75            fs_extra::dir::copy(custom_assets_dir, assets_dir, &options)
76                .context("Failed to copy custom assets")?;
77        }
78    }
79    Ok(())
80}

82/// Copy script files to the assets directory
83fn copy_script_files(config: &Config, assets_dir: &Path) -> Result<()> {
84    for script_path in &config.script_paths {
85        if script_path.exists() {
86            let file_name = script_path
87                .file_name()
88                .ok_or_else(|| anyhow::anyhow!("Invalid script filename"))?;
89            let dest_path = assets_dir.join(file_name);
90
91            fs::read(script_path)
92                .with_context(|| format!("Failed to read script file {}", script_path.display()))
93                .and_then(|content| {
94                    fs::write(&dest_path, content).with_context(|| {
95                        format!("Failed to write script file to {}", dest_path.display())
96                    })
97                })?;
98        }
99    }
100    Ok(())
101}

103/// Generate CSS from stylesheet
104fn generate_css(config: &Config) -> Result<String> {
105    // Use template CSS if available, otherwise use default CSS
106    let mut combined_css = if let Some(template_path) = config.get_template_path() {
107        let template_css_path = template_path.join("default.css");
108        if template_css_path.exists() {
109            fs::read_to_string(&template_css_path).with_context(|| {
110                format!(
111                    "Failed to read template CSS: {}",
112                    template_css_path.display()
113                )
114            })?
115        } else {
116            // No template CSS found, use default
117            String::from(DEFAULT_CSS)
118        }
119    } else {
120        // No template directory specified, use default CSS
121        String::from(DEFAULT_CSS)
122    };
123
124    // Add any custom stylesheets provided via --stylesheet
125    if !config.stylesheet_paths.is_empty() {
126        // Process each stylesheet in order
127        for (index, stylesheet_path) in config.stylesheet_paths.iter().enumerate() {
128            if stylesheet_path.exists() {
129                let content = fs::read_to_string(stylesheet_path).with_context(|| {
130                    format!(
131                        "Failed to read stylesheet {}: {}",
132                        index + 1,
133                        stylesheet_path.display()
134                    )
135                })?;
136
137                // Process SCSS if needed
138                let processed_content = if stylesheet_path
139                    .extension()
140                    .is_some_and(|ext| ext == "scss")
141                {
142                    grass::from_string(content.clone(), &grass::Options::default()).with_context(
143                        || format!("Failed to compile SCSS to CSS for stylesheet {}", index + 1),
144                    )?
145                } else {
146                    content
147                };
148
149                // Add a comment to separate multiple stylesheets
150                combined_css.push_str("\n\n/* Custom Stylesheet ");
151                combined_css.push_str(&(index + 1).to_string());
152                combined_css.push_str(": ");
153                combined_css.push_str(&stylesheet_path.display().to_string());
154                combined_css.push_str(" */\n");
155                combined_css.push_str(&processed_content);
156            }
157        }
158    }
159
160    Ok(combined_css)
161}

195/// Process markdown files
196pub fn process_markdown_files(config: &Config) -> Result<Vec<std::path::PathBuf>> {
197    if let Some(ref input_dir) = config.input_dir {
198        info!("Input directory: {}", input_dir.display());
199        let files = collect_markdown_files(input_dir);
200        info!("Found {} markdown files", files.len());
201
202        if !files.is_empty() {
203            // Process files in parallel
204            use rayon::prelude::*;
205
206            use crate::html::template;
207            files.par_iter().try_for_each(|file_path| {
208                let content = std::fs::read_to_string(file_path)?;
209                let mut options = ndg_commonmark::processor::MarkdownOptions::default();
210                if let Some(mappings_path) = &config.manpage_urls_path {
211                    options.manpage_urls_path = Some(mappings_path.to_string_lossy().to_string());
212                }
213                // Configure syntax highlighting based on config
214                options.highlight_code = config.highlight_code;
215                let processor = ndg_commonmark::processor::MarkdownProcessor::new(options);
216                let result = processor.render(&content);
217
218                // The HTML already includes syntax highlighting
219                let highlighted_html = result.html;
220
221                let headers = result.headers;
222                let title = result.title;
223                let input_dir = config.input_dir.as_ref().expect("input_dir required");
224                let rel_path = file_path
225                    .strip_prefix(input_dir)
226                    .expect("strip_prefix failed");
227                let mut output_path = config.output_dir.join(rel_path);
228                output_path.set_extension("html");
229
230                let html = template::render(
231                    config,
232                    &highlighted_html,
233                    title.as_deref().unwrap_or(&config.title),
234                    &headers,
235                    rel_path,
236                )?;
237
238                if let Some(parent) = output_path.parent() {
239                    std::fs::create_dir_all(parent)?;
240                }
241                std::fs::write(&output_path, html)?;
242                Ok::<(), anyhow::Error>(())
243            })?;
244        }
245
246        Ok(files)
247    } else {
248        info!("No input directory provided, skipping markdown processing");
249        Ok(Vec::new())
250    }
251}

264/// Create a fallback index page
265#[must_use]
266pub fn create_fallback_index(config: &Config, markdown_files: &[std::path::PathBuf]) -> String {
267    let mut content = format!(
268        "<h1>{}</h1>\n<p>This is a fallback page created by ndg.</p>",
269        &config.title
270    );
271
272    // Add file listing if we have an input directory
273    if let Some(input_dir) = &config.input_dir {
274        if !markdown_files.is_empty() {
275            let mut file_list = String::with_capacity(markdown_files.len() * 100); // Preallocate based on estimated size
276            file_list.push_str("<h2>Available Documents</h2>\n<ul>\n");
277
278            for file_path in markdown_files {
279                if let Ok(rel_path) = file_path.strip_prefix(input_dir) {
280                    let mut html_path = rel_path.to_path_buf();
281                    html_path.set_extension("html");
282
283                    // Get page title from first heading or filename
284                    let page_title = extract_page_title(file_path, &html_path);
285
286                    writeln!(
287                        file_list,
288                        "  <li><a href=\"{}\">{}</a></li>",
289                        html_path.to_string_lossy(),
290                        page_title
291                    )
292                    .unwrap();
293                }
294            }
295
296            file_list.push_str("</ul>");
297            content.push_str(&file_list);
298        }
299    }
300
301    content
302}

304/// Extract the page title from a markdown file
305#[must_use]
306pub fn extract_page_title(file_path: &std::path::Path, html_path: &std::path::Path) -> String {
307    let default_title = html_path
308        .file_stem()
309        .unwrap_or_default()
310        .to_string_lossy()
311        .to_string();
312
313    match fs::read_to_string(file_path) {
314        Ok(content) => {
315            ndg_commonmark::utils::extract_markdown_title(&content).unwrap_or(default_title)
316        }
317        Err(_) => default_title,
318    }
319}

321/// Ensure that index.html exists in the output directory
322pub fn ensure_index(
323    config: &Config,
324    options_processed: bool,
325    markdown_files: &[std::path::PathBuf],
326) -> Result<()> {
327    let index_path = config.output_dir.join("index.html");
328
329    // Check if index.html already exists (already generated from index.md)
330    if index_path.exists() {
331        return Ok(());
332    }
333
334    // If options were processed, try to use options.html as index
335    if options_processed {
336        let options_path = config.output_dir.join("options.html");
337        if options_path.exists() {
338            info!("Using options.html as index.html");
339
340            // Copy the content of options.html to index.html
341            let options_content = fs::read(&options_path)?;
342            fs::write(&index_path, options_content)?;
343            return Ok(());
344        }
345    }
346
347    // Create a fallback index page if input directory was provided but no index.md
348    // existed
349    if config.input_dir.is_some() {
350        info!("No index.md found, creating fallback index.html");
351
352        let content = create_fallback_index(config, markdown_files);
353
354        // Create a simple HTML document using our template system
355        let headers = vec![markdown::Header {
356            text: config.title.clone(),
357            level: 1,
358            id: "welcome".to_string(),
359        }];
360
361        let html = html::template::render(
362            config,
363            &content,
364            &config.title,
365            &headers,
366            &std::path::PathBuf::from("index.html"),
367        )?;
368
369        fs::write(&index_path, html)?;
370    }
371
372    Ok(())

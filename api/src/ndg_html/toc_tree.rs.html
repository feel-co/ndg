48fn insert_path(
49  children: &mut Vec<TocNode>,
50  path: &[TocPathPart],
51  leaf_html: String,
52  open_path: bool,
53) {
54  let Some((part, rest)) = path.split_first() else {
55    return;
56  };
57  let index = children
58    .iter()
59    .position(|node| node.key == part.key)
60    .unwrap_or_else(|| {
61      children.push(TocNode {
62        key: part.key.clone(),
63        label: part.label.clone(),
64        title: part.title.clone(),
65        ..Default::default()
66      });
67      children.len() - 1
68    });
69  let node = &mut children[index];
70  node.label.clone_from(&part.label);
71  node.title.clone_from(&part.title);
72  node.open |= open_path;
73
74  if rest.is_empty() {
75    node.leaf_html = Some(leaf_html);
76  } else {
77    insert_path(&mut node.children, rest, leaf_html, open_path);
78  }
79}

81fn render_node(
82  html: &mut String,
83  node: &TocNode,
84  indent: usize,
85  default_open_depth: usize,
86) {
87  let spaces = " ".repeat(indent);
88  if node.children.is_empty() {
89    if let Some(leaf_html) = &node.leaf_html {
90      let _ = writeln!(html, "{spaces}<li>{leaf_html}</li>");
91    }
92    return;
93  }
94
95  let safe_label = encode_text(&node.label);
96  let safe_title = encode_text(&node.title);
97  let count = count_leaves(node);
98  let open = if indent <= default_open_depth || node.open {
99    " open"
100  } else {
101    ""
102  };
103  let _ = writeln!(
104    html,
105    "{spaces}<li>\n{spaces}  <details \
106     class=\"toc-category\"{open}>\n{spaces}    <summary \
107     title=\"{safe_title}\"><span>{safe_label}</span><span \
108     class=\"toc-count\">{count}</span></summary>\n{spaces}    <ul>"
109  );
110
111  if let Some(leaf_html) = &node.leaf_html {
112    let _ = writeln!(html, "{spaces}      <li>{leaf_html}</li>");
113  }
114
115  for child in &node.children {
116    render_node(html, child, indent + 6, default_open_depth);
117  }
118
119  let _ = writeln!(
120    html,
121    "{spaces}    </ul>\n{spaces}  </details>\n{spaces}</li>"
122  );
123}

125fn count_leaves(node: &TocNode) -> usize {
126  usize::from(node.leaf_html.is_some())
127    + node.children.iter().map(count_leaves).sum::<usize>()

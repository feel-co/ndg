27/// Render a documentation page
28///
29/// # Errors
30///
31/// Returns an error if the template cannot be rendered or written.
32pub fn render(
33  config: &Config,
34  content: &str,
35  title: &str,
36  headers: &[Header],
37  rel_path: &Path,
38) -> Result<String> {
39  let mut tera = Tera::default();
40
41  // Check for file-specific template (e.g., foo.html for foo.md)
42  let file_stem = rel_path
43    .file_stem()
44    .and_then(|s| s.to_str())
45    .unwrap_or("default");
46  let specific_template_name = format!("{file_stem}.html");
47
48  // Try to load file-specific template first, then fall back to default
49  let template_content = if let Ok(specific_content) = get_template_content(
50    config,
51    &specific_template_name,
52    "", // no fallback for specific templates
53  ) {
54    if specific_content.is_empty() {
55      get_template_content(config, "default.html", DEFAULT_TEMPLATE)?
56    } else {
57      specific_content
58    }
59  } else {
60    get_template_content(config, "default.html", DEFAULT_TEMPLATE)?
61  };
62
63  tera.add_raw_template("default", &template_content)?;
64
65  // Load navbar template
66  let navbar_content =
67    get_template_content(config, "navbar.html", NAVBAR_TEMPLATE)?;
68  tera.add_raw_template("navbar", &navbar_content)?;
69
70  // Load footer template
71  let footer_content =
72    get_template_content(config, "footer.html", FOOTER_TEMPLATE)?;
73  tera.add_raw_template("footer", &footer_content)?;
74
75  // Generate table of contents from headers
76  let toc = generate_toc(headers);
77
78  // Generate document navigation
79  let doc_nav = generate_doc_nav(config, rel_path);
80
81  // Check if options are available
82  let has_options = if config.module_options.is_some() {
83    ""
84  } else {
85    "style=\"display:none;\""
86  };
87
88  let custom_scripts = generate_custom_scripts(config, rel_path)?;
89  let asset_paths = generate_asset_paths(rel_path);
90  let root_prefix = calculate_root_relative_path(rel_path);
91
92  // Prepare meta tags HTML
93  let meta_tags_html =
94    get_meta_tags(config).map_or_else(String::new, |meta_tags| {
95      meta_tags
96        .iter()
97        .map(|(k, v)| format!(r#"<meta name="{k}" content="{v}" />"#))
98        .collect::<Vec<_>>()
99        .join("\n    ")
100    });
101
102  // Prepare OpenGraph tags HTML, handling og:image as local path or URL
103  #[allow(
104    clippy::option_if_let_else,
105    reason = "Complex image handling logic clearer with if-let"
106  )]
107  let opengraph_html = if let Some(opengraph) = get_opengraph(config) {
108    opengraph
109      .iter()
110      .map(|(k, v)| {
111        if k == "og:image"
112          && !v.starts_with("http://")
113          && !v.starts_with("https://")
114        {
115          // Local file path: copy to assets/ and use relative path
116          let image_path = std::path::Path::new(v.as_str());
117          let assets_dir = config.output_dir.join("assets");
118          let file_name = image_path
119            .file_name()
120            .and_then(|f| f.to_str())
121            .unwrap_or(v.as_str());
122          let dest_path = assets_dir.join(file_name);
123          // Only copy if not already present
124          if let Err(e) = std::fs::create_dir_all(&assets_dir) {
125            log::error!("Failed to create assets dir for og:image: {e}");
126          }
127          if let Err(e) = std::fs::copy(image_path, &dest_path) {
128            log::error!(
129              "Failed to copy og:image from {} to {}: {e}",
130              v,
131              dest_path.display()
132            );
133          }
134          let rel_path = format!("assets/{file_name}");
135          format!(r#"<meta property="{k}" content="{rel_path}" />"#)
136        } else {
137          format!(r#"<meta property="{k}" content="{v}" />"#)
138        }
139      })
140      .collect::<Vec<_>>()
141      .join("\n    ")
142  } else {
143    String::new()
144  };
145
146  // Render navbar and footer
147  let navbar_html = tera.render("navbar", &{
148    let mut ctx = tera::Context::new();
149    ctx.insert("has_options", has_options);
150    ctx.insert("generate_search", &config.is_search_enabled());
151    ctx.insert(
152      "options_path",
153      asset_paths
154        .get("options_path")
155        .map_or("options.html", std::string::String::as_str),
156    );
157    ctx.insert(
158      "search_path",
159      asset_paths
160        .get("search_path")
161        .map_or("search.html", std::string::String::as_str),
162    );
163    ctx
164  })?;
165
166  let footer_html = tera.render("footer", &{
167    let mut ctx = tera::Context::new();
168    ctx.insert("footer_text", &config.footer_text);
169    ctx
170  })?;
171
172  // Create context
173  let mut tera_context = tera::Context::new();
174  tera_context.insert("content", content);
175  tera_context.insert("title", title);
176  tera_context.insert("site_title", &config.title);
177  tera_context.insert("footer_text", &config.footer_text);
178  tera_context.insert("navbar_html", &navbar_html);
179  tera_context.insert("footer_html", &footer_html);
180  tera_context.insert("toc", &toc);
181  tera_context.insert("doc_nav", &doc_nav);
182  tera_context.insert("has_options", has_options);
183  tera_context.insert("custom_scripts", &custom_scripts);
184  tera_context.insert("generate_search", &config.is_search_enabled());
185  tera_context.insert("meta_tags_html", &meta_tags_html);
186  tera_context.insert("opengraph_html", &opengraph_html);
187
188  // Populate Tera context with asset paths
189  tera_context.insert(
190    "stylesheet_path",
191    asset_paths
192      .get("stylesheet_path")
193      .map_or("assets/style.css", String::as_str),
194  );
195  tera_context.insert(
196    "main_js_path",
197    asset_paths
198      .get("main_js_path")
199      .map_or("assets/main.js", String::as_str),
200  );
201  tera_context.insert(
202    "search_js_path",
203    asset_paths
204      .get("search_js_path")
205      .map_or("assets/search.js", String::as_str),
206  );
207  tera_context.insert(
208    "index_path",
209    asset_paths
210      .get("index_path")
211      .map_or("index.html", String::as_str),
212  );
213  tera_context.insert(
214    "options_path",
215    asset_paths
216      .get("options_path")
217      .map_or("options.html", String::as_str),
218  );
219  tera_context.insert(
220    "search_path",
221    asset_paths
222      .get("search_path")
223      .map_or("search.html", String::as_str),
224  );
225  tera_context.insert("root_prefix", &root_prefix);
226
227  // Render the template
228  let html = tera.render("default", &tera_context)?;
229  Ok(html)
230}

390/// Generate specialized TOC for options page
391fn generate_options_toc(
392  options: &HashMap<String, NixOption>,
393  config: &Config,
394  tera: &Tera,
395) -> Result<String> {
396  // Get depth from sidebar.options config or fallback to legacy
397  // options_toc_depth
398  let default_depth = config
399    .sidebar
400    .as_ref()
401    .and_then(|s| s.options.as_ref())
402    .map_or(config.options_toc_depth, |o| o.depth);
403
404  let mut grouped_options: HashMap<String, Vec<&NixOption>> = HashMap::new();
405  let mut direct_parent_options: HashMap<String, &NixOption> = HashMap::new();
406  let mut option_custom_names: HashMap<String, String> = HashMap::new();
407  let mut option_positions: HashMap<String, usize> = HashMap::new();
408
409  for option in options.values() {
410    // Check if this option has a matching rule in sidebar.options config
411    let match_result = config
412      .sidebar
413      .as_ref()
414      .and_then(|s| s.options.as_ref())
415      .and_then(|o| o.find_match(&option.name));
416
417    // Skip if option is marked as hidden
418    if let Some(matched) = &match_result {
419      if matched.is_hidden() {
420        continue;
421      }
422
423      // Store custom name if provided
424      if let Some(name) = matched.get_name() {
425        option_custom_names.insert(option.name.clone(), name.to_string());
426      }
427
428      // Store custom position if provided
429      if let Some(position) = matched.get_position() {
430        option_positions.insert(option.name.clone(), position);
431      }
432    }
433
434    // Use custom depth if specified, otherwise use default
435    let depth = match_result
436      .as_ref()
437      .and_then(|m| m.get_depth())
438      .unwrap_or(default_depth);
439
440    let parent = get_option_parent(&option.name, depth);
441
442    // Check if this option exactly matches its parent category
443    if option.name == parent {
444      direct_parent_options.insert(parent.clone(), option);
445    }
446
447    // Add to grouped options
448    grouped_options.entry(parent).or_default().push(option);
449  }
450
451  // Separate categories into single options and dropdown categories
452  let mut single_options: Vec<tera::Value> = Vec::new();
453  let mut dropdown_categories: Vec<tera::Value> = Vec::new();
454
455  for (parent, opts) in &grouped_options {
456    let has_multiple_options = opts.len() > 1;
457    let has_child_options =
458      opts.len() > usize::from(direct_parent_options.contains_key(parent));
459
460    if !has_multiple_options && !has_child_options {
461      // Single option with no children
462      let option = opts[0];
463
464      // Use custom name if available, otherwise use option name
465      #[allow(clippy::map_unwrap_or)]
466      let display_name = option_custom_names
467        .get(&option.name)
468        .map(String::as_str)
469        .unwrap_or(&option.name);
470
471      let option_value = tera::to_value({
472        let mut map = tera::Map::new();
473        map.insert("name".to_string(), tera::to_value(&option.name)?);
474        map.insert("display_name".to_string(), tera::to_value(display_name)?);
475        map.insert("internal".to_string(), tera::to_value(option.internal)?);
476        map.insert("read_only".to_string(), tera::to_value(option.read_only)?);
477
478        // Add position if custom position is set
479        if let Some(position) = option_positions.get(&option.name) {
480          map.insert("position".to_string(), tera::to_value(position)?);
481        }
482
483        map
484      })?;
485      single_options.push(option_value);
486    } else {
487      // Category with multiple options or child options
488      let mut category = tera::Map::new();
489
490      // Use custom name for category if the parent option has one
491      #[allow(clippy::map_unwrap_or)]
492      let category_display_name = option_custom_names
493        .get(parent)
494        .map(String::as_str)
495        .unwrap_or(parent);
496
497      category.insert("name".to_string(), tera::to_value(parent)?);
498      category.insert(
499        "display_name".to_string(),
500        tera::to_value(category_display_name)?,
501      );
502      category.insert("count".to_string(), tera::to_value(opts.len())?);
503
504      // Add parent option if it exists
505      if let Some(parent_option) = direct_parent_options.get(parent) {
506        let parent_option_value = tera::to_value({
507          let mut map = tera::Map::new();
508          map.insert("name".to_string(), tera::to_value(&parent_option.name)?);
509          map.insert(
510            "internal".to_string(),
511            tera::to_value(parent_option.internal)?,
512          );
513          map.insert(
514            "read_only".to_string(),
515            tera::to_value(parent_option.read_only)?,
516          );
517          map
518        })?;
519        category.insert("parent_option".to_string(), parent_option_value);
520      }
521
522      // Add child options
523      let mut children = Vec::new();
524      let mut child_options: Vec<&NixOption> = opts
525        .iter()
526        .filter(|opt| opt.name != *parent)
527        .copied()
528        .collect();
529
530      // Sort by suffix
531      child_options.sort_by(|a, b| {
532        let a_suffix = a
533          .name
534          .strip_prefix(&format!("{parent}."))
535          .unwrap_or(&a.name);
536        let b_suffix = b
537          .name
538          .strip_prefix(&format!("{parent}."))
539          .unwrap_or(&b.name);
540        a_suffix.cmp(b_suffix)
541      });
542
543      for option in child_options {
544        let display_name = option
545          .name
546          .strip_prefix(&format!("{parent}."))
547          .unwrap_or(&option.name);
548
549        let child_value = tera::to_value({
550          let mut map = tera::Map::new();
551          map.insert("name".to_string(), tera::to_value(&option.name)?);
552          map.insert("display_name".to_string(), tera::to_value(display_name)?);
553          map.insert("internal".to_string(), tera::to_value(option.internal)?);
554          map
555            .insert("read_only".to_string(), tera::to_value(option.read_only)?);
556          map
557        })?;
558        children.push(child_value);
559      }
560
561      category.insert("children".to_string(), tera::to_value(children)?);
562
563      // Add position if custom position is set for parent
564      if let Some(position) = option_positions.get(parent) {
565        category.insert("position".to_string(), tera::to_value(position)?);
566      }
567
568      dropdown_categories.push(tera::to_value(category)?);
569    }
570  }
571
572  // Sort single options - by position first if available, then alphabetically
573  single_options.sort_by(|a, b| {
574    let a_name = a.get("name").and_then(|v| v.as_str()).unwrap_or("");
575    let b_name = b.get("name").and_then(|v| v.as_str()).unwrap_or("");
576    let a_position = a.get("position").and_then(Value::as_u64);
577    let b_position = b.get("position").and_then(Value::as_u64);
578
579    match (a_position, b_position) {
580      (Some(a_pos), Some(b_pos)) => a_pos.cmp(&b_pos),
581      (Some(_), None) => std::cmp::Ordering::Less,
582      (None, Some(_)) => std::cmp::Ordering::Greater,
583      (None, None) => a_name.cmp(b_name),
584    }
585  });
586
587  // Sort dropdown categories - by position first if available, then by
588  // component count and alphabetically
589  dropdown_categories.sort_by(|a, b| {
590    let a_name = a.get("name").and_then(|v| v.as_str()).unwrap_or("");
591    let b_name = b.get("name").and_then(|v| v.as_str()).unwrap_or("");
592    let a_position = a.get("position").and_then(Value::as_u64);
593    let b_position = b.get("position").and_then(Value::as_u64);
594
595    match (a_position, b_position) {
596      (Some(a_pos), Some(b_pos)) => a_pos.cmp(&b_pos),
597      (Some(_), None) => std::cmp::Ordering::Less,
598      (None, Some(_)) => std::cmp::Ordering::Greater,
599      (None, None) => {
600        let a_components = a_name.split('.').count();
601        let b_components = b_name.split('.').count();
602
603        // Sort by component count first
604        match a_components.cmp(&b_components) {
605          std::cmp::Ordering::Equal => a_name.cmp(b_name), // then alphabetically
606          other => other,
607        }
608      }
609    }
610  });
611
612  let mut tera_context = tera::Context::new();
613  tera_context.insert("single_options", &single_options);
614  tera_context.insert("dropdown_categories", &dropdown_categories);
615
616  // Render the template
617  let rendered = tera.render("options_toc", &tera_context)?;
618
619  Ok(rendered)
620}

622/// Extract the parent category from an option name with configurable depth
623fn get_option_parent(option_name: &str, depth: usize) -> String {
624  let parts: Vec<&str> = option_name.split('.').collect();
625  if parts.len() <= depth {
626    option_name.to_string()
627  } else {
628    parts[..depth].join(".")
629  }
630}

632/// Render search page
633///
634/// # Errors
635///
636/// Returns an error if the search template or any required template cannot be
637/// rendered or written.
638#[allow(
639  clippy::implicit_hasher,
640  reason = "Standard HashMap sufficient for this use case"
641)]
642pub fn render_search(
643  config: &Config,
644  context: &HashMap<&str, String>,
645) -> Result<String> {
646  // Skip rendering if search is disabled
647  if !config.is_search_enabled() {
648    bail!("Search functionality is disabled");
649  }
650
651  let mut tera = Tera::default();
652  let search_template =
653    get_template_content(config, "search.html", SEARCH_TEMPLATE)?;
654  tera.add_raw_template("search", &search_template)?;
655
656  // Load navbar template
657  let navbar_content =
658    get_template_content(config, "navbar.html", NAVBAR_TEMPLATE)?;
659  tera.add_raw_template("navbar", &navbar_content)?;
660
661  // Load footer template
662  let footer_content =
663    get_template_content(config, "footer.html", FOOTER_TEMPLATE)?;
664  tera.add_raw_template("footer", &footer_content)?;
665
666  let title_str = context
667    .get("title")
668    .cloned()
669    .unwrap_or_else(|| format!("{} - Search", config.title));
670
671  // Generate document navigation for root level (system-generated search.html
672  // is always at root)
673  let root_path = Path::new("search.html");
674  let doc_nav = generate_doc_nav(config, root_path);
675
676  // Generate custom scripts HTML for root level
677  let custom_scripts = generate_custom_scripts(config, root_path)?;
678
679  // Check if options are available
680  let has_options = if config.module_options.is_some() {
681    ""
682  } else {
683    "style=\"display:none;\""
684  };
685
686  // Generate asset and navigation paths (search page is at root)
687  let asset_paths = ndg_utils::html::generate_asset_paths(root_path);
688
689  // Calculate root prefix for JavaScript path resolution (search page is at
690  // root)
691  let root_prefix = ndg_utils::html::calculate_root_relative_path(root_path);
692
693  // Render navbar and footer
694  let navbar_html = tera.render("navbar", &{
695    let mut ctx = tera::Context::new();
696    ctx.insert("has_options", has_options);
697    ctx.insert("generate_search", &config.is_search_enabled());
698    ctx.insert(
699      "options_path",
700      asset_paths
701        .get("options_path")
702        .map_or("options.html", std::string::String::as_str),
703    );
704    ctx.insert(
705      "search_path",
706      asset_paths
707        .get("search_path")
708        .map_or("search.html", std::string::String::as_str),
709    );
710    ctx
711  })?;
712
713  let footer_html = tera.render("footer", &{
714    let mut ctx = tera::Context::new();
715    ctx.insert("footer_text", &config.footer_text);
716    ctx
717  })?;
718
719  // Create Tera context
720  let mut tera_context = tera::Context::new();
721  tera_context.insert("title", &title_str);
722  tera_context.insert("site_title", &config.title);
723  tera_context.insert("heading", "Search");
724  tera_context.insert("footer_text", &config.footer_text);
725  tera_context.insert("navbar_html", &navbar_html);
726  tera_context.insert("footer_html", &footer_html);
727  tera_context.insert("custom_scripts", &custom_scripts);
728  tera_context.insert("doc_nav", &doc_nav);
729  tera_context.insert("has_options", has_options);
730  tera_context.insert("toc", ""); // no TOC for search page
731  tera_context.insert("generate_search", &true); // always true for search page
732
733  // Add meta and opengraph tags
734  let meta_tags_html =
735    get_meta_tags(config).map_or_else(String::new, |meta_tags| {
736      meta_tags
737        .iter()
738        .map(|(k, v)| format!(r#"<meta name="{k}" content="{v}" />"#))
739        .collect::<Vec<_>>()
740        .join("\n    ")
741    });
742
743  let opengraph_html =
744    get_opengraph(config).map_or_else(String::new, |opengraph| {
745      opengraph
746        .iter()
747        .map(|(k, v)| format!(r#"<meta property="{k}" content="{v}" />"#))
748        .collect::<Vec<_>>()
749        .join("\n    ")
750    });
751  tera_context.insert("meta_tags_html", &meta_tags_html);
752  tera_context.insert("opengraph_html", &opengraph_html);
753
754  // Add asset paths with fallback values in case keys are missing
755  tera_context.insert(
756    "stylesheet_path",
757    asset_paths
758      .get("stylesheet_path")
759      .map_or("assets/style.css", std::string::String::as_str),
760  );
761  tera_context.insert(
762    "main_js_path",
763    asset_paths
764      .get("main_js_path")
765      .map_or("assets/main.js", std::string::String::as_str),
766  );
767  tera_context.insert(
768    "search_js_path",
769    asset_paths
770      .get("search_js_path")
771      .map_or("assets/search.js", std::string::String::as_str),
772  );
773  tera_context.insert(
774    "index_path",
775    asset_paths
776      .get("index_path")
777      .map_or("index.html", std::string::String::as_str),
778  );
779  tera_context.insert(
780    "options_path",
781    asset_paths
782      .get("options_path")
783      .map_or("options.html", std::string::String::as_str),
784  );
785  tera_context.insert(
786    "search_path",
787    asset_paths
788      .get("search_path")
789      .map_or("search.html", std::string::String::as_str),
790  );
791  tera_context.insert("root_prefix", &root_prefix);
792
793  // Render the template
794  let html = tera.render("search", &tera_context)?;
795  Ok(html)
796}

798/// Get the template content from file in template directory or use default
799fn get_template_content(
800  config: &Config,
801  template_name: &str,
802  fallback: &str,
803) -> Result<String> {
804  // Try to get the template from the configured template directory
805  if let Some(template_dir) = config.get_template_path() {
806    let template_path = template_dir.join(template_name);
807    if template_path.exists() {
808      return fs::read_to_string(&template_path).wrap_err({
809        format!("Failed to read template file: {}", template_path.display())
810      });
811    }
812  }
813  // If template_path is specified but doesn't point to a directory with our
814  // template
815  if let Some(template_path) = &config.template_path
816    && template_path.exists()
817    && template_name == "default.html"
818  {
819    // XXX: For backward compatibility
820    // If template_path is a file, use it for default.html
821    return fs::read_to_string(template_path).wrap_err({
822      format!(
823        "Failed to read custom template file: {}. Check file permissions and \
824         ensure the file is valid UTF-8",
825        template_path.display()
826      )
827    });
828  }
829  // Use fallback embedded template if no custom template found
830  // If fallback is empty, return empty string (used for optional templates)
831  Ok(fallback.to_string())
832}

834/// Represents a navigation item with its metadata for rendering
835struct NavItem {
836  path:     String,
837  title:    String,
838  position: Option<usize>,
839  number:   Option<usize>,
840}

842/// Extract page title from markdown file.
843///
844/// First attempts to read the file and extract the title from the first H1
845/// heading. Falls back to using the file stem as the title if reading fails or
846/// no H1 is found.
847fn extract_page_title(path: &Path, html_path: &Path) -> String {
848  std::fs::read_to_string(path).map_or_else(
849    |_| {
850      html_path
851        .file_stem()
852        .unwrap_or_default()
853        .to_string_lossy()
854        .to_string()
855    },
856    |content| {
857      content.lines().next().map_or_else(
858        || {
859          html_path
860            .file_stem()
861            .unwrap_or_default()
862            .to_string_lossy()
863            .to_string()
864        },
865        |first_line| {
866          first_line.strip_prefix("# ").map_or_else(
867            || {
868              html_path
869                .file_stem()
870                .unwrap_or_default()
871                .to_string_lossy()
872                .to_string()
873            },
874            ndg_commonmark::utils::clean_anchor_patterns,
875          )
876        },
877      )
878    },
879  )
880}

882/// Generate the document navigation HTML
883fn generate_doc_nav(config: &Config, current_file_rel_path: &Path) -> String {
884  let mut doc_nav = String::new();
885  let root_prefix = calculate_root_relative_path(current_file_rel_path);
886
887  // Only process markdown files if input_dir is provided
888  if let Some(input_dir) = &config.input_dir {
889    let entries: Vec<_> = walkdir::WalkDir::new(input_dir)
890      .follow_links(true)
891      .into_iter()
892      .filter_map(std::result::Result::ok)
893      .filter(|e| {
894        if !e.path().is_file()
895          || e.path().extension().is_none_or(|ext| ext != "md")
896        {
897          return false;
898        }
899
900        // Filter out excluded files (included files)
901        e.path()
902          .strip_prefix(input_dir)
903          .is_ok_and(|rel_path| !config.included_files.contains_key(rel_path))
904      })
905      .collect();
906
907    if !entries.is_empty() {
908      // Partition entries into special and regular
909      let mut special_entries = Vec::new();
910      let mut regular_entries = Vec::new();
911
912      for entry in &entries {
913        let path = entry.path();
914        if let Ok(rel_doc_path) = path.strip_prefix(input_dir) {
915          let file_name = rel_doc_path
916            .file_name()
917            .and_then(|n| n.to_str())
918            .unwrap_or("")
919            .to_ascii_lowercase();
920          if file_name == "index.md" || file_name == "readme.md" {
921            special_entries.push(entry);
922          } else {
923            regular_entries.push(entry);
924          }
925        }
926      }
927
928      // Process regular entries with sidebar configuration
929      let mut nav_items: Vec<NavItem> = regular_entries
930        .iter()
931        .filter_map(|entry| {
932          let path = entry.path();
933          let rel_doc_path = path.strip_prefix(input_dir).ok()?;
934          let mut html_path = rel_doc_path.to_path_buf();
935          html_path.set_extension("html");
936
937          let target_path =
938            format!("{}{}", root_prefix, html_path.to_string_lossy());
939
940          // Extract page title
941          let page_title = extract_page_title(path, &html_path);
942
943          // Apply sidebar configuration if available
944          let (display_title, position) =
945            if let Some(sidebar_config) = &config.sidebar {
946              let path_str = rel_doc_path.to_string_lossy();
947              if let Some(matched_rule) =
948                sidebar_config.find_match(&path_str, &page_title)
949              {
950                let title = matched_rule
951                  .get_title()
952                  .map_or_else(|| page_title.clone(), String::from);
953                let pos = matched_rule.get_position();
954                (title, pos)
955              } else {
956                (page_title, None)
957              }
958            } else {
959              (page_title, None)
960            };
961
962          Some(NavItem {
963            path: target_path,
964            title: display_title,
965            position,
966            number: None,
967          })
968        })
969        .collect();
970
971      // Sort items based on sidebar ordering configuration
972      if let Some(sidebar_config) = &config.sidebar {
973        use ndg_config::sidebar::SidebarOrdering;
974        match sidebar_config.ordering {
975          SidebarOrdering::Alphabetical => {
976            nav_items.sort_by(|a, b| a.title.cmp(&b.title));
977          },
978          SidebarOrdering::Custom => {
979            // Sort by position first, then alphabetically
980            nav_items.sort_by(|a, b| {
981              match (a.position, b.position) {
982                (Some(pos_a), Some(pos_b)) => pos_a.cmp(&pos_b),
983                (Some(_), None) => std::cmp::Ordering::Less,
984                (None, Some(_)) => std::cmp::Ordering::Greater,
985                (None, None) => a.title.cmp(&b.title),
986              }
987            });
988          },
989          SidebarOrdering::Filesystem => {
990            // Keep filesystem order (already in order thanks to walkdir)
991          },
992        }
993
994        // Add numbering if enabled
995        if sidebar_config.numbered {
996          for (idx, item) in nav_items.iter_mut().enumerate() {
997            item.number = Some(idx + 1);
998          }
999        }
1000      } else {
1001        // Default: alphabetical sorting
1002        nav_items.sort_by(|a, b| a.title.cmp(&b.title));
1003      }
1004
1005      // Process special entries
1006      let mut special_nav_items: Vec<NavItem> = special_entries
1007        .iter()
1008        .filter_map(|entry| {
1009          let path = entry.path();
1010          let rel_doc_path = path.strip_prefix(input_dir).ok()?;
1011          let mut html_path = rel_doc_path.to_path_buf();
1012          html_path.set_extension("html");
1013
1014          let target_path =
1015            format!("{}{}", root_prefix, html_path.to_string_lossy());
1016
1017          let page_title = extract_page_title(path, &html_path);
1018
1019          // Apply sidebar configuration to special files if available
1020          let display_title = if let Some(sidebar_config) = &config.sidebar {
1021            let path_str = rel_doc_path.to_string_lossy();
1022            if let Some(matched_rule) =
1023              sidebar_config.find_match(&path_str, &page_title)
1024            {
1025              matched_rule
1026                .get_title()
1027                .map_or_else(|| page_title.clone(), String::from)
1028            } else {
1029              page_title
1030            }
1031          } else {
1032            page_title
1033          };
1034
1035          Some(NavItem {
1036            path:     target_path,
1037            title:    display_title,
1038            position: None,
1039            number:   None,
1040          })
1041        })
1042        .collect();
1043
1044      // Sort special entries according to sidebar ordering configuration
1045      if let Some(sidebar_config) = &config.sidebar {
1046        match sidebar_config.ordering {
1047          SidebarOrdering::Alphabetical => {
1048            special_nav_items.sort_by(|a, b| a.title.cmp(&b.title));
1049          },
1050          SidebarOrdering::Custom => {
1051            // Custom ordering uses position from matches
1052            special_nav_items.sort_by_key(|item| item.position);
1053          },
1054          SidebarOrdering::Filesystem => {
1055            // Filesystem ordering keeps the order from directory iteration
1056          },
1057        }
1058      } else {
1059        // Default: alphabetical sorting
1060        special_nav_items.sort_by(|a, b| a.title.cmp(&b.title));
1061      }
1062
1063      // Determine if we should number special files
1064      let should_number_special = config
1065        .sidebar
1066        .as_ref()
1067        .is_some_and(|s| s.numbered && s.number_special_files);
1068
1069      // Render navigation items
1070      if should_number_special {
1071        // Combine special and regular items with unified numbering
1072        let mut all_items = special_nav_items;
1073        all_items.extend(nav_items);
1074
1075        // Apply numbering to all items
1076        for (idx, item) in all_items.iter_mut().enumerate() {
1077          item.number = Some(idx + 1);
1078        }
1079
1080        // Render all items
1081        for item in all_items {
1082          if let Some(num) = item.number {
1083            let _ = writeln!(
1084              doc_nav,
1085              "<li><a href=\"{}\">{num}. {}</a></li>",
1086              item.path, item.title
1087            );
1088          } else {
1089            let _ = writeln!(
1090              doc_nav,
1091              "<li><a href=\"{}\">{}</a></li>",
1092              item.path, item.title
1093            );
1094          }
1095        }
1096      } else {
1097        // Render special entries first without numbering
1098        for item in special_nav_items {
1099          let _ = writeln!(
1100            doc_nav,
1101            "<li><a href=\"{}\">{}</a></li>",
1102            item.path, item.title
1103          );
1104        }
1105
1106        // Render regular entries with optional numbering
1107        for item in nav_items {
1108          if let Some(num) = item.number {
1109            let _ = writeln!(
1110              doc_nav,
1111              "<li><a href=\"{}\">{num}. {}</a></li>",
1112              item.path, item.title
1113            );
1114          } else {
1115            let _ = writeln!(
1116              doc_nav,
1117              "<li><a href=\"{}\">{}</a></li>",
1118              item.path, item.title
1119            );
1120          }
1121        }
1122      }
1123    }
1124  }
1125
1126  // Add link to options page if module_options is configured
1127  if doc_nav.is_empty() && config.module_options.is_some() {
1128    let _ = writeln!(
1129      doc_nav,
1130      "<li><a href=\"{root_prefix}options.html\">Module Options</a></li>"
1131    );
1132  }
1133
1134  // Add search link only if search is enabled
1135  if config.is_search_enabled() {
1136    let _ = writeln!(
1137      doc_nav,
1138      "<li><a href=\"{root_prefix}search.html\">Search</a></li>"
1139    );
1140  }
1141
1142  doc_nav
1143}

1170/// Generate table of contents from headers
1171fn generate_toc(headers: &[Header]) -> String {
1172  let mut toc = String::new();
1173  let mut current_level = 0;
1174
1175  for header in headers {
1176    // Only include h1, h2, h3 in TOC
1177    if header.level <= 3 {
1178      // Adjust TOC nesting
1179      while current_level < header.level - 1 {
1180        toc.push_str("<ul>");
1181        current_level += 1;
1182      }
1183      while current_level > header.level - 1 {
1184        toc.push_str("</ul>");
1185        current_level -= 1;
1186      }
1187
1188      // Add TOC item
1189      if current_level == header.level - 1 {
1190        toc.push_str("<li>");
1191      } else {
1192        toc.push_str("</li><li>");
1193      }
1194
1195      let mut visible_text = header.text.as_str();
1196      if let Some(idx) = visible_text.rfind("{#")
1197        && visible_text.ends_with('}')
1198      {
1199        visible_text = visible_text[..idx].trim_end();
1200      }
1201      // Writing to String is infallible
1202      let _ = writeln!(
1203        toc,
1204        "<a href=\"#{}\">{}</a>",
1205        header.id,
1206        encode_text(visible_text)
1207      );
1208    }
1209  }
1210
1211  // Close open tags
1212  while current_level > 0 {
1213    toc.push_str("</li></ul>");
1214    current_level -= 1;
1215  }
1216  if !headers.is_empty() && !toc.is_empty() && !toc.ends_with("</li>") {
1217    toc.push_str("</li>");
1218  }
1219
1220  toc
1221}

1223/// Generate the options HTML content
1224fn generate_options_html(options: &HashMap<String, NixOption>) -> String {
1225  let mut options_html = String::with_capacity(options.len() * 500); // FIXME: Rough estimate for capacity
1226
1227  // Sort options by name
1228  let mut option_keys: Vec<_> = options.keys().collect();
1229  option_keys.sort();
1230
1231  for key in option_keys {
1232    let option = &options[key];
1233    let option_id = format!("option-{}", option.name.replace('.', "-"));
1234
1235    // Open option container with ID for direct linking
1236    // Writing to String is infallible
1237    let _ = writeln!(options_html, "<div class=\"option\" id=\"{option_id}\">");
1238
1239    // Option name with anchor link and copy button
1240    let _ = write!(
1241      options_html,
1242      "  <h3 class=\"option-name\">\n    <a href=\"#{}\" \
1243       class=\"option-anchor\">{}</a>\n    <span class=\"copy-link\" \
1244       title=\"Copy link to this option\"></span>\n    <span \
1245       class=\"copy-feedback\">Link copied!</span>\n  </h3>\n",
1246      option_id, option.name
1247    );
1248
1249    // Option metadata (internal/readOnly)
1250    let mut metadata = Vec::new();
1251    if option.internal {
1252      metadata.push("internal");
1253    }
1254    if option.read_only {
1255      metadata.push("read-only");
1256    }
1257
1258    if !metadata.is_empty() {
1259      // Writing to String is infallible
1260      let _ = writeln!(
1261        options_html,
1262        "  <div class=\"option-metadata\">{}</div>",
1263        metadata.join(", ")
1264      );
1265    }
1266
1267    // Option type
1268    let _ = writeln!(
1269      options_html,
1270      "  <div class=\"option-type\">Type: <code>{}</code></div>",
1271      option.type_name
1272    );
1273
1274    // Option description
1275    let _ = writeln!(
1276      options_html,
1277      "  <div class=\"option-description\">{}</div>",
1278      option.description
1279    );
1280
1281    // Add default value if available
1282    add_default_value(&mut options_html, option);
1283
1284    // Add example if available
1285    add_example_value(&mut options_html, option);
1286
1287    // Option declared in - now with hyperlink support
1288    if let Some(declared_in) = &option.declared_in {
1289      // Writing to String is infallible
1290      if let Some(url) = &option.declared_in_url {
1291        let _ = writeln!(
1292          options_html,
1293          "  <div class=\"option-declared\">Declared in: <code><a \
1294           href=\"{url}\" target=\"_blank\">{declared_in}</a></code></div>"
1295        );
1296      } else {
1297        let _ = writeln!(
1298          options_html,
1299          "  <div class=\"option-declared\">Declared in: \
1300           <code>{declared_in}</code></div>"
1301        );
1302      }
1303    }
1304
1305    // Close option div
1306    options_html.push_str("</div>\n");
1307  }
1308
1309  options_html
1310}

1312/// Add default value to options HTML
1313fn add_default_value(html: &mut String, option: &NixOption) {
1314  if let Some(default_text) = &option.default_text {
1315    // Remove surrounding backticks if present (from literalExpression)
1316    let clean_default = if default_text.starts_with('`')
1317      && default_text.ends_with('`')
1318      && default_text.len() > 2
1319    {
1320      &default_text[1..default_text.len() - 1]
1321    } else {
1322      default_text
1323    };
1324
1325    // Writing to String is infallible
1326    let _ = writeln!(
1327      html,
1328      "  <div class=\"option-default\">Default: <code>{}</code></div>",
1329      html_escape::encode_text(clean_default)
1330    );
1331  } else if let Some(default_val) = &option.default {
1332    let _ = writeln!(
1333      html,
1334      "  <div class=\"option-default\">Default: <code>{}</code></div>",
1335      html_escape::encode_text(&default_val.to_string()),
1336    );
1337  }
1338}

1340/// Add example value to options HTML
1341fn add_example_value(html: &mut String, option: &NixOption) {
1342  if let Some(example_text) = &option.example_text {
1343    // Process the example text to preserve code formatting
1344    if example_text.contains('\n') {
1345      // Multi-line examples - preserve formatting with pre/code
1346      // Process special characters to ensure valid HTML
1347      let safe_example = example_text.replace('<', "&lt;").replace('>', "&gt;");
1348
1349      // Remove backticks if they're surrounding the entire content (from
1350      // literalExpression)
1351      let trimmed_example = if safe_example.starts_with('`')
1352        && safe_example.ends_with('`')
1353        && safe_example.len() > 2
1354      {
1355        &safe_example[1..safe_example.len() - 1]
1356      } else {
1357        &safe_example
1358      };
1359
1360      // Writing to String is infallible
1361      let _ = writeln!(
1362        html,
1363        "  <div class=\"option-example\">Example: \
1364         <pre><code>{trimmed_example}</code></pre></div>"
1365      );
1366    } else {
1367      // Check if this is already a code block (surrounded by backticks)
1368      if example_text.starts_with('`')
1369        && example_text.ends_with('`')
1370        && example_text.len() > 2
1371      {
1372        // This is inline code - extract the content and properly escape it
1373        let code_content = &example_text[1..example_text.len() - 1];
1374        let safe_content =
1375          code_content.replace('<', "&lt;").replace('>', "&gt;");
1376        let _ = writeln!(
1377          html,
1378          "  <div class=\"option-example\">Example: \
1379           <code>{safe_content}</code></div>"
1380        );
1381      } else {
1382        // Regular inline example - still needs escaping
1383        let safe_example =
1384          example_text.replace('<', "&lt;").replace('>', "&gt;");
1385        let _ = writeln!(
1386          html,
1387          "  <div class=\"option-example\">Example: \
1388           <code>{safe_example}</code></div>"
1389        );
1390      }
1391    }
1392  } else if let Some(example_val) = &option.example {
1393    let example_str = example_val.to_string();
1394    let safe_example = example_str.replace('<', "&lt;").replace('>', "&gt;");
1395    if example_str.contains('\n') {
1396      // Multi-line JSON examples need special handling
1397      let _ = writeln!(
1398        html,
1399        "  <div class=\"option-example\">Example: \
1400         <pre><code>{safe_example}</code></pre></div>"
1401      );
1402    } else {
1403      // Single-line JSON examples
1404      let _ = writeln!(
1405        html,
1406        "  <div class=\"option-example\">Example: \
1407         <code>{safe_example}</code></div>"
1408      );
1409    }
1410  }
1411}

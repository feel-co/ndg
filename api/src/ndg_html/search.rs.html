79/// Tokenize text into searchable terms
80fn tokenize(text: &str) -> Vec<String> {
81  static WORD_REGEX: OnceLock<Regex> = OnceLock::new();
82  let word_regex = WORD_REGEX.get_or_init(|| {
83    #[allow(
84      clippy::unwrap_used,
85      reason = "regex pattern is statically known to be valid"
86    )]
87    Regex::new(r"\b[a-zA-Z0-9_-]+\b").unwrap()
88  });
89
90  let mut tokens = HashSet::new();
91
92  for capture in word_regex.find_iter(text) {
93    let token = capture.as_str().to_lowercase();
94    if token.len() > 2 {
95      // Skip very short tokens
96      tokens.insert(token);
97    }
98  }
99
100  tokens.into_iter().collect()
101}

103/// Represents a processed markdown document for search indexing.

120/// Generate search index from standalone markdown files.
121///
122/// This function only processes markdown files that will be rendered as
123/// standalone HTML pages. Files that are included in other documents via
124/// `{=include=}` directives should not be passed to this function - their
125/// content is already indexed as part of the parent document.
126///
127/// # Arguments
128///
129/// * `config` - The loaded configuration for documentation generation.
130/// * `processed_docs` - Processed markdown documents with expanded includes.
131///
132/// # Errors
133///
134/// Returns an error if the search directory cannot be created or if any file
135/// cannot be read or written.
136///
137/// # Panics
138///
139/// Panics if `config.input_dir` is `None` when processing markdown files.
140pub fn generate_search_index(
141  config: &Config,
142  processed_docs: &[ProcessedDocument],
143) -> Result<()> {
144  if !config.is_search_enabled() {
145    return Ok(());
146  }
147
148  info!("Generating search index...");
149
150  // Create search directory
151  let search_dir = config.output_dir.join("assets");
152  fs::create_dir_all(&search_dir)?;
153
154  let mut search_index = SearchIndex::new();
155  let mut doc_id = 0;
156  let mut markdown_count = 0;
157
158  // Get max heading level for anchor indexing
159  let max_heading_level = config.search_max_heading_level();
160
161  // Process standalone markdown documents in parallel
162  if !processed_docs.is_empty() {
163    let documents: Result<Vec<_>> = processed_docs
164      .par_iter()
165      .map(|doc| {
166        // Convert HTML to plain text for indexing
167        // This includes all content from expanded includes
168        let plain_text = html::content_to_plaintext(&doc.html_content);
169
170        let title = doc.title.clone().unwrap_or_else(|| {
171          doc
172            .source_path
173            .file_stem()
174            .unwrap_or_default()
175            .to_string_lossy()
176            .to_string()
177        });
178
179        // Use the provided HTML path
180        let path = doc.html_path.clone();
181
182        // Extract anchors from headers, filtering by max level
183        let anchors: Vec<SearchAnchor> = doc
184          .headers
185          .iter()
186          .filter(|h| h.level <= max_heading_level)
187          .map(|h| {
188            SearchAnchor {
189              text:   h.text.clone(),
190              id:     h.id.clone(),
191              level:  h.level,
192              tokens: tokenize(&h.text),
193            }
194          })
195          .collect();
196
197        let tokens = tokenize(&plain_text);
198        let title_tokens = tokenize(&title);
199
200        Ok((title, plain_text, path, tokens, title_tokens, anchors))
201      })
202      .collect();
203
204    let documents = documents?;
205    let documents_count = documents.len();
206    markdown_count = documents_count;
207    for (index, (title, content, path, tokens, title_tokens, anchors)) in
208      documents.into_iter().enumerate()
209    {
210      let current_doc_id = doc_id + index + 1;
211      let doc = SearchDocument {
212        id: current_doc_id.to_string(),
213        title,
214        content,
215        path,
216        tokens,
217        title_tokens,
218        anchors,
219      };
220      search_index.add_document(doc);
221    }
222    doc_id += documents_count + 1;
223  }
224
225  // Process options if available
226  let mut options_count = 0;
227  if let Some(options_path) = &config.module_options
228    && let Ok(options_content) = fs::read_to_string(options_path)
229    && let Ok(options_data) = serde_json::from_str::<Value>(&options_content)
230    && let Some(options_obj) = options_data.as_object()
231  {
232    for (key, option_value) in options_obj {
233      let raw_description = option_value["description"].as_str().unwrap_or("");
234      let plain_description = html::content_to_plaintext(raw_description);
235
236      let title = format!("Option: {}", html_escape::encode_text(key));
237      let tokens = tokenize(&plain_description);
238      let title_tokens = tokenize(&title);
239
240      search_index.add_document(SearchDocument {
241        id: doc_id.to_string(),
242        title,
243        content: plain_description,
244        // XXX: this leads to breakage sometimes, I don't regret it
245        path: format!("options.html#option-{}", key.replace('.', "-")),
246        tokens,
247        title_tokens,
248        // options don't have sub-anchors (or at least we hope they don't)
249        anchors: Vec::new(),
250      });
251
252      doc_id += 1;
253      options_count += 1;
254    }
255  }
256
257  // Write search index as JSON with config
258  let search_config = config
259    .search
260    .as_ref()
261    .map_or_else(ndg_config::search::SearchConfig::default, Clone::clone);
262
263  let search_data = SearchData {
264    documents:       search_index.documents,
265    min_word_length: search_config.min_word_length,
266    stopwords:       search_config.stopwords.clone(),
267    boost_title:     search_config.get_title_boost(),
268    boost_content:   search_config.get_content_boost(),
269    boost_anchor:    search_config.get_anchor_boost(),
270  };
271
272  let search_data_path = search_dir.join("search-data.json");
273  fs::write(&search_data_path, serde_json::to_string(&search_data)?)
274    .wrap_err_with(|| {
275      format!(
276        "Failed to write search data to {}",
277        search_data_path.display()
278      )
279    })?;
280
281  // Create search page
282  create_search_page(config)?;
283
284  let total_count = search_data.documents.len();
285  info!(
286    "Search index generated successfully: {markdown_count} markdown \
287     documents, {options_count} options indexed ({total_count} total)"
288  );
289
290  Ok(())
291}

293/// Create the search page HTML and write it to the output directory.
294///
295/// # Errors
296///
297/// Returns an error if the search page cannot be rendered or written.
298pub fn create_search_page(config: &Config) -> Result<()> {
299  if !config.is_search_enabled() {
300    return Ok(());
301  }
302
303  info!("Creating search page...");
304
305  let mut context = HashMap::new();
306  context.insert("title", format!("{} - Search", config.title));
307
308  let html = template::render_search(config, &context)?;
309
310  // Apply postprocessing if configured
311  let processed_html = if let Some(ref postprocess) = config.postprocess {
312    postprocess::process_html(&html, postprocess)?
313  } else {
314    html
315  };
316
317  let search_page_path = config.output_dir.join("search.html");
318  fs::write(&search_page_path, &processed_html).wrap_err_with(|| {
319    format!(
320      "Failed to write search page to {}",
321      search_page_path.display()
322    )
323  })?;
324
325  Ok(())

20/// TOML frontmatter parsed from the `+++` delimited block at the start of a
21/// markdown document.
22///
23/// Frontmatter must appear at the very beginning of the file, wrapped in
24/// `+++` delimiters on their own lines. Any unrecognised keys are silently
25/// ignored by the TOML deserialiser.
26#[derive(Debug, Default, Clone, Deserialize)]
27pub struct PageFrontmatter {
28  /// Overrides the page title extracted from the first H1 heading.
29  pub title:       Option<String>,
30  /// Per-page meta description. Injected as `<meta name="description">` and
31  /// as `og:description` where OG tags are enabled.
32  pub description: Option<String>,
33
34  /// Per-page author. Injected as `<meta name="author">`.
35  pub author: Option<String>,
36
37  /// Name of the template file to use for this page (e.g. `"custom"` or
38  /// `"custom.html"`). Falls back to the default template resolution if the
39  /// named file cannot be found.
40  pub template: Option<String>,
41
42  /// When `false`, suppresses the table of contents for this page.
43  pub toc: Option<bool>,
44
45  /// Arbitrary user-defined data. Accessible in templates as
46  /// `{{ page.extra.key }}`.
47  pub extra: Option<toml::Table>,
48}

50/// Strips TOML frontmatter from the start of a markdown document.
51///
52/// Frontmatter must begin at the very first byte of the file, delimited by
53/// `+++` on its own line. The closing `+++` must also be on its own line.
54/// Returns the parsed [`PageFrontmatter`] (if present and valid) and the
55/// remaining markdown content with the frontmatter block removed.
56///
57/// Malformed TOML inside a valid `+++` block is logged as a warning and
58/// treated as absent; the full original content is returned unchanged.
59fn strip_frontmatter(content: &str) -> (Option<PageFrontmatter>, String) {
60  const DELIM: &str = "+++";
61
62  // Normalize CRLF to LF so that Windows-style line endings are handled.
63  let lf_content;
64  let content: &str = if content.contains("\r\n") {
65    lf_content = content.replace("\r\n", "\n");
66    &lf_content
67  } else {
68    content
69  };
70
71  // The opening delimiter must be the very first thing in the file.
72  let Some(after_open) = content.strip_prefix(DELIM) else {
73    return (None, content.to_owned());
74  };
75  // Must be immediately followed by a newline.
76  let Some(after_open) = after_open.strip_prefix('\n') else {
77    return (None, content.to_owned());
78  };
79
80  // Find the closing delimiter.
81  let needle = format!("\n{DELIM}");
82  let Some(end_idx) = after_open.find(needle.as_str()) else {
83    return (None, content.to_owned());
84  };
85
86  let toml_str = &after_open[..end_idx];
87  let after_close = &after_open[end_idx + needle.len()..];
88  // Strip the single newline that follows the closing delimiter, if present.
89  let remaining = after_close.strip_prefix('\n').unwrap_or(after_close);
90
91  match toml::from_str::<PageFrontmatter>(toml_str) {
92    Ok(fm) => (Some(fm), remaining.to_owned()),
93    Err(e) => {
94      warn!("Failed to parse frontmatter: {e}");
95      (None, content.to_owned())
96    },
97  }
98}

111/// Processes all markdown files in the input directory and returns processed
112/// data.
113///
114/// This function renders all markdown files, handles custom output paths,
115/// and returns structured data for each file. Files that are included in other
116/// files are marked but still returned (caller decides whether to skip them).
117///
118/// As a side effect, this function populates `config.included_files` with the
119/// mapping of included files to their parent files.
120///
121/// # Arguments
122///
123/// * `config` - The loaded configuration for documentation generation.
124/// * `processor` - Optional pre-created processor to reuse. If None, creates a
125///   new one.
126///
127/// # Returns
128///
129/// A vector of `ProcessedMarkdown` entries.
130///
131/// # Errors
132///
133/// Returns an error if any file cannot be read or rendered.
134///
135/// # Panics
136///
137/// Panics if `config.input_dir` is `None` when processing markdown files.
138pub fn process_markdown_files(
139  config: &mut Config,
140  processor: Option<&MarkdownProcessor>,
141) -> Result<Vec<ProcessedMarkdown>> {
142  if let Some(ref input_dir) = config.input_dir {
143    info!("Input directory: {}", input_dir.display());
144    let files = collect_markdown_files(input_dir);
145    info!("Found {} markdown files", files.len());
146
147    // Map: output html path -> output entry
148    #[allow(clippy::items_after_statements, reason = "Local type alias")]
149    type OutputEntry = (
150      String,
151      Vec<Header>,
152      Option<String>,
153      PathBuf,
154      bool,
155      Option<PageFrontmatter>,
156    );
157    let mut output_map: HashMap<String, OutputEntry> = HashMap::new();
158
159    // Track custom outputs keyed by included file path
160    let mut pending_custom_outputs: HashMap<PathBuf, Vec<String>> =
161      HashMap::new();
162
163    // Track all included files across all markdown files
164    let mut all_included_files: HashMap<PathBuf, PathBuf> = HashMap::new();
165
166    // Use provided processor or create a new one
167    let base_processor = processor
168      .map_or_else(|| create_processor(config, None), std::clone::Clone::clone);
169
170    // Store render results to avoid rendering twice
171    #[allow(clippy::items_after_statements, reason = "Local type alias")]
172    type RenderCache = HashMap<PathBuf, ndg_commonmark::MarkdownResult>;
173    let mut render_cache: RenderCache = HashMap::new();
174
175    // Store parsed frontmatter per source file
176    let mut frontmatter_cache: HashMap<PathBuf, Option<PageFrontmatter>> =
177      HashMap::new();
178
179    // First pass: render all files once and collect metadata
180    let progress = ProgressBar::new(files.len() as u64);
181    #[allow(
182      clippy::expect_used,
183      reason = "template string is a compile-time constant; failure is \
184                impossible"
185    )]
186    let style = ProgressStyle::default_bar()
187      .template(
188        "{spinner:.green} [{elapsed_precise}] [{bar:40.cyan/blue}] \
189         {pos}/{len} ({eta}) {msg}",
190      )
191      .expect("valid progress bar template string")
192      .progress_chars("#>-");
193    progress.set_style(style);
194    progress.set_message("Processing markdown");
195    progress.enable_steady_tick(std::time::Duration::from_millis(100));
196
197    for file_path in &files {
198      progress.inc(1);
199      let raw_content = fs::read_to_string(file_path).wrap_err_with(|| {
200        format!("Failed to read markdown file: {}", file_path.display())
201      })?;
202
203      let (frontmatter, content) = strip_frontmatter(&raw_content);
204      frontmatter_cache.insert(file_path.clone(), frontmatter);
205
206      let base_dir = file_path.parent().unwrap_or_else(|| {
207        config
208          .input_dir
209          .as_deref()
210          .unwrap_or_else(|| Path::new("."))
211      });
212      let processor = base_processor.clone().with_base_dir(base_dir);
213
214      let result = processor.render(&content);
215
216      // Track custom outputs for included files
217      for inc in &result.included_files {
218        if let Some(ref custom_output) = inc.custom_output {
219          let inc_path = base_dir.join(&inc.path);
220          if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
221            pending_custom_outputs
222              .entry(inc_rel.to_path_buf())
223              .or_default()
224              .push(custom_output.clone());
225          }
226        }
227      }
228
229      // Collect all included files for this markdown file
230      for inc in &result.included_files {
231        let inc_path = base_dir.join(&inc.path);
232        if let Ok(inc_rel) = inc_path.strip_prefix(input_dir) {
233          let Ok(includer_rel) = file_path.strip_prefix(input_dir) else {
234            continue;
235          };
236
237          all_included_files
238            .entry(inc_rel.to_path_buf())
239            .and_modify(|old| {
240              // For deterministic output
241              if includer_rel < old.as_path() {
242                *old = {
243                  let this = &includer_rel;
244                  this.to_path_buf()
245                };
246              }
247            })
248            .or_insert_with(|| includer_rel.to_owned());
249        }
250      }
251
252      // Cache the render result
253      render_cache.insert(file_path.clone(), result);
254    }
255
256    progress.finish_with_message("Markdown processing complete");
257
258    // Second pass: use cached results to build output map
259    for file_path in &files {
260      // Retrieve cached result instead of rendering again
261      #[allow(clippy::expect_used, reason = "File is guaranteed to be cached")]
262      let result = render_cache
263        .get(file_path)
264        .expect("File should have cached result");
265
266      let rel_path = file_path.strip_prefix(input_dir).wrap_err_with(|| {
267        format!(
268          "Failed to determine relative path for {}",
269          file_path.display()
270        )
271      })?;
272      let mut output_path = rel_path.to_path_buf();
273      output_path.set_extension("html");
274      let output_path_str = output_path.to_string_lossy().to_string();
275
276      // Check if this file is included in another file
277      let is_included = all_included_files.contains_key(rel_path);
278
279      // Get any pending custom outputs for this file
280      let custom_outputs =
281        pending_custom_outputs.remove(rel_path).unwrap_or_default();
282
283      let frontmatter = frontmatter_cache.remove(file_path).flatten();
284
285      // If this file has custom outputs via html:into-file, write to those
286      // locations
287      if custom_outputs.is_empty() {
288        output_map.insert(
289          output_path_str,
290          (
291            result.html.clone(),
292            result.headers.clone(),
293            result.title.clone(),
294            file_path.clone(),
295            is_included,
296            frontmatter,
297          ),
298        );
299      } else {
300        for custom in custom_outputs {
301          output_map.insert(
302            custom.clone(),
303            (
304              result.html.clone(),
305              result.headers.clone(),
306              result.title.clone(),
307              file_path.clone(),
308              false,
309              frontmatter.clone(),
310            ),
311          );
312        }
313      }
314    }
315
316    // Populate config.included_files so caller can use it
317    config.included_files.clone_from(&all_included_files);
318
319    // Convert output_map to Vec<ProcessedMarkdown>
320    let mut results: Vec<ProcessedMarkdown> = output_map
321      .into_iter()
322      .map(
323        |(
324          output_path,
325          (html_content, headers, title, source_path, is_included, frontmatter),
326        )| {
327          ProcessedMarkdown {
328            html_content,
329            headers,
330            title,
331            source_path,
332            output_path,
333            is_included,
334            frontmatter,
335          }
336        },
337      )
338      .collect();
339
340    // Sort for deterministic output
341    results.sort_by(|a, b| a.output_path.cmp(&b.output_path));
342
343    Ok(results)
344  } else {
345    info!("No input directory provided, skipping markdown processing");
346    Ok(Vec::new())
347  }
348}

350/// Creates a markdown processor from the NDG configuration.
351///
352/// # Arguments
353/// * `config` - The loaded configuration for documentation generation.
354/// * `valid_options` - Optional set of valid option names for validation.
355#[must_use]
356#[allow(
357  clippy::implicit_hasher,
358  reason = "Standard HashSet sufficient for this use case"
359)]
360pub fn create_processor(
361  config: &Config,
362  valid_options: Option<HashSet<String>>,
363) -> MarkdownProcessor {
364  let tab_style = match config.tab_style.as_str() {
365    "warn" => TabStyle::Warn,
366    "normalize" => TabStyle::Normalize,
367    _ => TabStyle::None,
368  };
369
370  let mut builder = MarkdownOptionsBuilder::new()
371    .gfm(true)
372    .highlight_code(config.highlight_code)
373    .tab_style(tab_style);
374
375  if let Some(mappings_path) = &config.manpage_urls_path {
376    builder = builder
377      .manpage_urls_path(Some(mappings_path.to_string_lossy().to_string()));
378  }
379
380  if let Some(options) = valid_options {
381    builder = builder.valid_options(Some(options));
382  }
383
384  MarkdownProcessor::new(builder.build())
385}

387/// Extracts the page title from a markdown file.
388///
389/// This attempts to extract the first heading as the title, falling back to the
390/// file name if no heading is found or the file cannot be read.
391///
392/// # Arguments
393///
394/// * `file_path` - The path to the markdown file.
395/// * `html_path` - The corresponding HTML output path (used for fallback).
396///
397/// # Returns
398///
399/// The extracted title or the file stem as a fallback.
400#[must_use]
401pub fn extract_page_title(file_path: &Path, html_path: &Path) -> String {
402  let default_title = html_path
403    .file_stem()
404    .unwrap_or_default()
405    .to_string_lossy()
406    .to_string();
407
408  match fs::read_to_string(file_path) {
409    Ok(content) => {
410      ndg_commonmark::utils::extract_markdown_title_and_id(&content)
411        .map_or(default_title, |(title, _)| title)
412    },
413    Err(e) => {
414      log::warn!(
415        "Failed to read {} for title extraction: {e}",
416        file_path.display()
417      );
418      default_title
419    },
420  }

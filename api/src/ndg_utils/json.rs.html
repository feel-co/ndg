1use serde_json::Value;
2
3/// Extract string value from JSON structures including special types like
4/// `literalExpression`
5///
6/// Handles both structured values such as `literalExpression`,
7/// `literalDocBook`, `literalMD`) and simple scalar values (strings, numbers,
8/// booleans and null).
9///
10/// # Arguments
11///
12/// * `value` - The JSON value to extract
13/// * `wrap_code` - Whether to wrap `literalExpression` values in backticks
14///
15/// # Returns
16///
17/// Extracted string value, or `None` if not extractable
18#[must_use]
19#[inline]
20pub fn extract_value(value: &Value, wrap_code: bool) -> Option<String> {
21  if let Value::Object(obj) = value
22    && let Some(Value::String(type_name)) = obj.get("_type")
23  {
24    match type_name.as_str() {
25      // literalDocBook and literalMD are deprecated as of 24.11
26      // and supported only for backwards compatibility
27      "literalExpression" => {
28        if let Some(Value::String(text)) = obj.get("text") {
29          if wrap_code {
30            return Some(format!("`{}`", text.clone()));
31          }
32          return Some(text.clone());
33        }
34      },
35      "literalMD" => {
36        if let Some(Value::String(text)) = obj.get("text") {
37          // literalMD contains markdown that should be rendered, not wrapped in
38          // code
39          return Some(text.clone());
40        }
41      },
42      "literalDocBook" => {
43        if let Some(Value::String(text)) = obj.get("text") {
44          return Some(text.clone());
45        }
46      },
47      _ => {},
48    }
49  }
50
51  // For simple scalar values, convert to string
52  match value {
53    Value::String(s) => Some(s.clone()),
54    Value::Number(n) => Some(n.to_string()),
55    Value::Bool(b) => Some(b.to_string()),
56    Value::Null => Some("null".to_string()),
57    _ => None,
58  }
59}

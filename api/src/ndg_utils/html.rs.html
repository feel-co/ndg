5/// Calculate the relative path prefix needed to reach the root from a given
6/// file path For example: "docs/subdir/file.html" would return "../"
7///              "docs/subdir/nested/file.html" would return "../../"
8#[must_use]
9pub fn calculate_root_relative_path(file_rel_path: &Path) -> String {
10  let depth = file_rel_path.components().count();
11  if depth <= 1 {
12    String::new() // file is at root level
13  } else {
14    "../".repeat(depth - 1)
15  }
16}

18/// Generate proper asset paths for templates based on file location
19#[must_use]
20pub fn generate_asset_paths(
21  file_rel_path: &Path,
22) -> HashMap<&'static str, String> {
23  let root_prefix = calculate_root_relative_path(file_rel_path);
24
25  let mut paths = HashMap::new();
26  paths.insert("stylesheet_path", format!("{root_prefix}assets/style.css"));
27  paths.insert("main_js_path", format!("{root_prefix}assets/main.js"));
28  paths.insert("search_js_path", format!("{root_prefix}assets/search.js"));
29
30  // Navigation paths
31  paths.insert("index_path", format!("{root_prefix}index.html"));
32  paths.insert("options_path", format!("{root_prefix}options.html"));
33  paths.insert("search_path", format!("{root_prefix}search.html"));
34
35  paths
36}

38/// Process content through the markdown pipeline and extract plain text.
39///
40/// Converts markdown to plain text while preserving document structure:
41///
42/// - Headings and paragraphs are separated by newlines
43/// - Multiple consecutive blank lines are collapsed to single blank lines
44/// - Whitespace within lines is normalized to single spaces
45/// - Anchor markers like {#id} are removed
46#[must_use]
47pub fn content_to_plaintext(content: &str) -> String {
48  let plain_text = strip_markdown(content);
49
50  // Normalize whitespace: collapse multiple blank lines and trim lines
51  let lines: Vec<String> = plain_text
52    .lines()
53    .map(|line| {
54      // Remove anchor markers {#id} and collapse whitespace
55      line
56        .split_whitespace()
57        .filter(|word| !(word.starts_with("{#") && word.ends_with('}')))
58        .collect::<Vec<_>>()
59        .join(" ")
60    })
61    .collect();
62
63  // Remove consecutive empty lines
64  let mut result = Vec::new();
65  let mut prev_empty = false;
66
67  for line in lines {
68    if line.is_empty() {
69      if !prev_empty && !result.is_empty() {
70        result.push(line);
71      }
72      prev_empty = true;
73    } else {
74      result.push(line);
75      prev_empty = false;
76    }
77  }
78
79  result.join("\n").trim().to_string()
80}

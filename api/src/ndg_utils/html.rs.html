6/// Calculate the relative path prefix needed to reach the root from a given
7/// file path For example: "docs/subdir/file.html" would return "../"
8///              "docs/subdir/nested/file.html" would return "../../"
9#[must_use]
10pub fn calculate_root_relative_path(file_rel_path: &Path) -> String {
11  let depth = file_rel_path.components().count();
12  if depth <= 1 {
13    String::new() // file is at root level
14  } else {
15    "../".repeat(depth - 1)
16  }
17}

19/// Generate proper asset paths for templates based on file location
20#[must_use]
21pub fn generate_asset_paths(
22  file_rel_path: &Path,
23) -> FxHashMap<&'static str, String> {
24  let root_prefix = calculate_root_relative_path(file_rel_path);
25
26  let mut paths = FxHashMap::default();
27  paths.reserve(6);
28  paths.insert("stylesheet_path", format!("{root_prefix}assets/style.css"));
29  paths.insert("main_js_path", format!("{root_prefix}assets/main.js"));
30  paths.insert("search_js_path", format!("{root_prefix}assets/search.js"));
31
32  // Navigation paths
33  paths.insert("index_path", format!("{root_prefix}index.html"));
34  paths.insert("options_path", format!("{root_prefix}options.html"));
35  paths.insert("search_path", format!("{root_prefix}search.html"));
36
37  paths
38}

40/// Process content through the markdown pipeline and extract plain text.
41///
42/// Converts markdown to plain text while preserving document structure:
43///
44/// - Headings and paragraphs are separated by newlines
45/// - Multiple consecutive blank lines are collapsed to single blank lines
46/// - Whitespace within lines is normalized to single spaces
47/// - Anchor markers like {#id} are removed
48#[must_use]
49pub fn content_to_plaintext(content: &str) -> String {
50  let plain_text = strip_markdown(content);
51
52  let mut result = String::with_capacity(plain_text.len());
53  let mut line = String::new();
54  let mut prev_empty = false;
55  let mut has_output = false;
56
57  for raw_line in plain_text.lines() {
58    line.clear();
59
60    // Remove anchor markers {#id} and collapse whitespace.
61    for word in raw_line
62      .split_whitespace()
63      .filter(|word| !(word.starts_with("{#") && word.ends_with('}')))
64    {
65      if !line.is_empty() {
66        line.push(' ');
67      }
68      line.push_str(word);
69    }
70
71    if line.is_empty() {
72      if !prev_empty && has_output {
73        result.push('\n');
74      }
75      prev_empty = true;
76    } else {
77      if has_output {
78        result.push('\n');
79      }
80      result.push_str(&line);
81      has_output = true;
82      prev_empty = false;
83    }
84  }
85
86  result.trim().to_string()
87}

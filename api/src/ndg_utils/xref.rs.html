1//! Cross-page anchor registry and link-rewrite utilities.
2use ndg_commonmark::rewrite_cross_page_anchor_links;
3use rustc_hash::FxHashMap;
4
5use crate::markdown::ProcessedMarkdown;
6
7/// A resolved anchor target, i.e., the output page that owns it and the heading
8/// title text.
9#[derive(Debug, Clone)]
10pub struct AnchorEntry {
11  pub page:  String,
12  pub title: String,
13}
14
15/// Registry mapping anchor IDs to their owning output page and heading title.
16pub type AnchorRegistry = FxHashMap<String, AnchorEntry>;
17
18/// Build an anchor registry from all rendered pages.
19///
20/// Iterates every [`ProcessedMarkdown`] item and registers each `Header.id`
21/// with its `output_path` and `Header.text` as the title. Duplicate IDs are
22/// registered with the first occurrence winning.
23#[must_use]
24pub fn build_anchor_registry(pages: &[ProcessedMarkdown]) -> AnchorRegistry {
25  let mut registry = AnchorRegistry::default();
26  for page in pages {
27    for header in &page.headers {
28      if header.id.is_empty() {
29        continue;
30      }
31      if registry.contains_key(&header.id) {
32        log::debug!(
33          "Duplicate anchor ID '{}' on '{}' (already registered from another \
34           page)",
35          header.id,
36          page.output_path
37        );
38        continue;
39      }
40      registry.insert(header.id.clone(), AnchorEntry {
41        page:  page.output_path.clone(),
42        title: header.text.clone(),
43      });
44    }
45  }
46  registry
47}
48
49/// Apply cross-page link rewrites to all non-included pages in `pages`.
50///
51/// Converts the registry to the flat form expected by
52/// [`rewrite_cross_page_anchor_links`] and mutates each page's `html_content`
53/// in place. Included pages are skipped, and their content is embedded within
54/// their parent page's output.
55pub fn apply_cross_page_link_rewrites(
56  pages: &mut [ProcessedMarkdown],
57  registry: &AnchorRegistry,
58) {
59  if registry.is_empty() {
60    return;
61  }
62
63  let flat: FxHashMap<String, (String, String)> = registry
64    .iter()
65    .map(|(id, entry)| (id.clone(), (entry.page.clone(), entry.title.clone())))
66    .collect();
67
68  for page in pages.iter_mut() {
69    if page.is_included {
70      continue;
71    }
72    page.html_content = rewrite_cross_page_anchor_links(
73      &page.html_content,
74      &page.output_path,
75      &flat,
76    );
77  }
78}

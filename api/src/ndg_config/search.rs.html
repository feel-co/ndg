1use serde::{Deserialize, Serialize};
2
3/// Configuration for search functionality
4#[derive(Debug, Clone, Serialize, Deserialize)]
5#[serde(default)]
6pub struct SearchConfig {
7  /// Whether search functionality is enabled
8  pub enable: bool,
9
10  /// Maximum heading level to index (1-6)
11  ///
12  /// Controls which heading levels are included in the search index:
13  /// - 1: Only H1 headings
14  /// - 3: H1, H2, H3 headings (default)
15  /// - 6: All headings H1-H6
16  pub max_heading_level: u8,
17
18  /// Minimum word length for indexing
19  ///
20  /// Words shorter than this length will be excluded from the search index.
21  /// Default is 2, meaning single-character words are ignored.
22  pub min_word_length: usize,
23
24  /// List of stopwords to exclude from indexing
25  ///
26  /// Common words that should be ignored during indexing.
27  /// Empty by default - users can add their own stopwords.
28  pub stopwords: Vec<String>,
29
30  /// Score multiplier for title matches
31  ///
32  /// If not set, falls back to `boost` value. Default is 100.0 for fuzzy
33  /// matches and 20.0 for exact matches.
34  pub boost_title: Option<f32>,
35
36  /// Score multiplier for content matches
37  ///
38  /// If not set, falls back to `boost` value. Default is 30.0 for fuzzy
39  /// matches and 2.0 for partial matches.
40  pub boost_content: Option<f32>,
41
42  /// Score multiplier for anchor/heading matches
43  ///
44  /// If not set, falls back to `boost` value. This affects matching section
45  /// headings.
46  pub boost_anchor: Option<f32>,
47
48  /// Global score multiplier
49  ///
50  /// Sets all boost values at once. Individual boost_* settings override
51  /// this for their respective categories.
52  pub boost: Option<f32>,
53}
54
55impl Default for SearchConfig {
56  fn default() -> Self {
57    Self {
58      enable:            true,
59      max_heading_level: 3,
60      min_word_length:   2,
61      stopwords:         Vec::new(),
62      boost_title:       None,
63      boost_content:     None,
64      boost_anchor:      None,
65      boost:             None,
66    }
67  }
68}
69
70impl SearchConfig {
71  /// Get the effective title boost value
72  #[must_use]
73  pub fn get_title_boost(&self) -> f32 {
74    self.boost_title.or(self.boost).unwrap_or(100.0)
75  }
76
77  /// Get the effective content boost value
78  #[must_use]
79  pub fn get_content_boost(&self) -> f32 {
80    self.boost_content.or(self.boost).unwrap_or(30.0)
81  }
82
83  /// Get the effective anchor boost value
84  #[must_use]
85  pub fn get_anchor_boost(&self) -> f32 {
86    self.boost_anchor.or(self.boost).unwrap_or(10.0)
87  }
88}

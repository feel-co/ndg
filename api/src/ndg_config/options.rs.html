17/// Filters applied to module options before rendering options documentation.
18#[derive(Debug, Clone, Serialize, Deserialize, Configurable)]
19#[serde(default)]
20pub struct FilterConfig {
21  /// Include only options whose names start with this prefix.
22  #[config(key = "prefix", allow_empty)]
23  pub prefix: Option<String>,
24
25  /// Include only options whose type contains this text, case-insensitively.
26  #[serde(rename = "type")]
27  #[config(key = "type", allow_empty)]
28  pub type_name: Option<String>,
29
30  /// Include only options whose names or descriptions contain this text.
31  #[config(key = "search", allow_empty)]
32  pub search: Option<String>,
33
34  /// Include only options that define a default value.
35  #[config(key = "has_default")]
36  pub has_default: bool,
37
38  /// Include only options that have a non-empty description.
39  #[config(key = "has_description")]
40  pub has_description: bool,
41
42  /// Whether internal/hidden options should be included.
43  #[serde(default = "default_true")]
44  #[config(key = "include_internal")]
45  pub include_internal: bool,
46}

48impl Default for FilterConfig {
49  fn default() -> Self {
50    Self {

58  }

61impl FilterConfig {
62  /// Check whether an option should be included after applying all filters.
63  #[must_use]
64  pub fn matches(
65    &self,
66    name: &str,
67    type_name: &str,
68    description: &str,
69    has_default: bool,
70    has_description: bool,
71    internal: bool,
72  ) -> bool {
73    if !self.include_internal && internal {
74      return false;
75    }
76
77    if let Some(prefix) = self.prefix.as_deref()
78      && !prefix.is_empty()
79      && !name.starts_with(prefix)
80    {
81      return false;
82    }
83
84    if let Some(type_filter) = self.type_name.as_deref()
85      && !type_filter.is_empty()
86      && !contains_ignore_ascii_case(type_name, type_filter)
87    {
88      return false;
89    }
90
91    if let Some(search) = self.search.as_deref()
92      && !search.is_empty()
93      && !contains_ignore_ascii_case(name, search)
94      && !contains_ignore_ascii_case(description, search)
95    {
96      return false;
97    }
98
99    if self.has_default && !has_default {
100      return false;
101    }
102
103    if self.has_description && !has_description {
104      return false;
105    }
106
107    true
108  }
109}
110
111fn contains_ignore_ascii_case(haystack: &str, needle: &str) -> bool {
112  haystack
113    .to_ascii_lowercase()
114    .contains(&needle.to_ascii_lowercase())
115}

133  #[test]
134  fn combines_filters() {
135    let filter = FilterConfig {
136      prefix:           Some("services.nginx".to_string()),
137      type_name:        Some("bool".to_string()),
138      search:           Some("enable".to_string()),
139      has_default:      true,
140      has_description:  true,
141      include_internal: false,
142    };
143
144    assert!(filter.matches(
145      "services.nginx.enable",
146      "boolean",
147      "Whether to enable nginx",
148      true,
149      true,
150      false,
151    ));
152    assert!(!filter.matches(
153      "services.httpd.enable",
154      "boolean",
155      "Whether to enable httpd",
156      true,
157      true,
158      false,
159    ));
160    assert!(!filter.matches(
161      "services.nginx.internal",
162      "boolean",
163      "Internal option",
164      true,
165      true,
166      true,
167    ));
